20  package edu.umd.cs.findbugs.gui2;
21  
22  import com.apple.eawt.ApplicationAdapter;
23  import com.apple.eawt.ApplicationEvent;
24  
25  /*
26   * Based on sample code from Apple.
27   *
28   * This is the only class that uses the Apple specific EAWT classes.
29   * This class should only ever be referenced via reflection after
30   * checking that we are running on Mac OS X.
31   */
32  public class OSXAdapter extends ApplicationAdapter {
33  
34      // pseudo-singleton model; no point in making multiple instances
35      // of the EAWT application or our adapter
36      private static OSXAdapter theAdapter = new OSXAdapter();
37  
38      private static final com.apple.eawt.Application theApplication = new com.apple.eawt.Application();
39  
40      // reference to the app where the existing quit, about, prefs code is
41      private static MainFrame mainApp;
42  
43      private OSXAdapter() {
44      }
45  
46      // implemented handler methods. These are basically hooks into
47      // existing functionality from the main app, as if it came
48      // over from another platform.
49  
50      @Override
51      public void handleAbout(ApplicationEvent ae) {
52          if (mainApp != null) {
53              ae.setHandled(true);
54              // We need to invoke modal About Dialog asynchronously
55              // otherwise the Application queue is locked for the duration
56              // of the about Dialog, which results in a deadlock if a URL is
57              // selected, and we get a ReOpenApplication event when user
58              // switches back to Findbugs.
59              javax.swing.SwingUtilities.invokeLater(() -> mainApp.about());
60          } else {
61              throw new IllegalStateException("handleAbout: " + "MyApp instance detached from listener");
62          }
63      }
64  
65      @Override
66      public void handlePreferences(ApplicationEvent ae) {
67          if (mainApp != null) {
68              mainApp.preferences();
69              ae.setHandled(true);
70          } else {
71              throw new IllegalStateException("handlePreferences: MyApp instance " + "detached from listener");
72          }
73      }
74  
75      @Override
76      public void handleQuit(ApplicationEvent ae) {
77          if (mainApp != null) {
78  
79              /*
80               * You MUST setHandled(false) if you want to delay or cancel the
81               * quit. This is important for cross-platform development -- have a
82               * universal quit routine that chooses whether or not to quit, so
83               * the functionality is identical on all platforms. This example
84               * simply cancels the AppleEvent-based quit and defers to that
85               * universal method.
86               */
87  
88              ae.setHandled(false);
89              mainApp.callOnClose();
90          } else {
91              throw new IllegalStateException("handleQuit: MyApp instance detached " + "from listener");
92          }
93      }
94  
95      // The main entry-point for this functionality. This is the only method
96      // that needs to be called at runtime, and it can easily be done using
97      // reflection (see MyApp.java)
98      public static void registerMacOSXApplication(MainFrame inApp) {
99          if (mainApp != null) {
100             throw new IllegalStateException("application already set");
101         }
102 
103         mainApp = inApp;
104 
105         theApplication.addApplicationListener(theAdapter);
106 
107         theApplication.addPreferencesMenuItem();
108     }
109 
110     // Another static entry point for EAWT functionality. Enables the
111     // "Preferences..." menu item in the application menu.
112     public static void enablePrefs(boolean enabled) {
113 
114         theApplication.setEnabledPreferencesMenu(enabled);
115     }
116 }
