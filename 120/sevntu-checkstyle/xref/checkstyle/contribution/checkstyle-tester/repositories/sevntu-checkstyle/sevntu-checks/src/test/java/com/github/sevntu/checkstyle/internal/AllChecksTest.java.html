47      @Test
48      public void testDefaultTokensAreSubsetOfAcceptableTokens() throws Exception {
49          for (Class<?> check : CheckUtil.getCheckstyleChecks()) {
50              if (AbstractCheck.class.isAssignableFrom(check)) {
51                  final AbstractCheck testedCheck = (AbstractCheck) check.getDeclaredConstructor()
52                          .newInstance();
53                  final int[] defaultTokens = testedCheck.getDefaultTokens();
54                  final int[] acceptableTokens = testedCheck.getAcceptableTokens();
55  
56                  if (!isSubset(defaultTokens, acceptableTokens)) {
57                      final String errorMessage = String.format(Locale.ROOT,
58                              "%s's default tokens must be a subset"
59                              + " of acceptable tokens.", check.getName());
60                      Assert.fail(errorMessage);
61                  }
62              }
63          }
64      }

138     @Test
139     public void testAllCheckstyleMessages() throws Exception {
140         final Map<String, List<String>> usedMessages = new TreeMap<>();
141 
142         // test validity of messages from checks
143         for (Class<?> module : CheckUtil.getCheckstyleModules()) {
144             for (Field message : CheckUtil.getCheckMessages(module)) {
145                 Assert.assertEquals(module.getSimpleName() + "." + message.getName()
146                         + " should be 'public static final'", Modifier.PUBLIC | Modifier.STATIC
147                         | Modifier.FINAL, message.getModifiers());
148 
149                 // below is required for package/private classes
150                 if (!message.isAccessible()) {
151                     message.setAccessible(true);
152                 }
153 
154                 verifyCheckstyleMessage(usedMessages, module, message);
155             }
156         }
157 
158         // test properties for messages not used by checks
159         for (Entry<String, List<String>> entry : usedMessages.entrySet()) {
160             final Properties pr = new Properties();
161             pr.load(AllChecksTest.class.getResourceAsStream(
162                     "/" + entry.getKey().replace('.', '/') + "/messages.properties"));
163 
164             for (Object key : pr.keySet()) {
165                 Assert.assertTrue("property '" + key + "' isn't used by any check in package '"
166                         + entry.getKey() + "'", entry.getValue().contains(key.toString()));
167             }
168         }
169     }

171     @Test
172     public void testAllInputsHaveTest() throws Exception {
173         final Map<String, List<String>> allTests = new HashMap<>();
174 
175         Files.walk(Paths.get("src/test/java/com/github/sevntu/checkstyle"))
176             .forEach(filePath -> {
177                 grabAllTests(allTests, filePath.toFile());
178             });
179         Files.walk(Paths.get("src/test/resources/com/github/sevntu/checkstyle"))
180             .forEach(filePath -> {
181                 verifyInputFile(allTests, filePath.toFile());
182             });
183         Files.walk(Paths.get("src/test/resources-noncompilable/com/github/sevntu/checkstyle"))
184             .forEach(filePath -> {
185                 verifyInputFile(allTests, filePath.toFile());
186             });
187     }

189     private static void verifyCheckstyleMessage(Map<String, List<String>> usedMessages,
190             Class<?> module, Field message) throws Exception {
191         final String messageString = message.get(null).toString();
192         final String packageName = module.getPackage().getName();
193         List<String> packageMessages = usedMessages.get(packageName);
194 
195         if (packageMessages == null) {
196             packageMessages = new ArrayList<>();
197             usedMessages.put(packageName, packageMessages);
198         }
199 
200         packageMessages.add(messageString);
201 
202         String result = null;
203 
204         try {
205             result = CheckUtil.getCheckMessage(module, messageString);
206         }
207         catch (IllegalArgumentException ex) {
208             Assert.fail(module.getSimpleName() + " with the message '" + messageString
209                     + "' failed with: "
210                     + ex.getClass().getSimpleName() + " - " + ex.getMessage());
211         }
212 
213         Assert.assertNotNull(
214                 module.getSimpleName() + " should have text for the message '"
215                         + messageString + "'",
216                 result);
217         Assert.assertFalse(
218                 module.getSimpleName() + " should have non-empty text for the message '"
219                         + messageString + "'",
220                 result.trim().isEmpty());
221         Assert.assertFalse(
222                 module.getSimpleName() + " should have non-TODO text for the message '"
223                         + messageString + "'",
224                 result.trim().startsWith("TODO"));
225     }

227     private static void grabAllTests(Map<String, List<String>> allTests, File file) {
228         if (file.isFile() && file.getName().endsWith("Test.java")) {
229             String path = null;
230 
231             try {
232                 path = getSimplePath(file.getCanonicalPath()).replace("Test.java", "");
233             }
234             catch (IOException ex) {
235                 throw new IllegalStateException(ex);
236             }
237 
238             final int slash = path.lastIndexOf(File.separatorChar);
239             final String pkg = path.substring(0, slash);
240 
241             List<String> classes = allTests.get(pkg);
242 
243             if (classes == null) {
244                 classes = new ArrayList<>();
245 
246                 allTests.put(pkg, classes);
247             }
248 
249             classes.add(path.substring(slash + 1));
250         }
251     }

253     private static void verifyInputFile(Map<String, List<String>> allTests, File file) {
254         if (file.isFile()) {
255             String path = null;
256 
257             try {
258                 path = getSimplePath(file.getCanonicalPath());
259             }
260             catch (IOException ex) {
261                 throw new IllegalStateException(ex);
262             }
263 
264             String fileName = file.getName().toString();
265 
266             Assert.assertTrue("Resource must start with 'Input': " + path,
267                     fileName.startsWith("Input"));
268 
269             final int period = fileName.lastIndexOf('.');
270 
271             Assert.assertTrue("Resource must have an extension: " + path, period > 0);
272 
273             fileName = fileName.substring(5, period);
274 
275             final int slash = path.lastIndexOf(File.separatorChar);
276             final String pkg = path.substring(0, slash);
277             final List<String> classes = allTests.get(pkg);
278 
279             if (classes != null || !pkg.endsWith("external")) {
280                 Assert.assertNotNull("Resource must be in a package that has tests: " + path,
281                         classes);
282 
283                 boolean found = false;
284 
285                 for (String clss : classes) {
286                     if (fileName.startsWith(clss)) {
287                         found = true;
288                         break;
289                     }
290                 }
291 
292                 Assert.assertTrue("Resource must be named after a Test like 'InputMyCheck.java' "
293                         + "and be in the same package as the test: " + path, found);
294             }
295         }
296     }

298     /**
299      * Removes 'Check' suffix from each class name in the set.
300      * @param checks class instances.
301      * @return a set of simple names.
302      */
303     private static Set<String> getFullNames(Set<Class<?>> checks) {
304         return checks.stream().map(check -> check.getName())
305             .collect(Collectors.toSet());
306     }
307 
308     private static String getSimplePath(String path) {
309         return path.substring(path.lastIndexOf("com" + File.separator + "github"));
310     }

312     /**
313      * Checks that an array is a subset of other array.
314      * @param array to check whether it is a subset.
315      * @param arrayToCheckIn array to check in.
316      */
317     private static boolean isSubset(int[] array, int... arrayToCheckIn) {
318         boolean result = true;
319         if (arrayToCheckIn == null) {
320             result = array == null || array.length == 0;
321         }
322         else {
323             Arrays.sort(arrayToCheckIn);
324             for (final int element : array) {
325                 if (Arrays.binarySearch(arrayToCheckIn, element) < 0) {
326                     result = false;
327                     break;
328                 }
329             }
330         }
331         return result;
332     }
