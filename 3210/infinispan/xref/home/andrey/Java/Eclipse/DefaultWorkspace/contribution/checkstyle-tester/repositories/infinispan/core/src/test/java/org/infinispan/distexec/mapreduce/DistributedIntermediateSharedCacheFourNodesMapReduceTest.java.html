14  /**
15   * DistributedIntermediateSharedCacheFourNodesMapReduceTest tests Map/Reduce functionality using four Infinispan nodes,
16   * distributed reduce and shared intermediate key/value cache specified by application
17   *
18   * @author Vladimir Blagojevic
19   * @since 7.0
20   */
21  @Test(groups = "functional", testName = "distexec.mapreduce.DistributedIntermediateSharedCacheFourNodesMapReduceTest")
22  public class DistributedIntermediateSharedCacheFourNodesMapReduceTest extends BaseWordCountMapReduceTest {
23  
24     private String intermediateCacheName = "tmpCache";
25     private String intermediateCacheNameConfig = "tmpCacheConfig";
26  
27     @Override
28     protected void createCacheManagers() throws Throwable {
29        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(getCacheMode(), true);
30        createClusteredCaches(4, cacheName(), builder);
31     }
32  
33     /**
34      * We are overriding createMapReduceTask factory method from superclass so that all test methods
35      * from superclass are run with application specified intermediate cache.
36      *
37      */
38     @SuppressWarnings({ "rawtypes", "unchecked" })
39     @Override
40     protected MapReduceTask<String, String, String, Integer> createMapReduceTask(Cache c) {
41        //run distributed reduce with per task cache - cache specified by the user
42        MapReduceTask<String, String, String, Integer> t = new MapReduceTask<String, String, String, Integer>(c, true,
43              false);
44        ConfigurationBuilder cacheConfig = new ConfigurationBuilder();
45        cacheConfig.unsafe().unreliableReturnValues(true)
46        .clustering().cacheMode(CacheMode.DIST_SYNC).hash().numOwners(2).sync();
47  
48        //In the real world people will define configurations using xml files
49        defineConfigurationOnAllManagers(intermediateCacheNameConfig, cacheConfig);
50        return t.usingSharedIntermediateCache(intermediateCacheName, intermediateCacheNameConfig);
51     }
52     
53     @Test(expectedExceptions={CacheException.class})
54     public void testIntermediateCacheNotCreatedOnAllNodes() throws Exception {
55        String cacheNameConfig = "notCreatedOnAllNodes";
56        Cache c = cache(0, cacheName());
57        MapReduceTask<String, String, String, Integer> t = new MapReduceTask<String, String, String, Integer>(c, true,
58              false);
59        ConfigurationBuilder cacheConfig = new ConfigurationBuilder();
60        cacheConfig.unsafe().unreliableReturnValues(true).clustering().cacheMode(CacheMode.DIST_SYNC).hash().numOwners(2)
61              .sync();
62  
63        //define configuration only on first node
64        Iterator<EmbeddedCacheManager> iterator = getCacheManagers().iterator();
65        iterator.next().defineConfiguration(cacheNameConfig, cacheConfig.build());
66  
67        t.usingSharedIntermediateCache("irrelevant", cacheNameConfig);
68        t.mappedWith(new WordCountMapper()).reducedWith(new WordCountReducer());
69        t.execute();
70     }
