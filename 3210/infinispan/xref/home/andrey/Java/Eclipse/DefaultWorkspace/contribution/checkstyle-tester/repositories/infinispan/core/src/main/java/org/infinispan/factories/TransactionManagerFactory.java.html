1   package org.infinispan.factories;
2   
3   import javax.transaction.TransactionManager;
4   
5   import org.infinispan.commons.CacheException;
6   import org.infinispan.factories.annotations.DefaultFactoryFor;
7   import org.infinispan.transaction.lookup.TransactionManagerLookup;
8   import org.infinispan.transaction.tm.BatchModeTransactionManager;
9   import org.infinispan.util.logging.Log;
10  import org.infinispan.util.logging.LogFactory;
11  
12  /**
13   * Uses a number of mechanisms to retrieve a transaction manager.
14   *
15   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
16   * @author Galder Zamarre√±o
17   * @since 4.0
18   */
19  @DefaultFactoryFor(classes = {TransactionManager.class})
20  public class TransactionManagerFactory extends AbstractNamedCacheComponentFactory implements AutoInstantiableFactory {
21  
22     private static final Log log = LogFactory.getLog(TransactionManagerFactory.class);
23  
24     @Override
25     public <T> T construct(Class<T> componentType) {
26        if (!configuration.transaction().transactionMode().isTransactional()) {
27           return null;
28        }
29  
30        // See if we had a TransactionManager injected into our config
31        TransactionManager transactionManager = null;
32  
33        TransactionManagerLookup lookup = configuration.transaction().transactionManagerLookup();
34        try {
35           if (lookup != null) {
36              componentRegistry.wireDependencies(lookup);
37              transactionManager = lookup.getTransactionManager();
38           }
39        } catch (Exception e) {
40           log.couldNotInstantiateTransactionManager(e);
41        }
42  
43        if (transactionManager == null && configuration.invocationBatching().enabled()) {
44           log.usingBatchModeTransactionManager();
45           transactionManager = BatchModeTransactionManager.getInstance();
46        }
47  
48        if (transactionManager == null) {
49           throw new CacheException("This is transactional cache but no transaction manager could be found. " +
50                                          "Configure the transaction manager lookup properly.");
51        }
52  
53        return componentType.cast(transactionManager);
54     }
55  }
