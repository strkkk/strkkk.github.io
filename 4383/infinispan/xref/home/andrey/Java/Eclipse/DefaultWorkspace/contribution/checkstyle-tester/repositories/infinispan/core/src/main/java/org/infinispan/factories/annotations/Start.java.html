1   package org.infinispan.factories.annotations;
2   
3   import static java.lang.annotation.ElementType.METHOD;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   /**
9    * Method level annotation that indicates a (no-param) method to be called on a component registered in the
10   * ComponentRegistry when the cache starts.
11   * <p/>
12   *
13   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
14   * @since 4.0
15   */
16  @Target(METHOD)
17  @Retention(RetentionPolicy.RUNTIME)
18  public @interface Start {
19     /**
20      * Optional parameter which defines the order in which this method will be called when the ComponentRegistry moves to
21      * the STARTED state.  Defaults to 10.
22      *
23      * @return execution priority
24      * @since 4.0
25      */
26     public abstract int priority() default 10;
27  }
