26  /**
27   * Local file-based cache store, optimized for write-through use with strong consistency guarantees
28   * (ability to flush disk operations before returning from the store call).
29   *
30   * * DESIGN:
31   * There are three threads operating in the cache-store:
32   * - LogAppender:  Requests to store entries are passed to the LogAppender thread
33   *                 via queue, then the requestor threads wait until LogAppender notifies
34   *                 them about successful store. LogAppender serializes the writes
35   *                 into append-only file, writes the offset into TemporaryTable
36   *                 and enqueues request to update index into UpdateQueue.
37   *                 The append-only files have limited size, when the file is full,
38   *                 new file is started.
39   * - IndexUpdater: Reads the UpdateQueue, applies the operation into B-tree-like
40   *                 structure Index (exact description below) and then removes
41   *                 the entry from TemporaryTable. When the Index is overwriten,
42   *                 the current entry offset is retrieved and IndexUpdater increases
43   *                 the unused space statistics in FileStats.
44   * - Compactor:    When a limit of unused space in some file is reached (according
45   *                 to FileStats), the Compactor starts reading this file sequentially,
46   *                 querying TemporaryTable or Index for the current entry position
47   *                 and copying the unchanged entries into another file. For the entries
48   *                 that are still valid in the original file, a compare-and-set
49   *                 (file-offset based) request is enqueued into UpdateQueue - therefore
50   *                 this operation cannot interfere with concurrent writes overwriting
51   *                 the entry. Multiple files can be merged into single file during
52   *                 compaction.
53   *
54   * Structures:
55   * - TemporaryTable: keeps the records about current entry location until this is
56   *                   applied to the Index. Each read request goes to the TemporaryTable,
57   *                   if the key is not found here, Index is queried.
58   * - UpdateQueue:    bounded queue (to prevent grow the TemporaryTable too much) of either
59   *                   forced writes (used for regular stores) or compare-and-set writes
60   *                   (used by Compactor).
61   * - FileStats:      simple (Concurrent)HashTable with actual file size and amount of unused
62   *                   space for each file.
63   * - Index:          B+-tree of IndexNodes. The tree is dropped and built a new if the process
64   *                   crashes, it does not need to flush disk operations. On disk it is kept as single random-accessed file, with free blocks list stored in memory.
65   *
66   * As IndexUpdater may easily become a bottleneck under heavy load, the IndexUpdater thread,
67   * UpdateQueue and tree of IndexNodes may be multiplied several times - the Index is divided
68   * into Segments. Each segment owns keys according to the hashCode() of the key.
69   *
70   * Amount of entries in IndexNode is limited by the size it occupies on disk. This size is
71   * limited by configurable nodeSize (4096 bytes by default?), only in case that the node
72   * contains single pivot (too long) it can be longer. A key_prefix common for all keys
73   * in the IndexNode is stored in order to reduce space requirements. For implementation
74   * reasons the keys are limited to 32kB - this requirement may be circumvented later.
75   *
76   * The pivots are not whole keys - it is the shortest part of key that is greater than all
77   * left children (but lesser or equal to all right children) - let us call this key_part.
78   * The key_parts are sorted in the IndexNode, naturally. On disk it has this format:
79   *
80   *  key_prefix_length(2 bytes), key_prefix, num_parts(2 bytes),
81   *     ( key_part_length (2 bytes), key_part, left_child_index_node_offset (8 bytes))+,
82   *     right_child_index_node_offset (8 bytes)
83   *
84   * In memory, for every child a SoftReference<IndexNode> is held. When this reference
85   * is empty (but the offset in file is set), any reader may load the reference using
86   * double-locking pattern (synchronized over the reference itself). The entry is never
87   * loaded by multiple threads in parallel and even may block other threads trying to
88   * read this node.
89   *
90   * For each node in memory a RW-lock is held. When the IndexUpdater thread updates
91   * the Index (modifying some IndexNodes), it prepares a copy of these nodes (already
92   * stored into index file). Then, in locks only the uppermost node for writing, overwrites
93   * the references to new data and unlocks the this node. After that the changed nodes are
94   * traversed from top down, write locked and their record in index file is released.
95   * Reader threads crawl the tree from top down, locking the parent node (for reading),
96   * locking child node and unlocking parent node.
97   *
98   * @author Radim Vansa &lt;rvansa@redhat.com&gt;
99   */
100 public class SoftIndexFileStore implements AdvancedLoadWriteStore {

120    @Override
121    public void init(InitializationContext ctx) {
122       configuration = ctx.getConfiguration();

131    @Override
132    public void start() {
133       log.info("Starting using configuration " + configuration);
134       temporaryTable = new TemporaryTable(configuration.indexQueueLength() * configuration.indexSegments(), keyEquivalence);
135       storeQueue = new SyncProcessingQueue<LogRequest>();
136       indexQueue = new IndexQueue(configuration.indexSegments(), configuration.indexQueueLength(), keyEquivalence);
137       fileProvider = new FileProvider(configuration.dataLocation(), configuration.openFilesLimit());
138       compactor = new Compactor(fileProvider, temporaryTable, indexQueue, marshaller, timeService, configuration.maxFileSize(), configuration.compactionThreshold());
139       logAppender = new LogAppender(storeQueue, indexQueue, temporaryTable, compactor, fileProvider, configuration.syncWrites(), configuration.maxFileSize());
140       try {
141          index = new Index(fileProvider, configuration.indexLocation(), configuration.indexSegments(),
142                configuration.minNodeSize(), configuration.maxNodeSize(),
143                indexQueue, temporaryTable, compactor, timeService, keyEquivalence);
144       } catch (IOException e) {
145          throw new PersistenceException("Cannot open index file in " + configuration.indexLocation(), e);
146       }
147       compactor.setIndex(index);
148       final AtomicLong maxSeqId = new AtomicLong(0);
149       if (configuration.purgeOnStartup()) {
150          log.debug("Not building the index - purge will be executed");
151       } else {
152          log.debug("Building the index");
153          forEachOnDisk(false, false, new EntryFunctor() {
154             @Override
155             public boolean apply(int file, int offset, int size, byte[] serializedKey, byte[] serializedMetadata, byte[] serializedValue, long seqId, long expiration) throws IOException, ClassNotFoundException {
156                long prevSeqId;
157                while (seqId > (prevSeqId = maxSeqId.get()) && !maxSeqId.compareAndSet(prevSeqId, seqId));
158                Object key = marshaller.objectFromByteBuffer(serializedKey);
159                if (trace) log.tracef("Loaded %d:%d (seqId %d, expiration %d)", file, offset, seqId, expiration);
160                try {
161                   // We may check the seqId safely as we are the only thread writing to index
162                   EntryPosition entry = temporaryTable.get(key);
163                   if (entry == null) {
164                      entry = index.getPosition(key, serializedKey);
165                   }
166                   if (entry != null) {
167                      FileProvider.Handle handle = fileProvider.getFile(entry.file);
168                      try {
169                         EntryHeader header = EntryRecord.readEntryHeader(handle, entry.offset);
170                         if (header == null) {
171                            throw new IllegalStateException("Cannot read " + entry.file + ":" + entry.offset);
172                         }
173                         if (seqId < header.seqId()) {
174                            if (trace) log.tracef("Record on %d:%d has seqId %d > %d", entry.file, entry.offset, header.seqId(), seqId);
175                            return true;
176                         }
177                      } finally {
178                         handle.close();
179                      }
180                   }
181                   temporaryTable.set(key, file, offset);
182                   indexQueue.put(new IndexRequest(key, serializedKey, file, offset, size));
183                } catch (InterruptedException e) {
184                   log.error("Interrupted building of index, the index won't be built properly!", e);
185                   return false;
186                }
187                return true;
188             }
189          }, new FileFunctor() {
190             @Override
191             public void afterFile(int file) {
192                compactor.completeFile(file);
193             }
194          });
195       }
196       logAppender.setSeqId(maxSeqId.get() + 1);
197    }

199    @Override
200    public void stop() {
201       try {
202          logAppender.stopOperations();
203          logAppender = null;
204          compactor.stopOperations();
205          compactor = null;
206          index.stopOperations();
207          index = null;
208          fileProvider.stop();
209          fileProvider = null;
210          temporaryTable = null;
211          indexQueue = null;
212          storeQueue = null;
213       } catch (InterruptedException e) {
214          throw new PersistenceException("Cannot stop cache store", e);
215       }
216    }

218    @Override
219    public synchronized void clear() throws PersistenceException {
220       try {
221          logAppender.clearAndPause();
222          compactor.clearAndPause();
223       } catch (InterruptedException e) {
224          throw new PersistenceException("Cannot pause cache store to clear it.", e);
225       }
226       try {
227          index.clear();
228       } catch (IOException e) {
229          throw new PersistenceException("Cannot clear/reopen index!", e);
230       }
231       try {
232          fileProvider.clear();
233       } catch (IOException e) {
234          throw new PersistenceException("Cannot clear data directory!", e);
235       }
236       temporaryTable.clear();
237       compactor.resumeAfterPause();
238       logAppender.resumeAfterPause();
239    }

241    @Override
242    public synchronized int size() {
243       try {
244          logAppender.pause();
245          return (int) index.size();
246       } catch (InterruptedException e) {
247          log.error("Interrupted", e);
248          Thread.currentThread().interrupt();
249          return -1;
250       } finally {
251          logAppender.resumeAfterPause();
252       }
253    }

261    @Override
262    public void write(MarshalledEntry entry) {
263       int keyLength = entry.getKeyBytes().getLength();
264       if (keyLength > maxKeyLength) {
265          throw new PersistenceException("Configuration 'maxNodeSize' is too low - with maxNodeSize="
266                + configuration.maxNodeSize() + " bytes you can use only keys serialized to " + maxKeyLength
267                + " bytes (key " + entry.getKey() + " is serialized to " + keyLength + " bytes)");
268       } else if (keyLength > Short.MAX_VALUE) {
269          // TODO this limitation could be removed by different key length encoding
270          throw new PersistenceException("SoftIndexFileStore is limited to keys with serialized size <= 32767 bytes");
271       }
272       try {
273          storeQueue.pushAndWait(LogRequest.storeRequest(entry));
274       } catch (Exception e) {
275          throw new PersistenceException(e);
276       }
277    }

279    @Override
280    public boolean delete(Object key) {
281       try {
282          LogRequest request = LogRequest.deleteRequest(key, toBuffer(marshaller.objectToByteBuffer(key)));
283          storeQueue.pushAndWait(request);
284          return (Boolean) request.getIndexRequest().getResult();
285       } catch (Exception e) {
286          throw new PersistenceException(e);
287       }
288    }

290    @Override
291    public boolean contains(Object key) {
292       try {
293          for (;;) {
294             // TODO: consider storing expiration timestamp in temporary table
295             EntryPosition entry = temporaryTable.get(key);
296             if (entry != null) {
297                if (entry.offset < 0) {
298                   return false;
299                }
300                FileProvider.Handle handle = fileProvider.getFile(entry.file);
301                if (handle != null) {
302                   try {
303                      EntryHeader header = EntryRecord.readEntryHeader(handle, entry.offset);
304                      if (header == null) {
305                         throw new IllegalStateException("Error reading from " + entry.file + ":" + entry.offset + " | " + handle.getFileSize());
306                      }
307                      return header.expiryTime() < 0 || header.expiryTime() > timeService.wallClockTime();
308                   } finally {
309                      handle.close();
310                   }
311                }
312             } else {
313                EntryPosition position = index.getPosition(key, marshaller.objectToByteBuffer(key));
314                return position != null;
315             }
316          }
317       } catch (Exception e) {
318          throw new PersistenceException("Cannot load key from index", e);
319       }
320    }

322    @Override
323    public MarshalledEntry load(Object key) {
324       try {
325          byte[] serializedValue;
326          byte[] serializedKey;
327          byte[] serializedMetadata;
328          for (;;) {
329             EntryPosition entry = temporaryTable.get(key);
330             if (entry != null) {
331                if (entry.offset < 0) {
332                   return null;
333                }
334                FileProvider.Handle handle = fileProvider.getFile(entry.file);
335                if (handle != null) {
336                   try {
337                      EntryHeader header = EntryRecord.readEntryHeader(handle, entry.offset);
338                      if (header == null) {
339                         throw new IllegalStateException("Error reading from " + entry.file + ":" + entry.offset + " | " + handle.getFileSize());
340                      }
341                      if (header.expiryTime() > 0 && header.expiryTime() <= timeService.wallClockTime()) {
342                         return null;
343                      }
344                      serializedKey = EntryRecord.readKey(handle, header, entry.offset);
345                      if (header.metadataLength() > 0) {
346                         serializedMetadata = EntryRecord.readMetadata(handle, header, entry.offset);
347                      } else {
348                         serializedMetadata = null;
349                      }
350                      if (header.valueLength() > 0) {
351                         serializedValue = EntryRecord.readValue(handle, header, entry.offset);
352                      } else {
353                         return null;
354                      }
355                      return marshalledEntryFactory.newMarshalledEntry(toBuffer(serializedKey), toBuffer(serializedValue), toBuffer(serializedMetadata));
356                   } finally {
357                      handle.close();
358                   }
359                }
360             } else {
361                EntryRecord record = index.getRecord(key, marshaller.objectToByteBuffer(key));
362                if (record == null) return null;
363                return marshalledEntryFactory.newMarshalledEntry(toBuffer(record.getKey()), toBuffer(record.getValue()), toBuffer(record.getMetadata()));
364             }
365          }
366       } catch (Exception e) {
367          throw new PersistenceException(e);
368       }
369    }

391    private ByteBuffer toBuffer(byte[] array) {
392       return array == null ? null : byteBufferFactory.newByteBuffer(array, 0, array.length);

395    private interface EntryFunctor {
396       boolean apply(int file, int offset, int size, byte[] serializedKey, byte[] serializedMetadata, byte[] serializedValue, long seqId, long expiration) throws Exception;
397    }
398 
399    private interface FileFunctor {
400       void afterFile(int file);
401    }
402 
403    private void forEachOnDisk(boolean readMetadata, boolean readValues, EntryFunctor functor, FileFunctor fileFunctor) throws PersistenceException {
404       try {
405          Iterator<Integer> iterator = fileProvider.getFileIterator();
406          while (iterator.hasNext()) {
407             int file = iterator.next();
408             log.debug("Loading entries from file " + file);
409             FileProvider.Handle handle = fileProvider.getFile(file);
410             if (handle == null) {
411                log.debug("File " + file + " was deleted during iteration");
412                fileFunctor.afterFile(file);
413                continue;
414             }
415             try {
416                int offset = 0;
417                for (;;) {
418                   EntryHeader header = EntryRecord.readEntryHeader(handle, offset);
419                   if (header == null) {
420                      break; // end of file;
421                   }
422                   try {
423                      byte[] serializedKey = EntryRecord.readKey(handle, header, offset);
424                      if (serializedKey == null) {
425                         break; // we have read the file concurrently with writing there
426                         //throw new CacheLoaderException("File " + file + " appears corrupt when reading key from " + offset + ": header is " + header);
427                      }
428                      byte[] serializedMetadata = null;
429                      if (readMetadata && header.metadataLength() > 0) {
430                         serializedMetadata = EntryRecord.readMetadata(handle, header, offset);
431                      }
432                      byte[] serializedValue = null;
433                      int offsetOrNegation = offset;
434                      if (header.valueLength() > 0) {
435                         if (header.expiryTime() >= 0 && header.expiryTime() <= timeService.wallClockTime()) {
436                            offsetOrNegation = ~offset;
437                         } else if (readValues) {
438                            serializedValue = EntryRecord.readValue(handle, header, offset);
439                         }
440                      } else {
441                         offsetOrNegation = ~offset;
442                      }
443                      if (!functor.apply(file, offsetOrNegation, header.totalLength(), serializedKey, serializedMetadata, serializedValue, header.seqId(), header.expiryTime())) {
444                         return;
445                      }
446                   } finally {
447                      offset += header.totalLength();
448                   }
449                }
450             } finally {
451                handle.close();
452                fileFunctor.afterFile(file);
453             }
454          }
455       } catch (Exception e) {
456          throw new PersistenceException(e);
457       }
458    }

460    @Override
461    public void process(KeyFilter filter, final CacheLoaderTask task, final Executor executor, final boolean fetchValue, final boolean fetchMetadata) {
462       final TaskContext context = new TaskContextImpl();
463       final KeyFilter notNullFilter = PersistenceUtil.notNull(filter);
464       final AtomicLong tasksSubmitted = new AtomicLong();
465       final AtomicLong tasksFinished = new AtomicLong();
466       forEachOnDisk(fetchMetadata, fetchValue, new EntryFunctor() {
467          @Override
468          public boolean apply(int file, int offset, int size,
469                               final byte[] serializedKey, final byte[] serializedMetadata, final byte[] serializedValue,
470                               long seqId, long expiration) throws IOException, ClassNotFoundException {
471             if (context.isStopped()) {
472                return false;
473             }
474             final Object key = marshaller.objectFromByteBuffer(serializedKey);
475             if (!notNullFilter.accept(key)) {
476                return true;
477             }
478             EntryPosition entry = temporaryTable.get(key);
479             if (entry == null) {
480                entry = index.getPosition(key, serializedKey);
481             }
482             if (entry != null && entry.offset >= 0) {
483                FileProvider.Handle handle = fileProvider.getFile(entry.file);
484                try {
485                   EntryHeader header = EntryRecord.readEntryHeader(handle, entry.offset);
486                   if (header == null) {
487                      throw new IllegalStateException("Cannot read " + entry.file + ":" + entry.offset);
488                   }
489                   if (seqId < header.seqId()) {
490                      return true;
491                   }
492                } finally {
493                   handle.close();
494                }
495             } else {
496                // entry is not in index = it was deleted
497                return true;
498             }
499             if (serializedValue != null && (expiration < 0 || expiration > timeService.wallClockTime())) {
500                executor.execute(new Runnable() {
501                   @Override
502                   public void run() {
503                      try {
504                         task.processEntry(marshalledEntryFactory.newMarshalledEntry(key,
505                               serializedValue == null ? null : marshaller.objectFromByteBuffer(serializedValue),
506                               serializedMetadata == null ? null : (InternalMetadata) marshaller.objectFromByteBuffer(serializedMetadata)),
507                               context);
508                      } catch (Exception e) {
509                         log.error("Failed to process task for key " + key, e);
510                      } finally {
511                         long finished = tasksFinished.incrementAndGet();
512                         if (finished == tasksSubmitted.longValue()) {
513                            synchronized (context) {
514                               context.notifyAll();
515                            }
516                         }
517                      }
518                   }
519                });
520                tasksSubmitted.incrementAndGet();
521                return !context.isStopped();
522             }
523             return true;
524          }
525       }, new FileFunctor() {
526          @Override
527          public void afterFile(int file) {
528             // noop
529          }
530       });
531       while (tasksSubmitted.longValue() > tasksFinished.longValue()) {
532          synchronized (context) {
533             try {
534                context.wait(100);
535             } catch (InterruptedException e) {
536                log.error("Iteration was interrupted", e);
537                Thread.currentThread().interrupt();
538                return;
539             }
540          }
541       }
542    }
