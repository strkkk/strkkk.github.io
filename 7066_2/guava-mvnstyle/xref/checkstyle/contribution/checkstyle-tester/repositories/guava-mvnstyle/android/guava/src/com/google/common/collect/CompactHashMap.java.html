44  /**
45   * CompactHashMap is an implementation of a Map. All optional operations (put and remove) are
46   * supported. Null keys and values are supported.
47   *
48   * <p>{@code containsKey(k)}, {@code put(k, v)} and {@code remove(k)} are all (expected and
49   * amortized) constant time operations. Expected in the hashtable sense (depends on the hash
50   * function doing a good job of distributing the elements to the buckets to a distribution not far
51   * from uniform), and amortized since some operations can trigger a hash table resize.
52   *
53   * <p>Unlike {@code java.util.HashMap}, iteration is only proportional to the actual {@code size()},
54   * which is optimal, and <i>not</i> the size of the internal hashtable, which could be much larger
55   * than {@code size()}. Furthermore, this structure places significantly reduced load on the garbage
56   * collector by only using a constant number of internal objects.
57   *
58   * <p>If there are no removals, then iteration order for the {@link #entrySet}, {@link #keySet}, and
59   * {@link #values} views is the same as insertion order. Any removal invalidates any ordering
60   * guarantees.
61   *
62   * <p>This class should not be assumed to be universally superior to {@code java.util.HashMap}.
63   * Generally speaking, this class reduces object allocation and memory consumption at the price of
64   * moderately increased constant factors of CPU.  Only use this class when there is a specific
65   * reason to prioritize memory over CPU.
66   *
67   * @author Louis Wasserman
68   */
69  @GwtIncompatible // not worth using in GWT for now
70  class CompactHashMap<K, V> extends AbstractMap<K, V> implements Serializable {
71    /*
72     * TODO: Make this a drop-in replacement for j.u. versions, actually drop them in, and test the
73     * world. Figure out what sort of space-time tradeoff we're actually going to get here with the
74     * *Map variants. Followon optimizations, such as using 16-bit indices for small collections, will
75     * take more work to implement. This class is particularly hard to benchmark, because the benefit
76     * is not only in less allocation, but also having the GC do less work to scan the heap because of
77     * fewer references, which is particularly hard to quantify.
78     */

103   /** Bitmask that selects the low 32 bits. */
104   private static final long NEXT_MASK = (1L << 32) - 1;
105 
106   /** Bitmask that selects the high 32 bits. */
107   private static final long HASH_MASK = ~NEXT_MASK;

115   /**
116    * The hashtable. Its values are indexes to the keys, values, and entries arrays.
117    *
118    * <p>Currently, the UNSET value means "null pointer", and any non negative value x is the actual
119    * index.
120    *
121    * <p>Its size must be a power of two.
122    */
123   @MonotonicNonNullDecl private transient int[] table;
124 
125   /**
126    * Contains the logical entries, in the range of [0, size()). The high 32 bits of each long is the
127    * smeared hash of the element, whereas the low 32 bits is the "next" pointer (pointing to the
128    * next entry in the bucket chain). The pointers in [size(), entries.length) are all "null"
129    * (UNSET).
130    */
131   @MonotonicNonNullDecl @VisibleForTesting transient long[] entries;
132 
133   /**
134    * The keys of the entries in the map, in the range of [0, size()). The keys in [size(),
135    * keys.length) are all {@code null}.
136    */
137   @MonotonicNonNullDecl @VisibleForTesting transient Object[] keys;
138 
139   /**
140    * The values of the entries in the map, in the range of [0, size()). The values in [size(),
141    * values.length) are all {@code null}.
142    */
143   @MonotonicNonNullDecl @VisibleForTesting transient Object[] values;

179   /** Pseudoconstructor for serialization support. */
180   void init(int expectedSize, float loadFactor) {
181     Preconditions.checkArgument(expectedSize >= 0, "Initial capacity must be non-negative");
182     Preconditions.checkArgument(loadFactor > 0, "Illegal load factor");
183     int buckets = Hashing.closedTableSize(expectedSize, loadFactor);
184     this.table = newTable(buckets);
185     this.loadFactor = loadFactor;
186 
187     this.keys = new Object[expectedSize];
188     this.values = new Object[expectedSize];
189 
190     this.entries = newEntries(expectedSize);
191     this.threshold = Math.max(1, (int) (buckets * loadFactor));
192   }

206   private int hashTableMask() {

210   private static int getHash(long entry) {

214   /** Returns the index, or UNSET if the pointer is "null" */

219   /** Returns a new entry value by changing the "next" index of an existing entry */
220   private static long swapNext(long entry, int newNext) {
221     return (HASH_MASK & entry) | (NEXT_MASK & newNext);
222   }

232   @CanIgnoreReturnValue
233   @Override
234   @NullableDecl
235   public V put(@NullableDecl K key, @NullableDecl V value) {
236     long[] entries = this.entries;
237     Object[] keys = this.keys;
238     Object[] values = this.values;
239 
240     int hash = smearedHash(key);
241     int tableIndex = hash & hashTableMask();
242     int newEntryIndex = this.size; // current size, and pointer to the entry to be appended
243     int next = table[tableIndex];
244     if (next == UNSET) {
245       table[tableIndex] = newEntryIndex;
246     } else {
247       int last;
248       long entry;
249       do {
250         last = next;
251         entry = entries[next];
252         if (getHash(entry) == hash && Objects.equal(key, keys[next])) {
253           @SuppressWarnings("unchecked") // known to be a V
254           @NullableDecl
255           V oldValue = (V) values[next];
256 
257           values[next] = value;
258           accessEntry(next);
259           return oldValue;
260         }
261         next = getNext(entry);
262       } while (next != UNSET);
263       entries[last] = swapNext(entry, newEntryIndex);
264     }
265     if (newEntryIndex == Integer.MAX_VALUE) {
266       throw new IllegalStateException("Cannot contain more than Integer.MAX_VALUE elements!");
267     }
268     int newSize = newEntryIndex + 1;
269     resizeMeMaybe(newSize);
270     insertEntry(newEntryIndex, key, value, hash);
271     this.size = newSize;
272     if (newEntryIndex >= threshold) {
273       resizeTable(2 * table.length);
274     }
275     modCount++;
276     return null;
277   }

279   /**
280    * Creates a fresh entry with the specified object at the specified position in the entry arrays.
281    */
282   void insertEntry(int entryIndex, @NullableDecl K key, @NullableDecl V value, int hash) {
283     this.entries[entryIndex] = ((long) hash << 32) | (NEXT_MASK & UNSET);
284     this.keys[entryIndex] = key;
285     this.values[entryIndex] = value;
286   }
287 
288   /** Returns currentSize + 1, after resizing the entries storage if necessary. */
289   private void resizeMeMaybe(int newSize) {
290     int entriesSize = entries.length;
291     if (newSize > entriesSize) {
292       int newCapacity = entriesSize + Math.max(1, entriesSize >>> 1);
293       if (newCapacity < 0) {
294         newCapacity = Integer.MAX_VALUE;
295       }
296       if (newCapacity != entriesSize) {
297         resizeEntries(newCapacity);
298       }
299     }
300   }
301 
302   /**
303    * Resizes the internal entries array to the specified capacity, which may be greater or less than
304    * the current capacity.
305    */
306   void resizeEntries(int newCapacity) {
307     this.keys = Arrays.copyOf(keys, newCapacity);
308     this.values = Arrays.copyOf(values, newCapacity);
309     long[] entries = this.entries;
310     int oldCapacity = entries.length;
311     entries = Arrays.copyOf(entries, newCapacity);
312     if (newCapacity > oldCapacity) {
313       Arrays.fill(entries, oldCapacity, newCapacity, UNSET);
314     }
315     this.entries = entries;
316   }
317 
318   private void resizeTable(int newCapacity) { // newCapacity always a power of two
319     int[] oldTable = table;
320     int oldCapacity = oldTable.length;
321     if (oldCapacity >= MAXIMUM_CAPACITY) {
322       threshold = Integer.MAX_VALUE;
323       return;
324     }
325     int newThreshold = 1 + (int) (newCapacity * loadFactor);
326     int[] newTable = newTable(newCapacity);
327     long[] entries = this.entries;
328 
329     int mask = newTable.length - 1;
330     for (int i = 0; i < size; i++) {
331       long oldEntry = entries[i];
332       int hash = getHash(oldEntry);
333       int tableIndex = hash & mask;
334       int next = newTable[tableIndex];
335       newTable[tableIndex] = i;
336       entries[i] = ((long) hash << 32) | (NEXT_MASK & next);
337     }
338 
339     this.threshold = newThreshold;
340     this.table = newTable;
341   }

343   private int indexOf(@NullableDecl Object key) {
344     int hash = smearedHash(key);
345     int next = table[hash & hashTableMask()];
346     while (next != UNSET) {
347       long entry = entries[next];
348       if (getHash(entry) == hash && Objects.equal(key, keys[next])) {
349         return next;
350       }
351       next = getNext(entry);
352     }
353     return -1;
354   }

376   @NullableDecl
377   private V remove(@NullableDecl Object key, int hash) {
378     int tableIndex = hash & hashTableMask();
379     int next = table[tableIndex];
380     if (next == UNSET) { // empty bucket
381       return null;
382     }
383     int last = UNSET;
384     do {
385       if (getHash(entries[next]) == hash) {
386         if (Objects.equal(key, keys[next])) {
387           @SuppressWarnings("unchecked") // values only contains Vs
388           @NullableDecl
389           V oldValue = (V) values[next];
390 
391           if (last == UNSET) {
392             // we need to update the root link from table[]
393             table[tableIndex] = getNext(entries[next]);
394           } else {
395             // we need to update the link from the chain
396             entries[last] = swapNext(entries[last], getNext(entries[next]));
397           }
398 
399           moveLastEntry(next);
400           size--;
401           modCount++;
402           return oldValue;
403         }
404       }
405       last = next;
406       next = getNext(entries[next]);
407     } while (next != UNSET);
408     return null;
409   }

413     return remove(keys[entryIndex], getHash(entries[entryIndex]));

416   /**
417    * Moves the last entry in the entry array into {@code dstIndex}, and nulls out its old position.
418    */
419   void moveLastEntry(int dstIndex) {
420     int srcIndex = size() - 1;
421     if (dstIndex < srcIndex) {
422       // move last entry to deleted spot
423       keys[dstIndex] = keys[srcIndex];
424       values[dstIndex] = values[srcIndex];
425       keys[srcIndex] = null;
426       values[srcIndex] = null;
427 
428       // move the last entry to the removed spot, just like we moved the element
429       long lastEntry = entries[srcIndex];
430       entries[dstIndex] = lastEntry;
431       entries[srcIndex] = UNSET;
432 
433       // also need to update whoever's "next" pointer was pointing to the last entry place
434       // reusing "tableIndex" and "next"; these variables were no longer needed
435       int tableIndex = getHash(lastEntry) & hashTableMask();
436       int lastNext = table[tableIndex];
437       if (lastNext == srcIndex) {
438         // we need to update the root pointer
439         table[tableIndex] = dstIndex;
440       } else {
441         // we need to update a pointer in an entry
442         int previous;
443         long entry;
444         do {
445           previous = lastNext;
446           lastNext = getNext(entry = entries[lastNext]);
447         } while (lastNext != srcIndex);
448         // here, entries[previous] points to the old entry location; update it
449         entries[previous] = swapNext(entry, dstIndex);
450       }
451     } else {
452       keys[dstIndex] = null;
453       values[dstIndex] = null;
454       entries[dstIndex] = UNSET;
455     }
456   }

466   /**
467    * Updates the index an iterator is pointing to after a call to remove: returns the index of the
468    * entry that should be looked at after a removal on indexRemoved, with indexBeforeRemove as the
469    * index that *was* the next entry that would be looked at.
470    */
471   int adjustAfterRemove(int indexBeforeRemove, @SuppressWarnings("unused") int indexRemoved) {
472     return indexBeforeRemove - 1;
473   }

475   private abstract class Itr<T> implements Iterator<T> {
476     int expectedModCount = modCount;
477     int currentIndex = firstEntryIndex();
478     int indexToRemove = -1;
479 
480     @Override
481     public boolean hasNext() {
482       return currentIndex >= 0;
483     }
484 
485     abstract T getOutput(int entry);
486 
487     @Override
488     public T next() {
489       checkForConcurrentModification();
490       if (!hasNext()) {
491         throw new NoSuchElementException();
492       }
493       indexToRemove = currentIndex;
494       T result = getOutput(currentIndex);
495       currentIndex = getSuccessor(currentIndex);
496       return result;
497     }
498 
499     @Override
500     public void remove() {
501       checkForConcurrentModification();
502       checkRemove(indexToRemove >= 0);
503       expectedModCount++;
504       removeEntry(indexToRemove);
505       currentIndex = adjustAfterRemove(currentIndex, indexToRemove);
506       indexToRemove = -1;
507     }
508 
509     private void checkForConcurrentModification() {
510       if (modCount != expectedModCount) {
511         throw new ConcurrentModificationException();
512       }
513     }
514   }

638     @SuppressWarnings("unchecked") // keys only contains Ks
639     MapEntry(int index) {
640       this.key = (K) keys[index];
641       this.lastKnownIndex = index;
642     }

649     private void updateLastKnownIndex() {
650       if (lastKnownIndex == -1
651           || lastKnownIndex >= size()
652           || !Objects.equal(key, keys[lastKnownIndex])) {
653         lastKnownIndex = indexOf(key);
654       }
655     }
656 
657     @SuppressWarnings("unchecked") // values only contains Vs
658     @Override
659     public V getValue() {
660       updateLastKnownIndex();
661       return (lastKnownIndex == -1) ? null : (V) values[lastKnownIndex];
662     }
663 
664     @SuppressWarnings("unchecked") // values only contains Vs
665     @Override
666     public V setValue(V value) {
667       updateLastKnownIndex();
668       if (lastKnownIndex == -1) {
669         put(key, value);
670         return null;
671       } else {
672         V old = (V) values[lastKnownIndex];
673         values[lastKnownIndex] = value;
674         return old;
675       }
676     }
677   }

738   /**
739    * Ensures that this {@code CompactHashMap} has the smallest representation in memory, given its
740    * current size.
741    */
742   public void trimToSize() {
743     int size = this.size;
744     if (size < entries.length) {
745       resizeEntries(size);
746     }
747     // size / loadFactor gives the table size of the appropriate load factor,
748     // but that may not be a power of two. We floor it to a power of two by
749     // keeping its highest bit. But the smaller table may have a load factor
750     // larger than what we want; then we want to go to the next power of 2 if we can
751     int minimumTableSize = Math.max(1, Integer.highestOneBit((int) (size / loadFactor)));
752     if (minimumTableSize < MAXIMUM_CAPACITY) {
753       double load = (double) size / minimumTableSize;
754       if (load > loadFactor) {
755         minimumTableSize <<= 1; // increase to next power if possible
756       }
757     }
758 
759     if (minimumTableSize < table.length) {
760       resizeTable(minimumTableSize);
761     }
762   }
763 
764   @Override
765   public void clear() {
766     modCount++;
767     Arrays.fill(keys, 0, size, null);
768     Arrays.fill(values, 0, size, null);
769     Arrays.fill(table, UNSET);
770     Arrays.fill(entries, UNSET);
771     this.size = 0;
772   }

774   /**
775    * The serial form currently mimics Android's java.util.HashMap version, e.g. see
776    * http://omapzoom.org/?p=platform/libcore.git;a=blob;f=luni/src/main/java/java/util/HashMap.java
777    */
778   private void writeObject(ObjectOutputStream stream) throws IOException {
779     stream.defaultWriteObject();
780     stream.writeInt(size);
781     for (int i = 0; i < size; i++) {
782       stream.writeObject(keys[i]);
783       stream.writeObject(values[i]);
784     }
785   }

787   @SuppressWarnings("unchecked")
788   private void readObject(ObjectInputStream stream) throws IOException, ClassNotFoundException {
789     stream.defaultReadObject();
790     init(DEFAULT_SIZE, DEFAULT_LOAD_FACTOR);
791     int elementCount = stream.readInt();
792     for (int i = elementCount; --i >= 0; ) {
793       K key = (K) stream.readObject();
794       V value = (V) stream.readObject();
795       put(key, value);
796     }
797   }
