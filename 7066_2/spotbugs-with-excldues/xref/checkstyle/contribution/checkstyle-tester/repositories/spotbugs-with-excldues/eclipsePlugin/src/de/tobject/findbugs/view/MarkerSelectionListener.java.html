19  package de.tobject.findbugs.view;
20  
21  import org.eclipse.core.resources.IMarker;
22  import org.eclipse.jface.viewers.ISelection;
23  import org.eclipse.ui.ISelectionListener;
24  import org.eclipse.ui.IWorkbenchPart;
25  
26  import de.tobject.findbugs.reporter.MarkerUtil;
27  
28  class MarkerSelectionListener implements ISelectionListener {
29      private final IMarkerSelectionHandler handler;
30  
31      @Override
32      public String toString() {
33          return "MarkerSelectionListener for " + handler.getClass().getSimpleName();
34      }
35  
36      public MarkerSelectionListener(IMarkerSelectionHandler handler) {
37          this.handler = handler;
38      }
39  
40      @Override
41      public void selectionChanged(IWorkbenchPart thePart, ISelection theSelection) {
42          if (thePart == handler || !handler.isVisible()) {
43              return;
44          }
45          IMarker marker = MarkerUtil.getMarkerFromSingleSelection(theSelection);
46          // only handle a null if it's from the bug explorer. this way if the
47          // user
48          // selects a bug pattern or other "folder" in the tree, the annotation
49          // view
50          // and the property view are cleared. BUT if the user clicks into
51          // the editor, or some other view, it stays up. not quite sure what the
52          // best behavior is here but for now this works. -Keiths
53          if (marker == null && !(thePart instanceof BugExplorerView)) {
54              return;
55          }
56          handler.markerSelected(thePart, marker);
57      }
58  }
