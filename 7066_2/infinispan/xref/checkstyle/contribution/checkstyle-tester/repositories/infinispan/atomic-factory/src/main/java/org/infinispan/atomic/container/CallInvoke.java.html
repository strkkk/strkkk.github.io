7   class CallInvoke extends Call {
8       String method;
9       Object[] arguments;
10  
11      public CallInvoke(long id, String m, Object[] args) {
12          super(id);
13          method = m;
14          arguments = args;
15      }
16  
17      @Override
18      public String toString(){
19          String args = " ";
20          for(Object a : arguments){
21              args+=a.toString()+" ";
22          }
23          return super.toString()+"  - INV - "+method+ " ("+args+")";
24      }
25  }
