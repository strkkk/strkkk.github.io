25     @Field(store = Store.YES, analyze = Analyze.NO)
26     private int id;
27  
28     @Field(store = Store.YES, analyze = Analyze.NO)

31  
32     @Field(store = Store.YES, analyze = Analyze.NO)
33     private String name;

35     @Field(store = Store.YES, analyze = Analyze.NO, indexNullAs = Field.DEFAULT_NULL_TOKEN)

37  
38     @Field(store = Store.YES, analyze = Analyze.NO, indexNullAs = Field.DEFAULT_NULL_TOKEN)
39     @NumericField
40     private Integer age;  // yes, not the birth date :)
41  
42     @Field(store = Store.YES, analyze = Analyze.NO)
43     private Gender gender;
44  
45     @IndexedEmbedded(targetElement = AddressHS.class, indexNullAs = Field.DEFAULT_NULL_TOKEN)
46     private List<Address> addresses;
47  
48     /**
49      * This is not indexed!

51     private String notes;

119    @Override
120    public boolean equals(Object o) {
121       if (this == o) return true;
122       if (o == null || getClass() != o.getClass()) return false;
123 
124       UserHS other = (UserHS) o;
125 
126       if (age != null ? !age.equals(other.age) : other.age != null) return false;
127       if (id != other.id) return false;
128       if (accountIds != null ? !accountIds.equals(other.accountIds) : other.accountIds != null) return false;
129       if (addresses != null ? !addresses.equals(other.addresses) : other.addresses != null) return false;
130       if (gender != other.gender) return false;
131       if (name != null ? !name.equals(other.name) : other.name != null) return false;
132       if (surname != null ? !surname.equals(other.surname) : other.surname != null) return false;
133       if (notes != null ? !notes.equals(other.notes) : other.notes != null) return false;
134 
135       return true;
136    }
137 
138    @Override
139    public int hashCode() {
140       int result = id;
141       result = 31 * result + (accountIds != null ? accountIds.hashCode() : 0);
142       result = 31 * result + (name != null ? name.hashCode() : 0);
143       result = 31 * result + (surname != null ? surname.hashCode() : 0);
144       result = 31 * result + (age != null ? age : 0);
145       result = 31 * result + (gender != null ? gender.hashCode() : 0);
146       result = 31 * result + (addresses != null ? addresses.hashCode() : 0);
147       result = 31 * result + (notes != null ? notes.hashCode() : 0);
148       return result;
149    }
