34  /**
35   * The AsyncCacheWriter is a delegating CacheStore that buffers changes and writes them asynchronously to
36   * the underlying CacheStore.
37   * <p/>
38   * Read operations are done synchronously, taking into account the current state of buffered changes.
39   * <p/>
40   * There is no provision for exception handling for problems encountered with the underlying store
41   * during a write operation, and the exception is just logged.
42   * <p/>
43   * When configuring the loader, use the following element:
44   * <p/>
45   * <code> &lt;async enabled="true" /&gt; </code>
46   * <p/>
47   * to define whether cache loader operations are to be asynchronous. If not specified, a cache loader operation is
48   * assumed synchronous and this decorator is not applied.
49   * <p/>
50   * Write operations affecting same key are now coalesced so that only the final state is actually stored.
51   * <p/>
52   *
53   * @author Manik Surtani
54   * @author Galder Zamarre√±o
55   * @author Sanne Grinovero
56   * @author Karsten Blees
57   * @author Mircea Markus
58   * @since 4.0
59   */
60  public class AsyncCacheWriter extends DelegatingCacheWriter {
61     private static final Log log = LogFactory.getLog(AsyncCacheWriter.class);
62     private static final boolean trace = log.isTraceEnabled();
63     private static final AtomicInteger threadId = new AtomicInteger(0);
64  
65     private ExecutorService executor;
66     private Thread coordinator;
67     private int concurrencyLevel;
68     private String cacheName;
69  
70     protected BufferLock stateLock;
71     @GuardedBy("stateLock")
72     protected final AtomicReference<State> state = new AtomicReference<State>();
73     @GuardedBy("stateLock")
74     private boolean stopped;
75  
76     protected AsyncStoreConfiguration asyncConfiguration;
77  
78     public AsyncCacheWriter(CacheWriter delegate) {
79        super(delegate);
80     }
81  
82     @Override
83     public void init(InitializationContext ctx) {
84        super.init(ctx);
85        this.asyncConfiguration = ctx.getConfiguration().async();
86  
87        Cache cache = ctx.getCache();
88        Configuration cacheCfg = cache != null ? cache.getCacheConfiguration() : null;
89        concurrencyLevel = cacheCfg != null ? cacheCfg.locking().concurrencyLevel() : 16;
90        cacheName = cache != null ? cache.getName() : null;
91     }
92  
93     @Override
94     public void start() {
95        log.debugf("Async cache loader starting %s", this);
96        state.set(newState(false, null));
97        stopped = false;
98        stateLock = new BufferLock(asyncConfiguration.modificationQueueSize());
99  
100       // Create a thread pool with unbounded work queue, so that all work is accepted and eventually
101       // executed. A bounded queue could throw RejectedExecutionException and thus lose data.
102       int poolSize = asyncConfiguration.threadPoolSize();
103       executor = new ThreadPoolExecutor(poolSize, poolSize, 120L, TimeUnit.SECONDS, new LinkedBlockingQueue<Runnable>(),
104                                         new ThreadFactory() {
105                                            @Override
106                                            public Thread newThread(Runnable r) {
107                                               Thread t = new Thread(r, "AsyncStoreProcessor-" + cacheName + "-" + threadId.getAndIncrement());
108                                               t.setDaemon(true);
109                                               return t;
110                                            }
111                                         });
112       ((ThreadPoolExecutor) executor).allowCoreThreadTimeOut(true);
113       coordinator = new Thread(new AsyncStoreCoordinator(), "AsyncStoreCoordinator-" + cacheName);
114       coordinator.setDaemon(true);
115       coordinator.start();
116    }
117 
118    @Override
119    public void stop() {
120       if (trace) log.tracef("Stop async store %s", this);
121       stateLock.writeLock(0);
122       stopped = true;
123       stateLock.writeUnlock();
124       try {
125          // It is safe to wait without timeout because the thread pool uses an unbounded work queue (i.e.
126          // all work handed to the pool will be accepted and eventually executed) and AsyncStoreProcessors
127          // decrement the workerThreads latch in a finally block (i.e. even if the back-end store throws
128          // java.lang.Error). The coordinator thread can only block forever if the back-end's write() /
129          // remove() methods block, but this is no different from PassivationManager.stop() being blocked
130          // in a synchronous call to write() / remove().
131          coordinator.join();
132          // The coordinator thread waits for AsyncStoreProcessor threads to count down their latch (nearly
133          // at the end). Thus the threads should have terminated or terminate instantly.
134          executor.shutdown();
135          if (!executor.awaitTermination(1, TimeUnit.SECONDS))
136             log.errorAsyncStoreNotStopped();
137       } catch (InterruptedException e) {
138          log.interruptedWaitingAsyncStorePush(e);
139          Thread.currentThread().interrupt();
140       }
141    }
142 
143    @Override
144    public void write(MarshalledEntry entry) {
145       put(new Store(entry.getKey(), entry), 1);
146    }
147 
148    @Override
149    public boolean delete(Object key) {
150       put(new Remove(key), 1);
151       return true;
152    }
153 
154    protected void applyModificationsSync(List<Modification> mods) throws PersistenceException {
155       for (Modification m : mods) {
156          switch (m.getType()) {
157             case STORE:
158                actual.write(((Store) m).getStoredValue());
159                break;
160             case REMOVE:
161                actual.delete(((Remove) m).getKey());
162                break;
163             default:
164                throw new IllegalArgumentException("Unknown modification type " + m.getType());
165          }
166       }
167    }
168 
169 
170    protected State newState(boolean clear, State next) {
171       ConcurrentMap<Object, Modification> map = CollectionFactory.makeConcurrentMap(64, concurrencyLevel);
172       return new State(clear, map, next);
173    }
174 
175    void assertNotStopped() throws CacheException {
176       if (stopped)
177          throw new CacheException("AsyncCacheWriter stopped; no longer accepting more entries.");
178    }
179 
180    private void put(Modification mod, int count) {
181       stateLock.writeLock(count);
182       try {
183          if (log.isTraceEnabled())
184             log.tracef("Queue modification: %s", mod);
185 
186          assertNotStopped();
187          state.get().put(mod);
188       } finally {
189          stateLock.writeUnlock();
190       }
191    }
192 
193    public AtomicReference<State> getState() {
194       return state;
195    }
196 
197    protected void clearStore() {
198       // No-op, not supported for async
199    }

201    private class AsyncStoreCoordinator implements Runnable {
202 
203       @Override
204       public void run() {
205          LogFactory.pushNDC(cacheName, trace);
206          try {
207             for (;;) {
208                final State s, head, tail;
209                final boolean shouldStop;
210                stateLock.readLock();
211                try {
212                   s = state.get();
213                   shouldStop = stopped;
214                   tail = s.next;
215                   assert tail == null || tail.next == null : "State chain longer than 3 entries!";
216                   head = newState(false, s);
217                   state.set(head);
218                } finally {
219                   stateLock.reset(0);
220                   stateLock.readUnlock();
221                }
222 
223                try {
224                   if (s.clear) {
225                      // clear() must be called synchronously, wait until background threads are done
226                      if (tail != null)
227                         tail.workerThreads.await();
228 
229                      clearStore();
230                   }
231 
232                   final List<Modification> mods = new ArrayList<>(s.modifications.size());
233                   final List<Modification> deferredMods = new ArrayList<>();
234                   if (tail != null && tail.workerThreads.getCount() > 0) {
235                      // sort out modifications that are still in use by tail's AsyncStoreProcessors
236                      for (Map.Entry<Object, Modification> e : s.modifications.entrySet()) {
237                         if (!tail.modifications.containsKey(e.getKey()))
238                            mods.add(e.getValue());
239                         else
240                            deferredMods.add(e.getValue());
241                      }
242                   } else {
243                      mods.addAll(s.modifications.values());
244                   }
245 
246                   // create AsyncStoreProcessors
247                   final List<AsyncStoreProcessor> procs = createProcessors(s, mods);
248                   final List<AsyncStoreProcessor> deferredProcs = createProcessors(s, deferredMods);
249                   s.workerThreads = new CountDownLatch(procs.size() + deferredProcs.size());
250 
251                   // schedule AsyncStoreProcessors that don't conflict with tail's processors
252                   for (AsyncStoreProcessor processor : procs)
253                      executor.execute(processor);
254 
255                   // wait until background threads of previous round are done
256                   if (tail != null) {
257                      tail.workerThreads.await();
258                      s.next = null;
259                   }
260 
261                   // schedule remaining AsyncStoreProcessors
262                   for (AsyncStoreProcessor processor : deferredProcs)
263                      executor.execute(processor);
264 
265                   // if this is the last state to process, wait for background threads, then quit
266                   if (shouldStop) {
267                      s.workerThreads.await();
268                      return;
269                   }
270                } catch (Exception e) {
271                   log.unexpectedErrorInAsyncStoreCoordinator(e);
272                }
273             }
274          } finally {
275             LogFactory.popNDC(trace);
276          }
277       }
278 
279       private List<AsyncStoreProcessor> createProcessors(State state, List<Modification> mods) {
280          List<AsyncStoreProcessor> result = new ArrayList<>();
281          // distribute modifications evenly across worker threads
282          int threads = Math.min(mods.size(), asyncConfiguration.threadPoolSize());
283          if (threads > 0) {
284             // create background threads
285             int start = 0;
286             int quotient = mods.size() / threads;
287             int remainder = mods.size() % threads;
288             for (int i = 0; i < threads; i++) {
289                int end = start + quotient + (i < remainder ? 1 : 0);
290                result.add(new AsyncStoreProcessor(mods.subList(start, end), state));
291                start = end;
292             }
293             assert start == mods.size() : "Thread distribution is broken!";
294          }
295          return result;
296       }
297    }
298 
299    private class AsyncStoreProcessor implements Runnable {
300       private final List<Modification> modifications;
301       private final State myState;
302 
303       AsyncStoreProcessor(List<Modification> modifications, State myState) {
304          this.modifications = modifications;
305          this.myState = myState;
306       }
307 
308       @Override
309       public void run() {
310          try {
311             // try 3 times to store the modifications
312             retryWork(3);
313 
314          } finally {
315             // decrement active worker threads and disconnect myState if this was the last one
316             myState.workerThreads.countDown();
317             if (myState.workerThreads.getCount() == 0 && myState.next == null)
318                for (State s = state.get(); s != null; s = s.next)
319                   if (s.next == myState)
320                      s.next = null;
321          }
322       }
323 
324       private void retryWork(int maxRetries) {
325          for (int attempt = 0; attempt < maxRetries; attempt++) {
326             if (attempt > 0 && log.isDebugEnabled())
327                log.debugf("Retrying due to previous failure. %s attempts left.", maxRetries - attempt);
328 
329             try {
330                AsyncCacheWriter.this.applyModificationsSync(modifications);
331                return;
332             } catch (Exception e) {
333                if (log.isDebugEnabled())
334                   log.debug("Failed to process async modifications", e);
335             }
336          }
337          log.unableToProcessAsyncModifications(maxRetries);
338       }
339    }
