18  package org.apache.tools.ant.taskdefs.optional.extension;
19  
20  import org.apache.tools.ant.BuildException;
21  import org.apache.tools.ant.types.DataType;
22  import org.apache.tools.ant.types.Reference;
23  import org.apache.tools.ant.util.DeweyDecimal;
24  
25  /**
26   * Simple class that represents an Extension and conforms to Ants
27   * patterns.
28   *
29   * @ant.datatype name="extension"
30   */
31  public class ExtensionAdapter extends DataType {
32      /**
33       * The name of the optional package being made available, or required.
34       */
35      private String extensionName;
36  
37      /**
38       * The version number (dotted decimal notation) of the specification
39       * to which this optional package conforms.
40       */
41      private DeweyDecimal specificationVersion;
42  
43      /**
44       * The name of the company or organization that originated the
45       * specification to which this optional package conforms.
46       */
47      private String specificationVendor;
48  
49      /**
50       * The unique identifier of the company that produced the optional
51       * package contained in this JAR file.
52       */
53      private String implementationVendorID;
54  
55      /**
56       * The name of the company or organization that produced this
57       * implementation of this optional package.
58       */
59      private String implementationVendor;
60  
61      /**
62       * The version number (dotted decimal notation) for this implementation
63       * of the optional package.
64       */
65      private DeweyDecimal implementationVersion;
66  
67      /**
68       * The URL from which the most recent version of this optional package
69       * can be obtained if it is not already installed.
70       */
71      private String implementationURL;
72  
73      /**
74       * Set the name of extension.
75       *
76       * @param extensionName the name of extension
77       */
78      public void setExtensionName(final String extensionName) {
79          verifyNotAReference();
80          this.extensionName = extensionName;
81      }
82  
83      /**
84       * Set the specificationVersion of extension.
85       *
86       * @param specificationVersion the specificationVersion of extension
87       */
88      public void setSpecificationVersion(final String specificationVersion) {
89          verifyNotAReference();
90          this.specificationVersion = new DeweyDecimal(specificationVersion);
91      }
92  
93      /**
94       * Set the specificationVendor of extension.
95       *
96       * @param specificationVendor the specificationVendor of extension
97       */
98      public void setSpecificationVendor(final String specificationVendor) {
99          verifyNotAReference();
100         this.specificationVendor = specificationVendor;
101     }
102 
103     /**
104      * Set the implementationVendorID of extension.
105      *
106      * @param implementationVendorID the implementationVendorID of extension
107      */
108     public void setImplementationVendorId(final String implementationVendorID) {
109         verifyNotAReference();
110         this.implementationVendorID = implementationVendorID;
111     }
112 
113     /**
114      * Set the implementationVendor of extension.
115      *
116      * @param implementationVendor the implementationVendor of extension
117      */
118     public void setImplementationVendor(final String implementationVendor) {
119         verifyNotAReference();
120         this.implementationVendor = implementationVendor;
121     }
122 
123     /**
124      * Set the implementationVersion of extension.
125      *
126      * @param implementationVersion the implementationVersion of extension
127      */
128     public void setImplementationVersion(final String implementationVersion) {
129         verifyNotAReference();
130         this.implementationVersion = new DeweyDecimal(implementationVersion);
131     }
132 
133     /**
134      * Set the implementationURL of extension.
135      *
136      * @param implementationURL the implementationURL of extension
137      */
138     public void setImplementationUrl(final String implementationURL) {
139         verifyNotAReference();
140         this.implementationURL = implementationURL;
141     }
142 
143     /**
144      * Makes this instance in effect a reference to another ExtensionAdapter
145      * instance.
146      *
147      * <p>You must not set another attribute or nest elements inside
148      * this element if you make it a reference.</p>
149      *
150      * @param reference the reference to which this instance is associated
151      * @exception BuildException if this instance already has been configured.
152      */
153     public void setRefid(final Reference reference)
154         throws BuildException {
155         if (null != extensionName
156             || null != specificationVersion
157             || null != specificationVendor
158             || null != implementationVersion
159             || null != implementationVendorID
160             || null != implementationVendor
161             || null != implementationURL) {
162             throw tooManyAttributes();
163         }
164         super.setRefid(reference);
165     }
166 
167     private void verifyNotAReference()
168         throws BuildException {
169         if (isReference()) {
170             throw tooManyAttributes();
171         }
172     }
173 
174     /**
175      * Convert this adpater object into an extension object.
176      *
177      * @return the extension object
178      */
179     Extension toExtension()
180         throws BuildException {
181         if (isReference()) {
182             return ((ExtensionAdapter) getCheckedRef()).toExtension();
183         }
184         dieOnCircularReference();
185         if (null == extensionName) {
186             final String message = "Extension is missing name.";
187             throw new BuildException(message);
188         }
189 
190         String specificationVersionString = null;
191         if (null != specificationVersion) {
192             specificationVersionString = specificationVersion.toString();
193         }
194         String implementationVersionString = null;
195         if (null != implementationVersion) {
196             implementationVersionString = implementationVersion.toString();
197         }
198         return new Extension(extensionName,
199                               specificationVersionString,
200                               specificationVendor,
201                               implementationVersionString,
202                               implementationVendor,
203                               implementationVendorID,
204                               implementationURL);
205     }
206 
207     /**
208      * a debug toString method.
209      * @return the extension in a string.
210      * @see java.lang.Object#toString()
211      */
212     public String toString() {
213         return "{" + toExtension().toString() + "}";
214     }
215 }
