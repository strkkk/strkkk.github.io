72  /**
73   * A class loader that loads classes and resources from a collection of
74   * modules, or from a single module where the class loader is a member
75   * of a pool of class loaders.
76   *
77   * <p> The delegation model used by this ClassLoader differs to the regular
78   * delegation model. When requested to load a class then this ClassLoader first
79   * maps the class name to its package name. If there a module defined to the
80   * Loader containing the package then the class loader attempts to load from
81   * that module. If the package is instead defined to a module in a "remote"
82   * ClassLoader then this class loader delegates directly to that class loader.
83   * The map of package name to remote class loader is created based on the
84   * modules read by modules defined to this class loader. If the package is not
85   * local or remote then this class loader will delegate to the parent class
86   * loader. This allows automatic modules (for example) to link to types in the
87   * unnamed module of the parent class loader.
88   *
89   * @see ModuleLayer#defineModulesWithOneLoader
90   * @see ModuleLayer#defineModulesWithManyLoaders
91   */
92  
93  public final class Loader extends SecureClassLoader {

109     private final Map<String, LoadedModule> localPackageToModule;

113         = new ConcurrentHashMap<>();

116     private final Map<ModuleReference, ModuleReader> moduleToReader

122     /**
123      * A module defined/loaded to a {@code Loader}.
124      */
125     private static class LoadedModule {
126         private final ModuleReference mref;
127         private final URL url;          // may be null
128         private final CodeSource cs;
129 
130         LoadedModule(ModuleReference mref) {
131             URL url = null;
132             if (mref.location().isPresent()) {
133                 try {
134                     url = mref.location().get().toURL();
135                 } catch (MalformedURLException | IllegalArgumentException e) { }
136             }
137             this.mref = mref;
138             this.url = url;
139             this.cs = new CodeSource(url, (CodeSigner[]) null);
140         }
141 
142         ModuleReference mref() { return mref; }
143         String name() { return mref.descriptor().name(); }
144         URL location() { return url; }
145         CodeSource codeSource() { return cs; }
146     }

149     /**
150      * Creates a {@code Loader} in a loader pool that loads classes/resources
151      * from one module.
152      */
153     public Loader(ResolvedModule resolvedModule,
154                   LoaderPool pool,
155                   ClassLoader parent)
156     {
157         super("Loader-" + resolvedModule.name(), parent);
158 
159         this.pool = pool;
160         this.parent = parent;
161 
162         ModuleReference mref = resolvedModule.reference();
163         ModuleDescriptor descriptor = mref.descriptor();
164         String mn = descriptor.name();
165         this.nameToModule = Map.of(mn, mref);
166 
167         Map<String, LoadedModule> localPackageToModule = new HashMap<>();
168         LoadedModule lm = new LoadedModule(mref);
169         descriptor.packages().forEach(pn -> localPackageToModule.put(pn, lm));
170         this.localPackageToModule = localPackageToModule;
171 
172         this.acc = AccessController.getContext();
173     }
174 
175     /**
176      * Creates a {@code Loader} that loads classes/resources from a collection
177      * of modules.
178      *
179      * @throws IllegalArgumentException
180      *         If two or more modules have the same package
181      */
182     public Loader(Collection<ResolvedModule> modules, ClassLoader parent) {
183         super(parent);
184 
185         this.pool = null;
186         this.parent = parent;
187 
188         Map<String, ModuleReference> nameToModule = new HashMap<>();
189         Map<String, LoadedModule> localPackageToModule = new HashMap<>();
190         for (ResolvedModule resolvedModule : modules) {
191             ModuleReference mref = resolvedModule.reference();
192             ModuleDescriptor descriptor = mref.descriptor();
193             nameToModule.put(descriptor.name(), mref);
194             descriptor.packages().forEach(pn -> {
195                 LoadedModule lm = new LoadedModule(mref);
196                 if (localPackageToModule.put(pn, lm) != null)
197                     throw new IllegalArgumentException("Package "
198                         + pn + " in more than one module");
199             });
200         }
201         this.nameToModule = nameToModule;
202         this.localPackageToModule = localPackageToModule;
203 
204         this.acc = AccessController.getContext();
205     }

208     /**
209      * Completes initialization of this Loader. This method populates
210      * remotePackageToLoader with the packages of the remote modules, where
211      * "remote modules" are the modules read by modules defined to this loader.
212      *
213      * @param cf the Configuration containing at least modules to be defined to
214      *           this class loader
215      *
216      * @param parentModuleLayers the parent ModuleLayers
217      */
218     public Loader initRemotePackageMap(Configuration cf,
219                                        List<ModuleLayer> parentModuleLayers)
220     {
221         for (String name : nameToModule.keySet()) {
222             ResolvedModule resolvedModule = cf.findModule(name).get();
223             assert resolvedModule.configuration() == cf;
224 
225             for (ResolvedModule other : resolvedModule.reads()) {
226                 String mn = other.name();
227                 ClassLoader loader;
228 
229                 if (other.configuration() == cf) {
230 
231                     // The module reads another module in the newly created
232                     // layer. If all modules are defined to the same class
233                     // loader then the packages are local.
234                     if (pool == null) {
235                         assert nameToModule.containsKey(mn);
236                         continue;
237                     }
238 
239                     loader = pool.loaderFor(mn);
240                     assert loader != null;
241 
242                 } else {
243 
244                     // find the layer for the target module
245                     ModuleLayer layer = parentModuleLayers.stream()
246                         .map(parent -> findModuleLayer(parent, other.configuration()))
247                         .flatMap(Optional::stream)
248                         .findAny()
249                         .orElseThrow(() ->
250                             new InternalError("Unable to find parent layer"));
251 
252                     // find the class loader for the module
253                     // For now we use the platform loader for modules defined to the
254                     // boot loader
255                     assert layer.findModule(mn).isPresent();
256                     loader = layer.findLoader(mn);
257                     if (loader == null)
258                         loader = ClassLoaders.platformClassLoader();
259                 }
260 
261                 // find the packages that are exported to the target module
262                 String target = resolvedModule.name();
263                 ModuleDescriptor descriptor = other.reference().descriptor();
264                 for (ModuleDescriptor.Exports e : descriptor.exports()) {
265                     boolean delegate;
266                     if (e.isQualified()) {
267                         // qualified export in same configuration
268                         delegate = (other.configuration() == cf)
269                                 && e.targets().contains(target);
270                     } else {
271                         // unqualified
272                         delegate = true;
273                     }
274 
275                     if (delegate) {
276                         String pn = e.source();
277                         ClassLoader l = remotePackageToLoader.putIfAbsent(pn, loader);
278                         if (l != null && l != loader) {
279                             throw new IllegalArgumentException("Package "
280                                 + pn + " cannot be imported from multiple loaders");
281                         }
282                     }
283                 }
284             }
285 
286         }
287 
288         return this;
289     }
290 
291     /**
292      * Find the layer corresponding to the given configuration in the tree
293      * of layers rooted at the given parent.
294      */
295     private Optional<ModuleLayer> findModuleLayer(ModuleLayer parent, Configuration cf) {
296         return SharedSecrets.getJavaLangAccess().layers(parent)
297                 .filter(l -> l.configuration() == cf)
298                 .findAny();
299     }

313     /**
314      * Returns a URL to a resource of the given name in a module defined to
315      * this class loader.
316      */
317     @Override
318     protected URL findResource(String mn, String name) throws IOException {
319         ModuleReference mref = (mn != null) ? nameToModule.get(mn) : null;
320         if (mref == null)
321             return null;   // not defined to this class loader
322 
323         // locate resource
324         URL url = null;
325         try {
326             url = AccessController.doPrivileged(
327                 new PrivilegedExceptionAction<URL>() {
328                     @Override
329                     public URL run() throws IOException {
330                         Optional<URI> ouri = moduleReaderFor(mref).find(name);
331                         if (ouri.isPresent()) {
332                             try {
333                                 return ouri.get().toURL();
334                             } catch (MalformedURLException |
335                                      IllegalArgumentException e) { }
336                         }
337                         return null;
338                     }
339                 });
340         } catch (PrivilegedActionException pae) {
341             throw (IOException) pae.getCause();
342         }
343 
344         // check access with permissions restricted by ACC
345         if (url != null && System.getSecurityManager() != null) {
346             try {
347                 URL urlToCheck = url;
348                 url = AccessController.doPrivileged(
349                     new PrivilegedExceptionAction<URL>() {
350                         @Override
351                         public URL run() throws IOException {
352                             return URLClassPath.checkURL(urlToCheck);
353                         }
354                     }, acc);
355             } catch (PrivilegedActionException pae) {
356                 url = null;
357             }
358         }
359 
360         return url;
361     }
362 
363     @Override
364     public URL findResource(String name) {
365         String pn = Resources.toPackageName(name);
366         LoadedModule module = localPackageToModule.get(pn);
367 
368         if (module != null) {
369             try {
370                 URL url = findResource(module.name(), name);
371                 if (url != null
372                     && (name.endsWith(".class")
373                         || url.toString().endsWith("/")
374                         || isOpen(module.mref(), pn))) {
375                     return url;
376                 }
377             } catch (IOException ioe) {
378                 // ignore
379             }
380 
381         } else {
382             for (ModuleReference mref : nameToModule.values()) {
383                 try {
384                     URL url = findResource(mref.descriptor().name(), name);
385                     if (url != null) return url;
386                 } catch (IOException ioe) {
387                     // ignore
388                 }
389             }
390         }
391 
392         return null;
393     }

395     @Override
396     public Enumeration<URL> findResources(String name) throws IOException {
397         return Collections.enumeration(findResourcesAsList(name));
398     }

400     @Override
401     public URL getResource(String name) {
402         Objects.requireNonNull(name);
403 
404         // this loader
405         URL url = findResource(name);
406         if (url != null) {
407             return url;
408         } else {
409             // parent loader
410             return parent.getResource(name);
411         }
412     }
413 
414     @Override
415     public Enumeration<URL> getResources(String name) throws IOException {
416         Objects.requireNonNull(name);
417 
418         // this loader
419         List<URL> urls = findResourcesAsList(name);
420 
421         // parent loader
422         Enumeration<URL> e = parent.getResources(name);
423 
424         // concat the URLs with the URLs returned by the parent
425         return new Enumeration<>() {
426             final Iterator<URL> iterator = urls.iterator();
427             @Override
428             public boolean hasMoreElements() {
429                 return (iterator.hasNext() || e.hasMoreElements());
430             }
431             @Override
432             public URL nextElement() {
433                 if (iterator.hasNext()) {
434                     return iterator.next();
435                 } else {
436                     return e.nextElement();
437                 }
438             }
439         };
440     }

442     @Override
443     public Stream<URL> resources(String name) {
444         Objects.requireNonNull(name);
445         // ordering not specified
446         int characteristics = (Spliterator.NONNULL | Spliterator.IMMUTABLE |
447                                Spliterator.SIZED | Spliterator.SUBSIZED);
448         Supplier<Spliterator<URL>> supplier = () -> {
449             try {
450                 List<URL> urls = findResourcesAsList(name);
451                 return Spliterators.spliterator(urls, characteristics);
452             } catch (IOException e) {
453                 throw new UncheckedIOException(e);
454             }
455         };
456         Stream<URL> s1 = StreamSupport.stream(supplier, characteristics, false);
457         Stream<URL> s2 = parent.resources(name);
458         return Stream.concat(s1, s2);
459     }

461     /**
462      * Finds the resources with the given name in this class loader.
463      */
464     private List<URL> findResourcesAsList(String name) throws IOException {
465         String pn = Resources.toPackageName(name);
466         LoadedModule module = localPackageToModule.get(pn);
467         if (module != null) {
468             URL url = findResource(module.name(), name);
469             if (url != null
470                     && (name.endsWith(".class")
471                     || url.toString().endsWith("/")
472                     || isOpen(module.mref(), pn))) {
473                 return List.of(url);
474             } else {
475                 return Collections.emptyList();
476             }
477         } else {
478             List<URL> urls = new ArrayList<>();
479             for (ModuleReference mref : nameToModule.values()) {
480                 URL url = findResource(mref.descriptor().name(), name);
481                 if (url != null) {
482                     urls.add(url);
483                 }
484             }
485             return urls;
486         }
487     }

492     /**
493      * Finds the class with the specified binary name.
494      */
495     @Override
496     protected Class<?> findClass(String cn) throws ClassNotFoundException {
497         Class<?> c = null;
498         LoadedModule loadedModule = findLoadedModule(cn);
499         if (loadedModule != null)
500             c = findClassInModuleOrNull(loadedModule, cn);
501         if (c == null)
502             throw new ClassNotFoundException(cn);
503         return c;
504     }
505 
506     /**
507      * Finds the class with the specified binary name in a given module.
508      * This method returns {@code null} if the class cannot be found.
509      */
510     @Override
511     protected Class<?> findClass(String mn, String cn) {
512         Class<?> c = null;
513         LoadedModule loadedModule = findLoadedModule(cn);
514         if (loadedModule != null && loadedModule.name().equals(mn))
515             c = findClassInModuleOrNull(loadedModule, cn);
516         return c;
517     }
518 
519     /**
520      * Loads the class with the specified binary name.
521      */
522     @Override
523     protected Class<?> loadClass(String cn, boolean resolve)
524         throws ClassNotFoundException
525     {
526         SecurityManager sm = System.getSecurityManager();
527         if (sm != null) {
528             String pn = packageName(cn);
529             if (!pn.isEmpty()) {
530                 sm.checkPackageAccess(pn);
531             }
532         }
533 
534         synchronized (getClassLoadingLock(cn)) {
535             // check if already loaded
536             Class<?> c = findLoadedClass(cn);
537 
538             if (c == null) {
539 
540                 LoadedModule loadedModule = findLoadedModule(cn);
541 
542                 if (loadedModule != null) {
543 
544                     // class is in module defined to this class loader
545                     c = findClassInModuleOrNull(loadedModule, cn);
546 
547                 } else {
548 
549                     // type in another module or visible via the parent loader
550                     String pn = packageName(cn);
551                     ClassLoader loader = remotePackageToLoader.get(pn);
552                     if (loader == null) {
553                         // type not in a module read by any of the modules
554                         // defined to this loader, so delegate to parent
555                         // class loader
556                         loader = parent;
557                     }
558                     if (loader == null) {
559                         c = BootLoader.loadClassOrNull(cn);
560                     } else {
561                         c = loader.loadClass(cn);
562                     }
563 
564                 }
565             }
566 
567             if (c == null)
568                 throw new ClassNotFoundException(cn);
569 
570             if (resolve)
571                 resolveClass(c);
572 
573             return c;
574         }
575     }

578     /**
579      * Finds the class with the specified binary name if in a module
580      * defined to this ClassLoader.
581      *
582      * @return the resulting Class or {@code null} if not found
583      */
584     private Class<?> findClassInModuleOrNull(LoadedModule loadedModule, String cn) {
585         PrivilegedAction<Class<?>> pa = () -> defineClass(cn, loadedModule);
586         return AccessController.doPrivileged(pa, acc);
587     }
588 
589     /**
590      * Defines the given binary class name to the VM, loading the class
591      * bytes from the given module.
592      *
593      * @return the resulting Class or {@code null} if an I/O error occurs
594      */
595     private Class<?> defineClass(String cn, LoadedModule loadedModule) {
596         ModuleReader reader = moduleReaderFor(loadedModule.mref());
597 
598         try {
599             // read class file
600             String rn = cn.replace('.', '/').concat(".class");
601             ByteBuffer bb = reader.read(rn).orElse(null);
602             if (bb == null) {
603                 // class not found
604                 return null;
605             }
606 
607             try {
608                 return defineClass(cn, bb, loadedModule.codeSource());
609             } finally {
610                 reader.release(bb);
611             }
612 
613         } catch (IOException ioe) {
614             // TBD on how I/O errors should be propagated
615             return null;
616         }
617     }

622     /**
623      * Returns the permissions for the given CodeSource.
624      */
625     @Override
626     protected PermissionCollection getPermissions(CodeSource cs) {
627         PermissionCollection perms = super.getPermissions(cs);
628 
629         URL url = cs.getLocation();
630         if (url == null)
631             return perms;
632 
633         // add the permission to access the resource
634         try {
635             Permission p = url.openConnection().getPermission();
636             if (p != null) {
637                 // for directories then need recursive access
638                 if (p instanceof FilePermission) {
639                     String path = p.getName();
640                     if (path.endsWith(File.separator)) {
641                         path += "-";
642                         p = new FilePermission(path, "read");
643                     }
644                 }
645                 perms.add(p);
646             }
647         } catch (IOException ioe) { }
648 
649         return perms;
650     }

655     /**
656      * Find the candidate module for the given class name.
657      * Returns {@code null} if none of the modules defined to this
658      * class loader contain the API package for the class.
659      */
660     private LoadedModule findLoadedModule(String cn) {
661         String pn = packageName(cn);
662         return pn.isEmpty() ? null : localPackageToModule.get(pn);
663     }
664 
665     /**
666      * Returns the package name for the given class name
667      */
668     private String packageName(String cn) {
669         int pos = cn.lastIndexOf('.');
670         return (pos < 0) ? "" : cn.substring(0, pos);
671     }

674     /**
675      * Returns the ModuleReader for the given module.
676      */
677     private ModuleReader moduleReaderFor(ModuleReference mref) {
678         return moduleToReader.computeIfAbsent(mref, m -> createModuleReader(mref));
679     }
680 
681     /**
682      * Creates a ModuleReader for the given module.
683      */
684     private ModuleReader createModuleReader(ModuleReference mref) {
685         try {
686             return mref.open();
687         } catch (IOException e) {
688             // Return a null module reader to avoid a future class load
689             // attempting to open the module again.
690             return new NullModuleReader();
691         }
692     }

694     /**
695      * A ModuleReader that doesn't read any resources.
696      */
697     private static class NullModuleReader implements ModuleReader {
698         @Override
699         public Optional<URI> find(String name) {
700             return Optional.empty();
701         }
702         @Override
703         public Stream<String> list() {
704             return Stream.empty();
705         }
706         @Override
707         public void close() {
708             throw new InternalError("Should not get here");
709         }
710     }

712     /**
713      * Returns true if the given module opens the given package
714      * unconditionally.
715      *
716      * @implNote This method currently iterates over each of the open
717      * packages. This will be replaced once the ModuleDescriptor.Opens
718      * API is updated.
719      */
720     private boolean isOpen(ModuleReference mref, String pn) {
721         ModuleDescriptor descriptor = mref.descriptor();
722         if (descriptor.isOpen() || descriptor.isAutomatic())
723             return true;
724         for (ModuleDescriptor.Opens opens : descriptor.opens()) {
725             String source = opens.source();
726             if (!opens.isQualified() && source.equals(pn)) {
727                 return true;
728             }
729         }
730         return false;
731     }
