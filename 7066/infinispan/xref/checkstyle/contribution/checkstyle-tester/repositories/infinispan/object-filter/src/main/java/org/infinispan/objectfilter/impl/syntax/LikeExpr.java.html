1   package org.infinispan.objectfilter.impl.syntax;
2   
3   /**
4    * @author anistor@redhat.com
5    * @since 7.0
6    */
7   public final class LikeExpr implements PrimaryPredicateExpr {
8   
9      private final ValueExpr child;
10     private final String pattern;
11  
12     public LikeExpr(ValueExpr child, String pattern) {
13        this.child = child;
14        this.pattern = pattern;
15     }
16  
17     @Override
18     public ValueExpr getChild() {
19        return child;
20     }
21  
22     public String getPattern() {
23        return pattern;
24     }
25  
26     @Override
27     public BooleanExpr acceptVisitor(Visitor visitor) {
28        return visitor.visit(this);
29     }
30  
31     @Override
32     public boolean equals(Object o) {
33        if (this == o) return true;
34        if (o == null || getClass() != o.getClass()) return false;
35        LikeExpr likeExpr = (LikeExpr) o;
36        return pattern.equals(likeExpr.pattern) && child.equals(likeExpr.child);
37     }
38  
39     @Override
40     public int hashCode() {
41        return 31 * child.hashCode() + pattern.hashCode();
42     }
43  
44     @Override
45     public String toString() {
46        return "LIKE(" + child + ", " + pattern + ')';
47     }
48  }
