391       CacheQuery cacheQuery = Search.getSearchManager(cache).getQuery(luceneQuery);
392 
393       int matchCounter = 0;
394       int i = 0;
395 
396       //The implementation is changed to this way as in case of NRT index manager the number of created documents may
397       //differ comparing to the simple configuration.
398       while (true) {
399          try {
400             Explanation found = cacheQuery.explain(i);
401 
402             if (found.isMatch())
403                matchCounter++;
404 
405             i++;
406             if (i >= 10 || matchCounter == 3)
407                 break;
408          } catch(ArrayIndexOutOfBoundsException ex) {
409             break;
410          }
411       }
412 
413       assertEquals(3, matchCounter);
414       StaticTestingErrorHandler.assertAllGood(cache);
415    }

593       ResultIterator eagerIterator = cacheQuery.iterator(new FetchOptions().fetchMode(FetchOptions.FetchMode.EAGER));
594       try {
595          assertEquals(1, countElements(eagerIterator));
596       } finally {
597          eagerIterator.close();
598       }
599       ResultIterator lazyIterator = cacheQuery.iterator(new FetchOptions().fetchMode(FetchOptions.FetchMode.LAZY));
600       try {
601          assertEquals(1, countElements(lazyIterator));
602       } finally {
603          lazyIterator.close();
604       }
605       ResultIterator defaultIterator = cacheQuery.iterator();
606       try {
607          assertEquals(1, countElements(defaultIterator));
608       } finally {
609          defaultIterator.close();
610       }
611       StaticTestingErrorHandler.assertAllGood(cache);
612    }

614    private int countElements(ResultIterator iterator) {
615       int count = 0;
616       while (iterator.hasNext()) {
617          iterator.next();
618          count++;
619       }
620       return count;
621    }

700    private void assertIndexingKnows(Cache<Object, Object> cache, Class... types) {
701       ComponentRegistry cr = cache.getAdvancedCache().getComponentRegistry();
702       SearchIntegrator searchIntegrator = cr.getComponent(SearchIntegrator.class);
703       assertNotNull(searchIntegrator);
704       Map<Class<?>, EntityIndexBinding> indexBindingForEntity = searchIntegrator.unwrap(ExtendedSearchIntegrator.class).getIndexBindings();
705       assertNotNull(indexBindingForEntity);
706       Set<Class<?>> keySet = indexBindingForEntity.keySet();
707       assertEquals(types.length, keySet.size());
708       assertTrue(keySet.containsAll(Arrays.asList(types)));
709    }
