62     /**
63      * Check that the transaction commit/rollback recovers if we receive a StateTransferInProgressException from the remote node
64      */
65     private void doStateTransferInProgressTest(boolean commit, final boolean failOnOriginator) throws Exception {
66        MagicKey k1 = new MagicKey("k1", c1);
67        MagicKey k2 = new MagicKey("k2", c2);
68  
69        tm(c1).begin();
70        c1.put(k1, "v1");
71        c1.put(k2, "v2");
72  
73        // We split the transaction commit in two phases by calling the TransactionCoordinator methods directly
74        TransactionTable txTable = TestingUtil.extractComponent(c1, TransactionTable.class);
75        TransactionCoordinator txCoordinator = TestingUtil.extractComponent(c1, TransactionCoordinator.class);
76  
77        // Execute the prepare on both nodes
78        LocalTransaction localTx = txTable.getLocalTransaction(tm(c1).getTransaction());
79        txCoordinator.prepare(localTx);
80  
81        final CountDownLatch commitLatch = new CountDownLatch(1);
82        Thread worker = new Thread("RehasherSim,StaleLocksWithCommitDuringStateTransferTest") {
83           @Override
84           public void run() {
85              try {
86                 // Before calling commit we block transactions on one of the nodes to simulate a state transfer
87                 final StateTransferLock blockFirst = TestingUtil.extractComponent(failOnOriginator ? c1 : c2, StateTransferLock.class);
88                 final StateTransferLock blockSecond = TestingUtil.extractComponent(failOnOriginator ? c2 : c1, StateTransferLock.class);
89  
90                 try {
91                    blockFirst.acquireExclusiveTopologyLock();
92                    blockSecond.acquireExclusiveTopologyLock();
93  
94                    commitLatch.countDown();
95  
96                    // should be much larger than the lock acquisition timeout
97                    Thread.sleep(1000);
98                 } finally {
99                    blockSecond.releaseExclusiveTopologyLock();
100                   blockFirst.releaseExclusiveTopologyLock();
101                }
102             } catch (Throwable t) {
103                log.errorf(t, "Error blocking/unblocking transactions");
104             }
105          }
106       };
107       worker.start();
108 
109       commitLatch.await(10, TimeUnit.SECONDS);
110 
111       try {
112          // finally commit or rollback the transaction
113          if (commit) {
114             tm(c1).commit();
115          } else {
116             tm(c1).rollback();
117          }
118 
119          // make the transaction manager forget about our tx so that we don't get rollback exceptions in the log
120          tm(c1).suspend();
121       } finally {
122          // don't leak threads
123          worker.join();
124       }
125 
126       // test that we don't leak locks
127       assertNotLocked(c1, k1);
128       assertNotLocked(c2, k1);
129       assertNotLocked(c1, k2);
130       assertNotLocked(c2, k2);
131    }

141    /**
142     * Check that the transaction commit/rollback recovers if the remote node dies during the RPC
143     */
144    private void doTestSuspect(boolean commit) throws Exception {
145       MagicKey k1 = new MagicKey("k1", c1);
146       MagicKey k2 = new MagicKey("k2", c2);
147 
148       tm(c1).begin();
149       c1.put(k1, "v1");
150       c1.put(k2, "v2");
151 
152       // We split the transaction commit in two phases by calling the TransactionCoordinator methods directly
153       TransactionTable txTable = TestingUtil.extractComponent(c1, TransactionTable.class);
154       TransactionCoordinator txCoordinator = TestingUtil.extractComponent(c1, TransactionCoordinator.class);
155 
156       // Execute the prepare on both nodes
157       LocalTransaction localTx = txTable.getLocalTransaction(tm(c1).getTransaction());
158       txCoordinator.prepare(localTx);
159 
160       // Delay the commit on the remote node. Can't used blockNewTransactions because we don't want a StateTransferInProgressException
161       InterceptorChain c2ic = TestingUtil.extractComponent(c2, InterceptorChain.class);
162       c2ic.addInterceptorBefore(new CommandInterceptor() {
163          @Override
164          protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
165             if (command instanceof CommitCommand) {
166                Thread.sleep(3000);
167             }
168             return super.handleDefault(ctx, command);
169          }
170       }, StateTransferInterceptor.class);
171 
172       // Schedule the remote node to stop on another thread since the main thread will be busy with the commit call
173       Thread worker = new Thread("RehasherSim,StaleLocksWithCommitDuringStateTransferTest") {
174          @Override
175          public void run() {
176             try {
177                // should be much larger than the lock acquisition timeout
178                Thread.sleep(1000);
179                manager(c2).stop();
180                // stLock.unblockNewTransactions(1000);
181             } catch (InterruptedException e) {
182                log.errorf(e, "Error stopping cache");
183             }
184          }
185       };
186       worker.start();
187 
188       try {
189          // finally commit or rollback the transaction
190          if (commit) {
191             txCoordinator.commit(localTx, false);
192          } else {
193             txCoordinator.rollback(localTx);
194          }
195 
196          // make the transaction manager forget about our tx so that we don't get rollback exceptions in the log
197          tm(c1).suspend();
198       } finally {
199          // don't leak threads
200          worker.join();
201       }
202 
203       // test that we don't leak locks
204       assertNotLocked(c1, k1);
205       assertNotLocked(c1, k2);
206    }
