7   package io.vavr.match.generator;
8   
9   import io.vavr.match.model.ClassModel;
10  
11  import java.util.*;
12  
13  import static java.util.stream.Collectors.joining;
14  
15  /**
16   * A <em>stateful</em> ImportManager which generates an import section of a Java class file.
17   *
18   * @author Daniel Dietrich
19   */
20  class ImportManager {
21  
22      // properties
23      private final String packageNameOfClass;
24      private List<String> directImports;
25  
26      // mutable state
27      private Map<FQN, Import> imports = new HashMap<>();
28  
29      private ImportManager(String packageNameOfClass, List<String> directImports) {
30          this.packageNameOfClass = packageNameOfClass;
31          this.directImports = directImports;
32      }
33  
34      // directImport FQN("io.vavr", "Match.API") will import "io.vavr.Match.API.Pattern0"
35      // otherwise "io.vavr.Match" is imported and "Match.API.Pattern0" is qualified
36      public static ImportManager forClass(ClassModel classModel, String... directImports) {
37          return new ImportManager(classModel.getPackageName(), reverseSort(directImports));
38      }
39  
40      // used by generator to register non-static imports
41      public String getType(ClassModel classModel) {
42          final FQN fqn = new FQN(classModel.getPackageName(), classModel.getClassName());
43          return getType(fqn, imports, packageNameOfClass, directImports);
44      }
45  
46      public String getType(String packageName, String className) {
47          final FQN fqn = new FQN(packageName, className);
48          return getType(fqn, imports, packageNameOfClass, directImports);
49      }
50  
51      // finally used by generator to get the import section
52      public String getImports() {
53          return optimizeImports(imports.values());
54      }
55  
56      private static String getType(FQN fqn, Map<FQN, Import> imports, String packageNameOfClass, List<String> directImports) {
57          if (fqn.packageName.isEmpty() && !packageNameOfClass.isEmpty()) {
58              throw new IllegalStateException("Can't import class '" + fqn.className + "' located in default package");
59          } else if (fqn.packageName.equals(packageNameOfClass)) {
60              final Import _import = createImport(fqn, directImports);
61              if (_import.type.equals(fqn.className)) {
62                  return fqn.className;
63              } else {
64                  imports.put(fqn, _import);
65                  return _import.type;
66              }
67          } else if (imports.containsKey(fqn)) {
68              return imports.get(fqn).type;
69          } else {
70              final Import _import = createImport(fqn, directImports);
71              imports.put(fqn, _import);
72              return _import.type;
73          }
74      }
75  
76      private static Import createImport(FQN fqn, List<String> directImports) {
77          final String qualifiedName = fqn.qualifiedName();
78          final Optional<String> directImportOption = directImports.stream()
79                  .filter(directImport -> qualifiedName.equals(directImport) || qualifiedName.startsWith(directImport + "."))
80                  .findFirst();
81          if (directImportOption.isPresent()) {
82              final String directImport = directImportOption.get();
83              if (qualifiedName.equals(directImport)) {
84                  final String type = directImport.substring(directImport.lastIndexOf('.') + 1);
85                  return new Import(directImport, type);
86              } else {
87                  final String type = qualifiedName.substring(directImport.length() + 1);
88                  final int index = type.indexOf(".");
89                  final String firstSegment = (index < 0) ? type : type.substring(0, index);
90                  return new Import(directImport + "." + firstSegment, type);
91              }
92          } else {
93              final int index = fqn.className.indexOf(".");
94              final String firstSegment = (index < 0) ? fqn.className : fqn.className.substring(0, index);
95              return new Import(fqn.packageName + "." + firstSegment, fqn.className);
96          }
97      }
98  
99      private static String optimizeImports(Collection<Import> imports) {
100         return imports.stream()
101                 .filter(currentImport -> !currentImport.name.startsWith("java.lang.") ||
102                         imports.stream()
103                                 .filter(otherImport -> !otherImport.equals(currentImport))
104                                 .map(otherImport -> otherImport.type)
105                                 .filter(otherType -> otherType.equals(currentImport.type))
106                                 .findFirst().isPresent())
107                 .map(_import -> _import.name)
108                 .distinct()
109                 .sorted()
110                 .map(s -> "import " + s + ";")
111                 .collect(joining("\n"));
112     }
113 
114     private static List<String> reverseSort(String[] strings) {
115         final String[] copy = new String[strings.length];
116         System.arraycopy(strings, 0, copy, 0, strings.length);
117         Arrays.sort(copy, Comparator.reverseOrder());
118         return Arrays.asList(copy);
119     }
120 
121     private static class FQN {
122 
123         private final String packageName;
124         private final String className;
125 
126         FQN(String packageName, String className) {
127             this.packageName = packageName;
128             this.className = className;
129         }
130 
131         String qualifiedName() {
132             return packageName.isEmpty() ? className : packageName + "." + className;
133         }
134 
135         @Override
136         public boolean equals(Object o) {
137             return (o == this) || (o instanceof FQN && qualifiedName().equals(((FQN) o).qualifiedName()));
138         }
139 
140         @Override
141         public int hashCode() {
142             return qualifiedName().hashCode();
143         }
144 
145         @Override
146         public String toString() {
147             return "FQN(" + packageName + ", " + className + ")";
148         }
149     }
150 
151     private static class Import {
152 
153         final String name;
154         final String type;
155 
156         Import(String name, String type) {
157             this.name = name;
158             this.type = type;
159         }
160 
161         @Override
162         public boolean equals(Object o) {
163             return (o == this) || (o instanceof Import && toString().equals(o.toString()));
164         }
165 
166         @Override
167         public int hashCode() {
168             return toString().hashCode();
169         }
170 
171         @Override
172         public String toString() {
173             return "Import(" + name + ", " + type + ")";
174         }
175     }
