45  public final class ProcessTools {
46      private static final class LineForwarder extends StreamPumper.LinePump {
47          private final PrintStream ps;
48          private final String prefix;
49          LineForwarder(String prefix, PrintStream os) {
50              this.ps = os;
51              this.prefix = prefix;
52          }
53          @Override
54          protected void processLine(String line) {
55              ps.println("[" + prefix + "] " + line);
56          }
57      }

62      /**
63       * <p>Starts a process from its builder.</p>
64       * <span>The default redirects of STDOUT and STDERR are started</span>
65       * @param name The process name
66       * @param processBuilder The process builder
67       * @return Returns the initialized process
68       * @throws IOException
69       */
70      public static Process startProcess(String name,
71                                         ProcessBuilder processBuilder)
72      throws IOException {
73          Process p = null;
74          try {
75              p = startProcess(name, processBuilder, null, -1, TimeUnit.NANOSECONDS);
76          } catch (InterruptedException | TimeoutException e) {
77              // can't ever happen
78          }
79          return p;
80      }

82      /**
83       * <p>Starts a process from its builder.</p>
84       * <span>The default redirects of STDOUT and STDERR are started</span>
85       * <p>
86       * It is possible to wait for the process to get to a warmed-up state
87       * via {@linkplain Predicate} condition on the STDOUT
88       * </p>
89       * @param name The process name
90       * @param processBuilder The process builder
91       * @param linePredicate The {@linkplain Predicate} to use on the STDOUT
92       *                      Used to determine the moment the target app is
93       *                      properly warmed-up.
94       *                      It can be null - in that case the warmup is skipped.
95       * @param timeout The timeout for the warmup waiting
96       * @param unit The timeout {@linkplain TimeUnit}
97       * @return Returns the initialized {@linkplain Process}
98       * @throws IOException
99       * @throws InterruptedException
100      * @throws TimeoutException
101      */
102     public static Process startProcess(String name,
103                                        ProcessBuilder processBuilder,
104                                        final Predicate<String> linePredicate,
105                                        long timeout,
106                                        TimeUnit unit)
107     throws IOException, InterruptedException, TimeoutException {
108         Process p = processBuilder.start();
109         StreamPumper stdout = new StreamPumper(p.getInputStream());
110         StreamPumper stderr = new StreamPumper(p.getErrorStream());
111 
112         stdout.addPump(new LineForwarder(name, System.out));
113         stderr.addPump(new LineForwarder(name, System.err));
114         final Phaser phs = new Phaser(1);
115         if (linePredicate != null) {
116             stdout.addPump(new StreamPumper.LinePump() {
117                 @Override
118                 protected void processLine(String line) {
119                     if (linePredicate.test(line)) {
120                         if (phs.getRegisteredParties() > 0) {
121                             phs.arriveAndDeregister();
122                         }
123                     }
124                 }
125             });
126         }
127         Future<Void> stdoutTask = stdout.process();
128         Future<Void> stderrTask = stderr.process();
129 
130         try {
131             if (timeout > -1) {
132                 phs.awaitAdvanceInterruptibly(0, timeout, unit);
133             }
134         } catch (TimeoutException | InterruptedException e) {
135             System.err.println("Failed to start a process (thread dump follows)");
136             for(Map.Entry<Thread, StackTraceElement[]> s : Thread.getAllStackTraces().entrySet()) {
137                 printStack(s.getKey(), s.getValue());
138             }
139             stdoutTask.cancel(true);
140             stderrTask.cancel(true);
141             throw e;
142         }
143 
144         return p;
145     }

148      * Pumps stdout and stderr from running the process into a String.
149      *
150      * @param processBuilder
151      *            ProcessHandler to run.
152      * @return Output from process.
153      * @throws IOException

157             throws IOException {
158         return getOutput(processBuilder.start());

161     /**
162      * Pumps stdout and stderr the running process into a String.
163      *
164      * @param process
165      *            Process to pump.
166      * @return Output from process.
167      * @throws IOException
168      *             If an I/O error occurs.
169      */
170     public static OutputBuffer getOutput(Process process) throws IOException {
171         ByteArrayOutputStream stderrBuffer = new ByteArrayOutputStream();
172         ByteArrayOutputStream stdoutBuffer = new ByteArrayOutputStream();
173         StreamPumper outPumper = new StreamPumper(process.getInputStream(),
174                 stdoutBuffer);
175         StreamPumper errPumper = new StreamPumper(process.getErrorStream(),
176                 stderrBuffer);
177 
178         Future<Void> outTask = outPumper.process();
179         Future<Void> errTask = errPumper.process();
180 
181         try {
182             process.waitFor();
183             outTask.get();
184             errTask.get();
185         } catch (InterruptedException e) {
186             Thread.currentThread().interrupt();
187             return null;
188         } catch (ExecutionException e) {
189             throw new IOException(e);
190         }
191 
192         return new OutputBuffer(stdoutBuffer.toString(),
193                 stderrBuffer.toString());
194     }

196     /**
197      * Get the process id of the current running Java process
198      *
199      * @return Process id
200      */
201     public static int getProcessId() throws Exception {
202 
203         // Get the current process id using a reflection hack
204         RuntimeMXBean runtime = ManagementFactory.getRuntimeMXBean();
205         Field jvm = runtime.getClass().getDeclaredField("jvm");
206 
207         jvm.setAccessible(true);
208         VMManagement mgmt = (sun.management.VMManagement) jvm.get(runtime);
209 
210         Method pid_method = mgmt.getClass().getDeclaredMethod("getProcessId");
211 
212         pid_method.setAccessible(true);
213 
214         int pid = (Integer) pid_method.invoke(mgmt);
215 
216         return pid;
217     }

219     /**
220      * Get platform specific VM arguments (e.g. -d64 on 64bit Solaris)
221      *
222      * @return String[] with platform specific arguments, empty if there are
223      *         none
224      */
225     public static String[] getPlatformSpecificVMArgs() {
226         String osName = System.getProperty("os.name");
227         String dataModel = System.getProperty("sun.arch.data.model");
228 
229         if (osName.equals("SunOS") && dataModel.equals("64")) {
230             return new String[] { "-d64" };
231         }
232 
233         return new String[] {};
234     }

236     /**
237      * Create ProcessBuilder using the java launcher from the jdk to be tested
238      * and with any platform specific arguments prepended
239      */
240     public static ProcessBuilder createJavaProcessBuilder(String... command)
241             throws Exception {
242         String javapath = JDKToolFinder.getJDKTool("java");
243 
244         ArrayList<String> args = new ArrayList<>();
245         args.add(javapath);
246         Collections.addAll(args, getPlatformSpecificVMArgs());
247         Collections.addAll(args, command);
248 
249         // Reporting
250         StringBuilder cmdLine = new StringBuilder();
251         for (String cmd : args)
252             cmdLine.append(cmd).append(' ');
253         System.out.println("Command line: [" + cmdLine.toString() + "]");
254 
255         return new ProcessBuilder(args.toArray(new String[args.size()]));
256     }

258     private static void printStack(Thread t, StackTraceElement[] stack) {
259         System.out.println("\t" +  t +
260                            " stack: (length = " + stack.length + ")");
261         if (t != null) {
262             for (StackTraceElement stack1 : stack) {
263                 System.out.println("\t" + stack1);
264             }
265             System.out.println();
266         }
267     }

269     /**
270      * Executes a test jvm process, waits for it to finish and returns the process output.
271      * The default jvm options from jtreg, test.vm.opts and test.java.opts, are added.
272      * The java from the test.jdk is used to execute the command.
273      *
274      * The command line will be like:
275      * {test.jdk}/bin/java {test.vm.opts} {test.java.opts} cmds
276      *
277      * @param cmds User specifed arguments.
278      * @return The output from the process.
279      */
280     public static OutputAnalyzer executeTestJvm(String... cmds) throws Throwable {
281         ProcessBuilder pb = createJavaProcessBuilder(Utils.addTestJavaOpts(cmds));
282         return executeProcess(pb);
283     }

285     /**
286      * Executes a process, waits for it to finish and returns the process output.
287      * @param pb The ProcessBuilder to execute.
288      * @return The output from the process.
289      */
290     public static OutputAnalyzer executeProcess(ProcessBuilder pb) throws Throwable {
291         OutputAnalyzer output = null;
292         try {
293             output = new OutputAnalyzer(pb.start());
294             return output;
295         } catch (Throwable t) {
296             System.out.println("executeProcess() failed: " + t);
297             throw t;
298         } finally {
299             System.out.println(getProcessLog(pb, output));
300         }
301     }
302 
303     /**
304      * Executes a process, waits for it to finish and returns the process output.
305      * @param cmds The command line to execute.
306      * @return The output from the process.
307      */
308     public static OutputAnalyzer executeProcess(String... cmds) throws Throwable {
309         return executeProcess(new ProcessBuilder(cmds));
310     }

312     /**
313      * Used to log command line, stdout, stderr and exit code from an executed process.
314      * @param pb The executed process.
315      * @param output The output from the process.
316      */
317     public static String getProcessLog(ProcessBuilder pb, OutputAnalyzer output) {
318         String stderr = output == null ? "null" : output.getStderr();
319         String stdout = output == null ? "null" : output.getStdout();
320         String exitValue = output == null ? "null": Integer.toString(output.getExitValue());
321         StringBuilder logMsg = new StringBuilder();
322         final String nl = System.getProperty("line.separator");
323         logMsg.append("--- ProcessLog ---" + nl);
324         logMsg.append("cmd: " + getCommandLine(pb) + nl);
325         logMsg.append("exitvalue: " + exitValue + nl);
326         logMsg.append("stderr: " + stderr + nl);
327         logMsg.append("stdout: " + stdout + nl);
328         return logMsg.toString();
329     }

331     /**
332      * @return The full command line for the ProcessBuilder.
333      */
334     public static String getCommandLine(ProcessBuilder pb) {
335         if (pb == null) {
336             return "null";
337         }
338         StringBuilder cmd = new StringBuilder();
339         for (String s : pb.command()) {
340             cmd.append(s).append(" ");
341         }
342         return cmd.toString().trim();
343     }
