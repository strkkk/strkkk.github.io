126     @Override
127     public void visitAfter(JavaClass obj) {
128         if (!obj.isClass()) {
129             return;
130         }
131         if ("java.lang.Object".equals(getDottedClassName())) {
132             return;
133         }
134         int accessFlags = obj.getAccessFlags();
135         if ((accessFlags & Const.ACC_INTERFACE) != 0) {
136             return;
137         }
138         visibleOutsidePackage = obj.isPublic() || obj.isProtected();
139 
140         String whereEqual = getDottedClassName();
141         boolean inheritedHashCodeIsFinal = false;
142         boolean inheritedEqualsIsFinal = false;
143         boolean inheritedEqualsIsAbstract = false;
144         boolean inheritedEqualsFromAbstractClass = false;
145         XMethod inheritedEquals = null;
146         if (!hasEqualsObject) {
147             XClass we = Lookup.findImplementor(getXClass(), "equals", "(Ljava/lang/Object;)Z", false, bugReporter);
148             if (we == null || we.equals(getXClass())) {
149                 whereEqual = "java.lang.Object";
150             } else {
151                 inheritedEqualsFromAbstractClass = we.isAbstract();
152                 whereEqual = we.getClassDescriptor().getDottedClassName();
153                 inheritedEquals = we.findMethod("equals", "(Ljava/lang/Object;)Z", false);
154                 if (inheritedEquals != null) {
155                     inheritedEqualsIsFinal = inheritedEquals.isFinal();
156                     inheritedEqualsIsAbstract = inheritedEquals.isAbstract();
157                 }
158             }
159         }
160         boolean usesDefaultEquals = "java.lang.Object".equals(whereEqual);
161         String whereHashCode = getDottedClassName();
162         if (!hasHashCode) {
163             XClass wh = Lookup.findSuperImplementor(getXClass(), "hashCode", "()I", false, bugReporter);
164             if (wh == null) {
165                 whereHashCode = "java.lang.Object";
166             } else {
167                 whereHashCode = wh.getClassDescriptor().getDottedClassName();
168                 XMethod m = wh.findMethod("hashCode", "()I", false);
169                 if (m != null && m.isFinal()) {
170                     inheritedHashCodeIsFinal = true;
171                 }
172             }
173         }
174         boolean usesDefaultHashCode = "java.lang.Object".equals(whereHashCode);
175         /*
176         if (false && (usesDefaultEquals || usesDefaultHashCode)) {
177             try {
178                 if (Repository.implementationOf(obj, "java/util/Set") || Repository.implementationOf(obj, "java/util/List")
179                         || Repository.implementationOf(obj, "java/util/Map")) {
180                     // System.out.println(getDottedClassName() + " uses default
181                     // hashCode or equals");
182                 }
183             } catch (ClassNotFoundException e) {
184                 // e.printStackTrace();
185             }
186         }
187          */
188 
189         if (!hasEqualsObject && !hasEqualsSelf && hasEqualsOther) {
190             BugInstance bug = new BugInstance(this, usesDefaultEquals ? "EQ_OTHER_USE_OBJECT" : "EQ_OTHER_NO_OBJECT",
191                     NORMAL_PRIORITY).addClass(this).addMethod(equalsOtherMethod).addClass(equalsOtherClass);
192             bugReporter.reportBug(bug);
193         }
194         if (!hasEqualsObject && hasEqualsSelf) {
195 
196             if (usesDefaultEquals) {
197                 int priority = HIGH_PRIORITY;
198                 if (usesDefaultHashCode || obj.isAbstract()) {
199                     priority++;
200                 }
201                 if (!visibleOutsidePackage) {
202                     priority++;
203                 }
204                 String bugPattern = "EQ_SELF_USE_OBJECT";
205 
206                 BugInstance bug = new BugInstance(this, bugPattern, priority).addClass(getDottedClassName());
207                 if (equalsMethod != null) {
208                     bug.addMethod(equalsMethod);
209                 }
210                 bugReporter.reportBug(bug);
211             } else {
212                 int priority = NORMAL_PRIORITY;
213                 if (hasFields) {
214                     priority--;
215                 }
216                 if (obj.isAbstract()) {
217                     priority++;
218                 }
219                 String bugPattern = "EQ_SELF_NO_OBJECT";
220                 String superclassName = obj.getSuperclassName();
221                 if ("java.lang.Enum".equals(superclassName)) {
222                     bugPattern = "EQ_DONT_DEFINE_EQUALS_FOR_ENUM";
223                     priority = HIGH_PRIORITY;
224                 }
225                 BugInstance bug = new BugInstance(this, bugPattern, priority).addClass(getDottedClassName());
226                 if (equalsMethod != null) {
227                     bug.addMethod(equalsMethod);
228                 }
229                 bugReporter.reportBug(bug);
230             }
231         }
232 
233         // System.out.println("Class " + getDottedClassName());
234         // System.out.println("usesDefaultEquals: " + usesDefaultEquals);
235         // System.out.println("hasHashCode: : " + hasHashCode);
236         // System.out.println("usesDefaultHashCode: " + usesDefaultHashCode);
237         // System.out.println("hasEquals: : " + hasEqualsObject);
238         // System.out.println("hasCompareToObject: : " + hasCompareToObject);
239         // System.out.println("hasCompareToSelf: : " + hasCompareToSelf);
240 
241         if ((hasCompareToObject || hasCompareToSelf) && usesDefaultEquals) {
242             BugInstance bug = new BugInstance(this, "EQ_COMPARETO_USE_OBJECT_EQUALS", obj.isAbstract() ? Priorities.LOW_PRIORITY
243                     : Priorities.NORMAL_PRIORITY).addClass(this);
244             if (compareToSelfMethod != null) {
245                 bug.addMethod(compareToSelfMethod);
246             } else {
247                 bug.addMethod(compareToObjectMethod);
248             }
249             bugReporter.reportBug(bug);
250         }
251         if (!hasCompareToObject && !hasCompareToBridgeMethod && hasCompareToSelf) {
252             if (!extendsObject) {
253                 bugReporter.reportBug(new BugInstance(this, "CO_SELF_NO_OBJECT", NORMAL_PRIORITY).addClass(getDottedClassName())
254                         .addMethod(compareToMethod));
255             }
256         }
257 
258         // if (!hasFields) return;
259         if (hasHashCode && !hashCodeIsAbstract && !(hasEqualsObject || hasEqualsSelf)) {
260             int priority = LOW_PRIORITY;
261             if (usesDefaultEquals) {
262                 bugReporter.reportBug(new BugInstance(this, "HE_HASHCODE_USE_OBJECT_EQUALS", priority).addClass(
263                         getDottedClassName()).addMethod(hashCodeMethod));
264             } else if (!inheritedEqualsIsFinal) {
265                 bugReporter.reportBug(new BugInstance(this, "HE_HASHCODE_NO_EQUALS", priority).addClass(getDottedClassName())
266                         .addMethod(hashCodeMethod));
267             }
268         }
269         if (equalsObjectIsAbstract) {
270             // no errors reported
271         } else if (!hasHashCode && (hasEqualsObject || hasEqualsSelf)) {
272             EqualsKindSummary.KindOfEquals equalsKind = AnalysisContext.currentAnalysisContext().getEqualsKindSummary()
273                     .get(new ClassAnnotation(obj.getClassName()));
274             if (equalsKind == EqualsKindSummary.KindOfEquals.ALWAYS_FALSE) {
275                 return;
276             }
277             if (usesDefaultHashCode) {
278                 int priority = HIGH_PRIORITY;
279                 if (equalsMethodIsInstanceOfEquals) {
280                     priority += 2;
281                 } else if (obj.isAbstract() || !hasEqualsObject) {
282                     priority++;
283                 }
284                 if (priority == HIGH_PRIORITY) {
285                     nonHashableClasses.add(getDottedClassName());
286                 }
287                 if (!visibleOutsidePackage) {
288                     priority++;
289                 }
290                 BugInstance bug = new BugInstance(this, "HE_EQUALS_USE_HASHCODE", priority).addClass(getDottedClassName());
291                 if (equalsMethod != null) {
292                     bug.addMethod(equalsMethod);
293                 }
294                 bugReporter.reportBug(bug);
295             } else if (!inheritedHashCodeIsFinal && !whereHashCode.startsWith("java.util.Abstract")) {
296                 int priority = LOW_PRIORITY;
297 
298                 if (hasEqualsObject && inheritedEqualsIsAbstract) {
299                     priority++;
300                 }
301                 if (hasFields) {
302                     priority--;
303                 }
304                 if (equalsMethodIsInstanceOfEquals || !hasEqualsObject) {
305                     priority += 2;
306                 } else if (obj.isAbstract()) {
307                     priority++;
308                 }
309                 BugInstance bug = new BugInstance(this, "HE_EQUALS_NO_HASHCODE", priority).addClass(getDottedClassName());
310                 if (equalsMethod != null) {
311                     bug.addMethod(equalsMethod);
312                 }
313                 bugReporter.reportBug(bug);
314             }
315         }
316         if (!hasHashCode && !hasEqualsObject && !hasEqualsSelf && !usesDefaultEquals && usesDefaultHashCode && !obj.isAbstract()
317                 && inheritedEqualsFromAbstractClass) {
318             BugInstance bug = new BugInstance(this, "HE_INHERITS_EQUALS_USE_HASHCODE", NORMAL_PRIORITY)
319             .addClass(getDottedClassName());
320             if (equalsMethod != null) {
321                 bug.addMethod(equalsMethod);
322             }
323             bugReporter.reportBug(bug);
324         }
325         if (!hasEqualsObject && !hasEqualsSelf && !usesDefaultEquals && !obj.isAbstract() && hasFields && inheritedEquals != null
326                 && !inheritedEqualsIsFinal && !inheritedEqualsFromAbstractClass
327                 && !inheritedEquals.getClassDescriptor().getSimpleName().contains("Abstract")
328                 && !"java/lang/Enum".equals(inheritedEquals.getClassDescriptor().getClassName())) {
329 
330             BugInstance bug = new BugInstance(this, "EQ_DOESNT_OVERRIDE_EQUALS", NORMAL_PRIORITY);
331 
332             // create annotation pointing to this class source line 1, otherwise the primary annotation shows parent class
333             SourceLineAnnotation sourceLine = new SourceLineAnnotation(getDottedClassName(), obj.getSourceFileName(), 1, 1, 0, 0);
334             bug.addClass(getDottedClassName()).add(sourceLine);
335             bug.addMethod(inheritedEquals).describe(MethodAnnotation.METHOD_DID_YOU_MEAN_TO_OVERRIDE);
336             bugReporter.reportBug(bug);
337         }
338     }

392     @Override
393     public void visit(Method obj) {
394 
395         int accessFlags = obj.getAccessFlags();
396         if ((accessFlags & Const.ACC_STATIC) != 0) {
397             return;
398         }
399         String name = obj.getName();
400         String sig = obj.getSignature();
401         if ((accessFlags & Const.ACC_ABSTRACT) != 0) {
402             if ("equals".equals(name) && sig.equals("(L" + getClassName() + ";)Z")) {
403                 bugReporter.reportBug(new BugInstance(this, "EQ_ABSTRACT_SELF", LOW_PRIORITY).addClass(getDottedClassName()));
404                 return;
405             } else if ("compareTo".equals(name) && sig.equals("(L" + getClassName() + ";)I")) {
406                 bugReporter.reportBug(new BugInstance(this, "CO_ABSTRACT_SELF", LOW_PRIORITY).addClass(getDottedClassName()));
407                 return;
408             }
409         }
410         boolean sigIsObject = "(Ljava/lang/Object;)Z".equals(sig);
411         if ("hashCode".equals(name) && "()I".equals(sig)) {
412             hasHashCode = true;
413             if (obj.isAbstract()) {
414                 hashCodeIsAbstract = true;
415             }
416             hashCodeMethod = MethodAnnotation.fromVisitedMethod(this);
417             // System.out.println("Found hashCode for " + betterClassName);
418         } else if (obj.isPublic() && "equals".equals(name)) {
419             Matcher m = predicateOverAnInstance.matcher(sig);
420             if (m.matches()) {
421                 if (sigIsObject) {
422                     equalsMethod = MethodAnnotation.fromVisitedMethod(this);
423                     hasEqualsObject = true;
424                     if (obj.isAbstract()) {
425                         equalsObjectIsAbstract = true;
426                     } else if (!obj.isNative()) {
427                         Code code = obj.getCode();
428                         byte[] codeBytes = code.getCode();
429                         if (codeBytes.length == 9) {
430                             int op0 = opcode(codeBytes, 0);
431                             int op1 = opcode(codeBytes, 1);
432                             int op2 = opcode(codeBytes, 2);
433                             int op5 = opcode(codeBytes, 5);
434                             int op6 = opcode(codeBytes, 6);
435                             int op7 = opcode(codeBytes, 7);
436                             int op8 = opcode(codeBytes, 8);
437                             if ((op0 == Const.ALOAD_0 && op1 == Const.ALOAD_1 || op0 == Const.ALOAD_1 && op1 == Const.ALOAD_0)
438                                     && (op2 == Const.IF_ACMPEQ || op2 == Const.IF_ACMPNE) && (op5 == Const.ICONST_0 || op5 == Const.ICONST_1)
439                                     && op6 == Const.IRETURN && (op7 == Const.ICONST_0 || op7 == Const.ICONST_1) && op8 == Const.IRETURN) {
440                                 equalsMethodIsInstanceOfEquals = true;
441                             }
442                         } else if (codeBytes.length == 11) {
443                             int op0 = opcode(codeBytes, 0);
444                             int op1 = opcode(codeBytes, 1);
445                             int op2 = opcode(codeBytes, 2);
446                             int op5 = opcode(codeBytes, 5);
447                             int op6 = opcode(codeBytes, 6);
448                             int op9 = opcode(codeBytes, 9);
449                             int op10 = opcode(codeBytes, 10);
450                             if ((op0 == Const.ALOAD_0 && op1 == Const.ALOAD_1 || op0 == Const.ALOAD_1 && op1 == Const.ALOAD_0)
451                                     && (op2 == Const.IF_ACMPEQ || op2 == Const.IF_ACMPNE) && (op5 == Const.ICONST_0 || op5 == Const.ICONST_1)
452                                     && op6 == Const.GOTO && (op9 == Const.ICONST_0 || op9 == Const.ICONST_1) && op10 == Const.IRETURN) {
453                                 equalsMethodIsInstanceOfEquals = true;
454                             }
455 
456                         } else if ((codeBytes.length == 5 && (codeBytes[1] & 0xff) == Const.INSTANCEOF)
457                                 || (codeBytes.length == 15 && (codeBytes[1] & 0xff) == Const.INSTANCEOF && (codeBytes[11] & 0xff) == Const.INVOKESPECIAL)) {
458                             equalsMethodIsInstanceOfEquals = true;
459                         }
460                     }
461                 } else if (sig.equals("(L" + getClassName() + ";)Z")) {
462                     hasEqualsSelf = true;
463                     if (equalsMethod == null) {
464                         equalsMethod = MethodAnnotation.fromVisitedMethod(this);
465                     }
466                 } else {
467                     String arg = m.group(1);
468                     if (getSuperclassName().equals(arg)) {
469                         JavaClass findSuperImplementor = Lookup.findSuperDefiner(getThisClass(), name, sig, bugReporter);
470                         if (findSuperImplementor == null) {
471                             hasEqualsOther = true;
472                             equalsOtherMethod = MethodAnnotation.fromVisitedMethod(this);
473                             equalsOtherClass = DescriptorFactory.createClassDescriptor(arg);
474                         }
475                     }
476 
477                 }
478             }
479         } else if ("compareTo".equals(name) && sig.endsWith(")I") && !obj.isStatic()) {
480             MethodAnnotation tmp = MethodAnnotation.fromVisitedMethod(this);
481             if (BCELUtil.isSynthetic(obj)) {
482                 hasCompareToBridgeMethod = true;
483             }
484             if ("(Ljava/lang/Object;)I".equals(sig)) {
485                 hasCompareToObject = true;
486                 compareToObjectMethod = compareToMethod = tmp;
487             } else if (sig.equals("(L" + getClassName() + ";)I")) {
488                 hasCompareToSelf = true;
489                 compareToSelfMethod = compareToMethod = tmp;
490             }
491         }
492     }

494     Method findMethod(JavaClass clazz, String name, String sig) {
495         Method[] m = clazz.getMethods();
496         for (Method aM : m) {
497             if (aM.getName().equals(name) && aM.getSignature().equals(sig)) {
498                 return aM;
499             }
500         }
501         return null;
502     }

504     @Override
505     public void sawOpcode(int seen) {
506         if (seen == Const.INVOKEVIRTUAL || seen == Const.INVOKEINTERFACE) {
507             String className = getClassConstantOperand();
508             if ("java/util/Map".equals(className) || "java/util/HashMap".equals(className)
509                     || "java/util/LinkedHashMap".equals(className) || "java/util/concurrent/ConcurrentHashMap".equals(className)
510                     || className.contains("Hash")
511                     && Subtypes2.instanceOf(ClassName.toDottedClassName(className), "java.util.Map")) {
512                 if ("put".equals(getNameConstantOperand())
513                         && "(Ljava/lang/Object;Ljava/lang/Object;)Ljava/lang/Object;".equals(getSigConstantOperand())
514                         && stack.getStackDepth() >= 3) {
515                     check(1);
516                 } else if (("get".equals(getNameConstantOperand()) || "remove".equals(getNameConstantOperand()))
517                         && getSigConstantOperand().startsWith("(Ljava/lang/Object;)") && stack.getStackDepth() >= 2) {
518                     check(0);
519                 }
520             } else if ("java/util/Set".equals(className) || "java/util/HashSet".equals(className) || className.contains("Hash")
521                     && Subtypes2.instanceOf(ClassName.toDottedClassName(className), "java.util.Set")) {
522                 if ("add".equals(getNameConstantOperand()) || "contains".equals(getNameConstantOperand())
523                         || "remove".equals(getNameConstantOperand()) && "(Ljava/lang/Object;)Z".equals(getSigConstantOperand())
524                         && stack.getStackDepth() >= 2) {
525                     check(0);
526                 }
527             }
528         }
529     }

531     private void check(int pos) {
532         OpcodeStack.Item item = stack.getStackItem(pos);
533         JavaClass type = null;
534 
535         try {
536             type = item.getJavaClass();
537         } catch (ClassNotFoundException e) {
538             AnalysisContext.reportMissingClass(e);
539         }
540         if (type == null) {
541             return;
542         }
543         String typeName = type.getClassName();
544         if (typeName.startsWith("java.lang")) {
545             return;
546         }
547         int priority = NORMAL_PRIORITY;
548 
549         OpcodeStack.Item collection = stack.getStackItem(PreorderVisitor.getNumberArguments(getSigConstantOperand()));
550         String collectionSignature = collection.getSignature();
551         if (collectionSignature.indexOf("Tree") >= 0
552                 || collectionSignature.indexOf("Sorted") >= 0
553                 || collectionSignature.indexOf("SkipList") >= 0 ) {
554             return;
555         }
556 
557         if (collectionSignature.indexOf("Hash") >= 0) {
558             priority--;
559         }
560         if (!AnalysisContext.currentAnalysisContext()/* .getSubtypes() */.isApplicationClass(type)) {
561             priority++;
562         }
563 
564         if (type.isAbstract() || type.isInterface()) {
565             priority++;
566         }
567         potentialBugs.put(
568                 type.getClassName(),
569                 new BugInstance(this, "HE_USE_OF_UNHASHABLE_CLASS", priority).addClassAndMethod(this)
570                 .addTypeOfNamedClass(type.getClassName()).describe(TypeAnnotation.UNHASHABLE_ROLE).addCalledMethod(this)
571                 .addSourceLine(this));
572     }

574     @CheckForNull
575     @DottedClassName
576     String findHashedClassInSignature(String sig) {
577         Matcher m = mapPattern.matcher(sig);
578         if (m.find()) {
579             return m.group(1).replace('/', '.');
580         }
581         m = hashTablePattern.matcher(sig);
582         if (m.find()) {
583             return m.group(1).replace('/', '.');
584         }
585 
586         m = setPattern.matcher(sig);
587         if (m.find()) {
588             return m.group(1).replace('/', '.');
589         }
590         return null;
591 
592     }

594     @Override
595     public void visit(Signature obj) {
596         if (!isApplicationClass) {
597             return;
598         }
599 
600         String sig = obj.getSignature();
601         String className = findHashedClassInSignature(sig);
602         if (className == null) {
603             return;
604         }
605         if (className.startsWith("java.lang")) {
606             return;
607         }
608         JavaClass type = null;
609 
610         try {
611             type = Repository.lookupClass(className);
612         } catch (ClassNotFoundException e) {
613             AnalysisContext.reportMissingClass(e);
614         }
615         if (type == null) {
616             return;
617         }
618 
619         int priority = NORMAL_PRIORITY;
620         if (sig.indexOf("Hash") >= 0) {
621             priority--;
622         }
623         if (type.isAbstract() || type.isInterface()) {
624             priority++;
625         }
626         if (!AnalysisContext.currentAnalysisContext()/* .getSubtypes() */.isApplicationClass(type)) {
627             priority++;
628         }
629 
630         BugInstance bug = null;
631 
632         if (visitingField()) {
633             bug = new BugInstance(this, "HE_SIGNATURE_DECLARES_HASHING_OF_UNHASHABLE_CLASS", priority).addClass(this)
634                     .addVisitedField(this).addTypeOfNamedClass(className).describe(TypeAnnotation.UNHASHABLE_ROLE);
635         } else if (visitingMethod()) {
636             bug = new BugInstance(this, "HE_SIGNATURE_DECLARES_HASHING_OF_UNHASHABLE_CLASS", priority).addClassAndMethod(this)
637                     .addTypeOfNamedClass(className).describe(TypeAnnotation.UNHASHABLE_ROLE);
638         } else {
639             bug = new BugInstance(this, "HE_SIGNATURE_DECLARES_HASHING_OF_UNHASHABLE_CLASS", priority).addClass(this)
640                     .addClass(this).addTypeOfNamedClass(className).describe(TypeAnnotation.UNHASHABLE_ROLE);
641         }
642         potentialBugs.put(className, bug);
643     }

647         for (Map.Entry<String, BugInstance> e : potentialBugs.entrySet()) {
648             if (!isHashableClassName(e.getKey())) {
649                 BugInstance bug = e.getValue();
650 
651                 bugReporter.reportBug(bug);
652             }
653         }
