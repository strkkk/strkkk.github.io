140     private final String name;
141 
142     private Attribute(final String name) {
143         this.name = name;
144     }
145 
146     /**
147      * Get the local name of this element.
148      *
149      * @return the local name
150      */
151     public String getLocalName() {
152         return name;
153     }
154 
155     private static final Map<String, Attribute> attributes;
156 
157     static {
158         final Map<String, Attribute> map = new HashMap<String, Attribute>(64);
159         for (Attribute attribute : values()) {
160             final String name = attribute.getLocalName();
161             if (name != null) map.put(name, attribute);
162         }
163         attributes = map;
164     }
165 
166     public static Attribute forName(String localName) {
167         final Attribute attribute = attributes.get(localName);
168         return attribute == null ? UNKNOWN : attribute;
169     }
