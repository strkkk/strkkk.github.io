24  import org.apache.logging.log4j.Logger;
25  import org.apache.logging.log4j.LogManager;
26  import org.apache.commons.jci.monitor.FilesystemAlterationListener;
27  import org.apache.commons.jci.monitor.FilesystemAlterationMonitor;
28  import org.apache.commons.jci.monitor.FilesystemAlterationObserver;
29  import org.apache.commons.lang3.StringUtils;
30  import org.apache.struts2.dispatcher.Dispatcher;
31  import org.springframework.web.context.support.XmlWebApplicationContext;
32  import org.springframework.beans.factory.support.DefaultListableBeanFactory;

34  import org.springframework.beans.BeansException;
35  
36  import javax.servlet.ServletContext;
37  import java.io.File;
38  import java.util.List;
39  import java.util.ArrayList;
40  import java.util.Set;
41  import java.util.HashSet;
42  import java.util.regex.Pattern;

44  /**
45   * <p>
46   * This class can be used instead of XmlWebApplicationContext, and it will watch jar files and directories for changes
47   * and reload then changed classes.
48   * </p>
49   *
50   * <p>
51   * To use this class:
52   * </p>
53   * <ul>
54   * <li>Set "struts.devMode" to "true" </li>
55   * <li>Set "struts.class.reloading.watchList" to a comma separated list of directories, or jar files (absolute paths)</li>
56   * <li>Add this to web.xml:
57   *  <pre>
58   *  &lt;context-param&gt;
59   *       &lt;param-name&gt;contextClass&lt;/param-name&gt;
60   *       &lt;param-value&gt;org.apache.struts2.spring.ClassReloadingXMLWebApplicationContext&lt;/param-value&gt;
61   *   &lt;/context-param&gt;
62   *   </pre>
63   *  </li>
64   * <li>Add Apache Commons JCI FAM to the classpath. If you are using maven, add this to pom.xml:
65   *  <pre>
66   *  &lt;dependency&gt;
67   *       &lt;groupId&gt;org.apache.commons&lt;/groupId&gt;
68   *       &lt;artifactId&gt;commons-jci-fam&lt;/artifactId&gt;
69   *       &lt;version&gt;1.0&lt;/version&gt;
70   *       &lt;optional&gt;true&lt;/optional&gt;
71   *  &lt;/dependency&gt;
72   *  </pre>
73   * </li>
74   * </ul>
75   */
76  public class ClassReloadingXMLWebApplicationContext extends XmlWebApplicationContext implements FilesystemAlterationListener {
77      private static final Logger LOG = LogManager.getLogger(ClassReloadingXMLWebApplicationContext.class);
78  
79      protected ReloadingClassLoader classLoader;
80      protected FilesystemAlterationMonitor filesystemAlterationMonitor;
81  
82      protected ClassReloadingBeanFactory beanFactory;
83      //reload the runtime configuration when a change is detected
84      private boolean reloadConfig;
85  
86      public void setupReloading(String[] watchList, String acceptClasses, ServletContext servletContext, boolean reloadConfig) {
87          this.reloadConfig = reloadConfig;
88  
89          classLoader = new ReloadingClassLoader(ClassReloadingXMLWebApplicationContext.class.getClassLoader());
90  
91          //make a list of accepted classes
92          if (StringUtils.isNotBlank(acceptClasses)) {
93              String[] splitted = acceptClasses.split(",");
94              Set<Pattern> patterns = new HashSet<>(splitted.length);
95              for (String pattern : splitted)
96                  patterns.add(Pattern.compile(pattern));
97  
98              classLoader.setAccepClasses(patterns);
99          }
100 
101         filesystemAlterationMonitor = new FilesystemAlterationMonitor();
102 
103         //setup stores
104         for (String watch : watchList) {
105             File file = new File(watch);
106 
107             //make it absolute, if it is a relative path
108             if (!file.isAbsolute())
109                 file = new File(servletContext.getRealPath(watch));
110 
111             if (watch.endsWith(".jar")) {
112                 classLoader.addResourceStore(new JarResourceStore(file));
113                 //register with the filesystemAlterationMonitor
114                 filesystemAlterationMonitor.addListener(file, this);
115                 LOG.debug("Watching [{}] for changes", file.getAbsolutePath());
116             } else {
117                 //get all subdirs
118                 List<File> dirs = new ArrayList<>();
119                 getAllPaths(file, dirs);
120 
121                 classLoader.addResourceStore(new FileResourceStore(file));
122 
123                 for (File dir : dirs) {
124                     //register with the filesystemAlterationMonitor
125                     filesystemAlterationMonitor.addListener(dir, this);
126                 	LOG.debug("Watching [{}] for changes", dir.getAbsolutePath());
127                 }
128             }
129         }
130         //setup the bean factory
131         beanFactory = new ClassReloadingBeanFactory();
132         beanFactory.setInstantiationStrategy(new ClassReloadingInstantiationStrategy());
133         beanFactory.setBeanClassLoader(classLoader);
134 
135         //start watch thread
136         filesystemAlterationMonitor.start();
137     }
138 
139     /**
140      * If root is a dir, find all the subdir paths
141      */
142     private void getAllPaths(File root, List<File> dirs) {
143         dirs.add(root);
144 
145         if (root.isDirectory()) {
146             File[] files = root.listFiles();
147             if (files != null) {
148                 for (File file : files) {
149                     if (file.isDirectory()) {
150                         getAllPaths(file, dirs);
151                     }
152                 }
153             }
154         }
155     }
156 
157     public void close() {
158         super.close();
159 
160         if (filesystemAlterationMonitor != null) {
161             filesystemAlterationMonitor.removeListener(this);
162             filesystemAlterationMonitor.stop();
163         }
164     }
165 
166     public void refresh() throws BeansException, IllegalStateException {
167         if (classLoader != null) {
168             classLoader.reload();
169         }
170 
171         super.refresh();
172     }
173 
174     protected DefaultListableBeanFactory createBeanFactory() {
175         return beanFactory != null ? beanFactory : super.createBeanFactory();
176     }
177 
178     protected void prepareBeanFactory(ConfigurableListableBeanFactory beanFactory) {
179         super.prepareBeanFactory(beanFactory);
180 
181         //overwrite the class loader in the bean factory
182         if (classLoader != null) {
183             beanFactory.setBeanClassLoader(classLoader);
184         }
185     }
186 
187     public void onDirectoryChange(File file) {
188         reload(file);
189     }
190 
191     public void onDirectoryCreate(File file) {
192         reload(file);
193     }
194 
195     public void onDirectoryDelete(File file) {
196     }
197 
198     public void onFileChange(File file) {
199         reload(file);
200     }
201 
202     public void onFileCreate(File file) {
203         reload(file);
204     }
205 
206     private void reload(File file) {
207         if (classLoader != null) {
208             LOG.debug("Change detected in file [{}], reloading class loader", file.getAbsolutePath());
209             classLoader.reload();
210             if (reloadConfig && Dispatcher.getInstance() != null) {
211                 LOG.debug("Change detected in file [{}], reloading configuration", file.getAbsolutePath());
212                 Dispatcher.getInstance().getConfigurationManager().reload();
213             }
214         }
215     }
216 
217     public void onFileDelete(File file) {
218     }
219 
220     public void onStart(FilesystemAlterationObserver filesystemAlterationObserver) {
221     }
222 
223     public void onStop(FilesystemAlterationObserver filesystemAlterationObserver) {
224     }
225 
226     public ReloadingClassLoader getReloadingClassLoader() {
227         return classLoader;
228     }
