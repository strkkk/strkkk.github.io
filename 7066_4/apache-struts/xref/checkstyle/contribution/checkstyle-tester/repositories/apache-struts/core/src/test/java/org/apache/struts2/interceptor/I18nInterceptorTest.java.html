55      public void testNoSessionNoLocale() throws Exception {
56          request.setSession(null);

60          } catch (Exception ignore) {
61              fail("Shouldn't throw any exception!");

226     public void testCookieCreation() throws Exception {
227 
228         prepare(I18nInterceptor.DEFAULT_COOKIE_PARAMETER, "da_DK");
229 
230         final Cookie cookie = new Cookie(I18nInterceptor.DEFAULT_COOKIE_ATTRIBUTE, "da_DK");
231 
232         HttpServletResponse response = EasyMock.createMock(HttpServletResponse.class);
233         response.addCookie(CookieMatcher.eqCookie(cookie));
234         EasyMock.replay(response);
235 
236         ac.put(StrutsStatics.HTTP_RESPONSE, response);
237         interceptor.setLocaleStorage(I18nInterceptor.Storage.COOKIE.name());
238         interceptor.intercept(mai);
239 
240         EasyMock.verify(response);
241 
242         assertNull(session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE)); // should not be stored here
243         assertNull(session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE)); // should not create a locale object
244     }

246     private void prepare(String key, Serializable value) {
247         Map<String, Serializable> params = new HashMap<>();
248         params.put(key, value);
249 
250         mai.getInvocationContext().setParameters(HttpParameters.create(params).build());
251     }
252 
253     public void setUp() throws Exception {
254         interceptor = new I18nInterceptor();
255         interceptor.setLocaleProviderFactory(new DefaultLocaleProviderFactory());
256         interceptor.init();
257         session = new HashMap();
258 
259         Map<String, Object> ctx = new HashMap<String, Object>();
260         ctx.put(ActionContext.PARAMETERS, HttpParameters.create().build());
261         ctx.put(ActionContext.SESSION, session);
262 
263         ac = new ActionContext(ctx);
264 
265         ServletActionContext.setContext(ac);
266         request = new MockHttpServletRequest();
267         request.setSession(new MockHttpSession());
268         ServletActionContext.setRequest(request);
269 
270         Action action = new Action() {
271             public String execute() throws Exception {
272                 return SUCCESS;
273             }
274         };
275 
276         MockActionProxy proxy = new MockActionProxy();
277         proxy.setAction(action);
278         proxy.setNamespace("i18n");
279         proxy.setActionName("anAction");
280 
281         mai = new MockActionInvocation();
282         ((MockActionInvocation) mai).setAction(action);
283         ((MockActionInvocation) mai).setInvocationContext(ac);
284         ((MockActionInvocation) mai).setProxy(proxy);
285     }

295     static class CookieMatcher implements IArgumentMatcher {
296         private Cookie expected;
297 
298         CookieMatcher(Cookie cookie) {
299             expected = cookie;
300         }
301 
302         public boolean matches(Object argument) {
303             Cookie cookie = ((Cookie) argument);
304             return
305                 (cookie.getName().equals(expected.getName()) &&
306                  cookie.getValue().equals(expected.getValue()));
307         }
308 
309         public static Cookie eqCookie(Cookie ck) {
310             EasyMock.reportMatcher(new CookieMatcher(ck));
311             return null;
312         }
313 
314         public void appendTo(StringBuffer buffer) {
315             buffer
316                 .append("Received")
317                 .append(expected.getName())
318                 .append("/")
319                 .append(expected.getValue());
320         }
321     }
