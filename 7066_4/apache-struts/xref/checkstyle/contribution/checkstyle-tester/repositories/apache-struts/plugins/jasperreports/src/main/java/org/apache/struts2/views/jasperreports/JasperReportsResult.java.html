58  /**
59   * <!-- START SNIPPET: description -->
60   * <p>
61   * Generates a JasperReports report using the specified format or PDF if no
62   * format is specified.
63   * </p>
64   * <!-- END SNIPPET: description -->
65   * <p>
66   * <b>This result type takes the following parameters:</b>
67   * </p>
68   * <!-- START SNIPPET: params -->
69   *
70   * <ul>
71   *
72   * <li><b>location (default)</b> - the location where the compiled jasper report
73   * definition is (foo.jasper), relative from current URL.</li>
74   * <li><b>dataSource (required)</b> - the EL expression used to retrieve the
75   * datasource from the value stack (usually a List).</li>
76   * <li><b>parse</b> - true by default. If set to false, the location param will
77   * not be parsed for EL expressions.</li>
78   * <li><b>format</b> - the format in which the report should be generated. Valid
79   * values can be found in {@link JasperReportConstants}. If no format is
80   * specified, PDF will be used.</li>
81   * <li><b>contentDisposition</b> - disposition (defaults to "inline", values are
82   * typically <i>filename="document.pdf"</i>).</li>
83   * <li><b>documentName</b> - name of the document (will generate the http header
84   * <code>Content-disposition = X; filename=X.[format]</code>).</li>
85   * <li><b>delimiter</b> - the delimiter used when generating CSV reports. By
86   * default, the character used is ",".</li>
87   * <li><b>imageServletUrl</b> - name of the url that, when prefixed with the
88   * context page, can return report images.</li>
89   * <li>
90   * <b>reportParameters</b> - (2.1.2+) OGNL expression used to retrieve a map of
91   * report parameters from the value stack. The parameters may be accessed
92   * in the report via the usual JR mechanism and might include data not
93   * part of the dataSource, such as the user name of the report creator, etc.
94   * </li>
95   * <li>
96   * <b>exportParameters</b> - (2.1.2+) OGNL expression used to retrieve a map of
97   * JR exporter parameters from the value stack. The export parameters are
98   * used to customize the JR export. For example, a PDF export might enable
99   * encryption and set the user password to a string known to the report creator.
100  * </li>
101  * <li>
102  * <b>connection</b> - (2.1.7+) JDBC Connection which can be passed to the
103  * report instead of dataSource
104  * </li>
105  * <li><b>wrapField</b> - (2.3.18+) defines if fields should warp with ValueStackDataSource
106  * see https://issues.apache.org/jira/browse/WW-3698 for more details
107  * </li>
108  * </ul>
109  * <p>
110  * This result follows the same rules from {@link StrutsResultSupport}.
111  * Specifically, all parameters will be parsed if the "parse" parameter
112  * is not set to false.
113  * </p>
114  * <!-- END SNIPPET: params -->
115  * <p><b>Example:</b></p>
116  * <pre>
117  * <!-- START SNIPPET: example1 -->
118  * &lt;result name="success" type="jasper"&gt;
119  *   &lt;param name="location"&gt;foo.jasper&lt;/param&gt;
120  *   &lt;param name="dataSource"&gt;mySource&lt;/param&gt;
121  *   &lt;param name="format"&gt;CSV&lt;/param&gt;
122  * &lt;/result&gt;
123  * <!-- END SNIPPET: example1 -->
124  * </pre>
125  * <p>
126  * or for pdf
127  *
128  * <pre>
129  * <!-- START SNIPPET: example2 -->
130  * &lt;result name="success" type="jasper"&gt;
131  *   &lt;param name="location"&gt;foo.jasper&lt;/param&gt;
132  *   &lt;param name="dataSource"&gt;mySource&lt;/param&gt;
133  * &lt;/result&gt;
134  * <!-- END SNIPPET: example2 -->
135  * </pre>
136  */

249     protected void doExecute(String finalLocation, ActionInvocation invocation) throws Exception {
250         // Will throw a runtime exception if no "datasource" property. TODO Best place for that is...?
251         initializeProperties(invocation);
252 
253         LOG.debug("Creating JasperReport for dataSource = {}, format = {}", dataSource, format);
254 
255         HttpServletRequest request = (HttpServletRequest) invocation.getInvocationContext()
256                 .get(ServletActionContext.HTTP_REQUEST);
257         HttpServletResponse response = (HttpServletResponse) invocation.getInvocationContext()
258                 .get(ServletActionContext.HTTP_RESPONSE);
259 
260         // Handle IE special case: it sends a "contype" request first.
261         // TODO Set content type to config settings?
262         if ("contype".equals(request.getHeader("User-Agent"))) {
263             try (OutputStream outputStream = response.getOutputStream()) {
264                 response.setContentType("application/pdf");
265                 response.setContentLength(0);
266             } catch (IOException e) {
267                 LOG.error("Error writing report output", e);
268                 throw new ServletException(e.getMessage(), e);
269             }
270             return;
271         }
272 
273         // Construct the data source for the report.
274         ValueStack stack = invocation.getStack();
275         ValueStackDataSource stackDataSource = null;
276 
277         Connection conn = (Connection) stack.findValue(connection);
278         if (conn == null) {
279             stackDataSource = new ValueStackDataSource(stack, dataSource, wrapField);
280         }
281 
282         if ("https".equalsIgnoreCase(request.getScheme())) {
283             // set the the HTTP Header to work around IE SSL weirdness
284             response.setHeader("CACHE-CONTROL", "PRIVATE");
285             response.setHeader("Cache-Control", "maxage=3600");
286             response.setHeader("Pragma", "public");
287             response.setHeader("Accept-Ranges", "none");
288         }
289 
290         // Determine the directory that the report file is in and set the reportDirectory parameter
291         // For WW 2.1.7:
292         //  ServletContext servletContext = ((ServletConfig) invocation.getInvocationContext().get(ServletActionContext.SERVLET_CONFIG)).getServletContext();
293         ServletContext servletContext = (ServletContext) invocation.getInvocationContext()
294                 .get(ServletActionContext.SERVLET_CONTEXT);
295         String systemId = servletContext.getRealPath(finalLocation);
296         Map<String, Object> parameters = new ValueStackShadowMap(stack);
297         File directory = new File(systemId.substring(0, systemId.lastIndexOf(File.separator)));
298         parameters.put("reportDirectory", directory);
299         parameters.put(JRParameter.REPORT_LOCALE, invocation.getInvocationContext().getLocale());
300 
301         // put timezone in jasper report parameter
302         if (timeZone != null) {
303             timeZone = conditionalParse(timeZone, invocation);
304             final TimeZone tz = TimeZone.getTimeZone(timeZone);
305             if (tz != null) {
306                 // put the report time zone
307                 parameters.put(JRParameter.REPORT_TIME_ZONE, tz);
308             }
309         }
310 
311         // Add any report parameters from action to param map.
312         Map reportParams = (Map) stack.findValue(reportParameters);
313         if (reportParams != null) {
314             LOG.debug("Found report parameters; adding to parameters...");
315             parameters.putAll(reportParams);
316         }
317 
318         ByteArrayOutputStream output;
319         JasperPrint jasperPrint;
320 
321         // Fill the report and produce a print object
322         try {
323             JasperReport jasperReport = (JasperReport) JRLoader.loadObject(new File(systemId));
324             if (conn == null) {
325                 jasperPrint = JasperFillManager.fillReport(jasperReport, parameters, stackDataSource);
326             } else {
327                 jasperPrint = JasperFillManager.fillReport(jasperReport, parameters, conn);
328             }
329         } catch (JRException e) {
330             LOG.error("Error building report for uri {}", systemId, e);
331             throw new ServletException(e.getMessage(), e);
332         }
333 
334         // Export the print object to the desired output format
335         try {
336             if (contentDisposition != null || documentName != null) {
337                 final StringBuilder tmp = new StringBuilder();
338                 tmp.append((contentDisposition == null) ? "inline" : contentDisposition);
339 
340                 if (documentName != null) {
341                     tmp.append("; filename=");
342                     tmp.append(documentName);
343                     tmp.append(".");
344                     tmp.append(format.toLowerCase());
345                 }
346 
347                 response.setHeader("Content-disposition", tmp.toString());
348             }
349 
350             //TODO: replace deprecated logic
351             JRExporter exporter;
352 
353             switch (format) {
354                 case FORMAT_PDF:
355                     response.setContentType("application/pdf");
356                     exporter = new JRPdfExporter();
357                     break;
358                 case FORMAT_CSV:
359                     response.setContentType("text/csv");
360                     exporter = new JRCsvExporter();
361                     break;
362                 case FORMAT_HTML:
363                     response.setContentType("text/html");
364 
365                     // IMAGES_MAPS seems to be only supported as "backward compatible" from JasperReports 1.1.0
366 
367                     Map imagesMap = new HashMap();
368                     request.getSession(true).setAttribute("IMAGES_MAP", imagesMap);
369 
370                     exporter = new HtmlExporter();
371                     exporter.setParameter(JRHtmlExporterParameter.IMAGES_MAP, imagesMap);
372                     exporter.setParameter(JRHtmlExporterParameter.IMAGES_URI, request.getContextPath() + imageServletUrl);
373 
374                     // Needed to support chart images:
375                     exporter.setParameter(JRExporterParameter.JASPER_PRINT, jasperPrint);
376                     request.getSession().setAttribute("net.sf.jasperreports.j2ee.jasper_print", jasperPrint);
377                     break;
378                 case FORMAT_XLS:
379                     response.setContentType("application/vnd.ms-excel");
380                     exporter = new JRXlsExporter();
381                     break;
382                 case FORMAT_XML:
383                     response.setContentType("text/xml");
384                     exporter = new JRXmlExporter();
385                     break;
386                 case FORMAT_RTF:
387                     response.setContentType("application/rtf");
388                     exporter = new JRRtfExporter();
389                     break;
390                 default:
391                     throw new ServletException("Unknown report format: " + format);
392             }
393 
394             Map exportParams = (Map) stack.findValue(exportParameters);
395             if (exportParams != null) {
396                 LOG.debug("Found export parameters; adding to exporter parameters...");
397                 exporter.getParameters().putAll(exportParams);
398             }
399 
400             output = exportReportToBytes(jasperPrint, exporter);
401         } catch (JRException e) {
402             LOG.error("Error producing {} report for uri {}", format, systemId, e);
403             throw new ServletException(e.getMessage(), e);
404         } finally {
405             try {
406                 if (conn != null) {
407                     // avoid NPE if connection was not used for the report
408                     conn.close();
409                 }
410             } catch (Exception e) {
411                 LOG.warn("Could not close db connection properly", e);
412             }
413         }
414 
415         response.setContentLength(output.size());
416         // Will throw ServletException on IOException.
417         writeReport(response, output);
418     }
419 
420     /**
421      * Writes report bytes to response output stream.
422      *
423      * @param response Current response.
424      * @param output   Report bytes to write.
425      * @throws ServletException on stream IOException.
426      */
427     private void writeReport(HttpServletResponse response, ByteArrayOutputStream output) throws ServletException {
428         try (OutputStream outputStream = response.getOutputStream()) {
429             output.writeTo(outputStream);
430             outputStream.flush();
431         } catch (IOException e) {
432             LOG.error("Error writing report output", e);
433             throw new ServletException(e.getMessage(), e);
434         }
435     }

437     /**
438      * Sets up result properties, parsing etc.
439      *
440      * @param invocation Current invocation.
441      * @throws Exception on initialization error.
442      */
443     private void initializeProperties(ActionInvocation invocation) {
444         if (dataSource == null && connection == null) {
445             String message = "No dataSource specified...";
446             LOG.error(message);
447             throw new RuntimeException(message);
448         }
449         if (dataSource != null) {
450             dataSource = conditionalParse(dataSource, invocation);
451         }
452 
453         format = conditionalParse(format, invocation);
454         if (StringUtils.isEmpty(format)) {
455             format = FORMAT_PDF;
456         }
457 
458         if (contentDisposition != null) {
459             contentDisposition = conditionalParse(contentDisposition, invocation);
460         }
461 
462         if (documentName != null) {
463             documentName = conditionalParse(documentName, invocation);
464         }
465 
466         reportParameters = conditionalParse(reportParameters, invocation);
467         exportParameters = conditionalParse(exportParameters, invocation);
468     }

470     /**
471      * Run a Jasper report to CSV format and put the results in a byte array
472      *
473      * @param jasperPrint The Print object to render as CSV
474      * @param exporter    The exporter to use to export the report
475      * @return A CSV formatted report
476      * @throws net.sf.jasperreports.engine.JRException If there is a problem running the report
477      */
478     private ByteArrayOutputStream exportReportToBytes(JasperPrint jasperPrint, JRExporter exporter) throws JRException {
479         ByteArrayOutputStream baos = new ByteArrayOutputStream();
480 
481         exporter.setParameter(JRExporterParameter.JASPER_PRINT, jasperPrint);
482         exporter.setParameter(JRExporterParameter.OUTPUT_STREAM, baos);
483         if (delimiter != null) {
484             exporter.setParameter(JRCsvExporterParameter.FIELD_DELIMITER, delimiter);
485         }
486 
487         exporter.exportReport();
488         return baos;
489     }
