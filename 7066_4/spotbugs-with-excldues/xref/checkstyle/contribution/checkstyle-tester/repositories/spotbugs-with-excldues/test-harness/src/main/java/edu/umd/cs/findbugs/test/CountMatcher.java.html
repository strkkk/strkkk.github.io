1   package edu.umd.cs.findbugs.test;
2   
3   import org.hamcrest.Description;
4   import org.hamcrest.Factory;
5   import org.hamcrest.Matcher;
6   import org.hamcrest.TypeSafeMatcher;
7   
8   /**
9    * @since ?
10   *
11   * @param <T>
12   *            matcher item type
13   */
14  public final class CountMatcher<T> extends TypeSafeMatcher<Iterable<T>> {
15  
16      private final int count;
17      private final Matcher<T> matcher;
18  
19      public CountMatcher(int count, Matcher<T> matcher) {
20          this.count = count;
21          this.matcher = matcher;
22      }
23  
24      /**
25       * Creates a matcher for {@link Iterable}s that only matches if exactly {@code count} items match the specified
26       * {@code matcher}.
27       *
28       * @param matcher
29       *            A non-{@code null} matcher that must match exactly {@code count} times.
30       * @param <T>
31       *            matcher item type
32       * @param count
33       *            How many times the {@code matcher} must match.
34       * @return new matcher instance
35       */
36      @Factory
37      public static <T> Matcher<Iterable<T>> containsExactly(final int count, final Matcher<T> matcher) {
38          return new CountMatcher<>(count, matcher);
39      }
40  
41      @Override
42      protected boolean matchesSafely(Iterable<T> iterable) {
43          int numberOfmatches = 0;
44  
45          for (final Object item : iterable) {
46              if (matcher.matches(item)) {
47                  numberOfmatches++;
48              }
49          }
50  
51          return numberOfmatches == count;
52      }
53  
54      @Override
55      public void describeTo(final Description desc) {
56          desc.appendText("Iterable containing exactly ").appendValue(count).appendText(" ").appendDescriptionOf(matcher);
57      }
58  }
