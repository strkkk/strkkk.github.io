258     /**
259      * Constructor taking the error code as defined for the CKR_* constants
260      * in PKCS#11.
261      */
262     public PKCS11Exception(long errorCode) {
263         errorCode_ = errorCode;
264     }
265 
266     /**
267      * This method gets the corresponding text error message from
268      * a property file. If this file is not available, it returns the error
269      * code as a hex-string.
270      *
271      * @return The message or the error code; e.g. "CKR_DEVICE_ERROR" or
272      *         "0x00000030".
273      * @preconditions
274      * @postconditions (result <> null)
275      */
276     public String getMessage() {
277         String message = errorMap.get(Long.valueOf(errorCode_));
278         if (message == null) {
279             message = "0x" + Functions.toFullHexString((int)errorCode_);
280         }
281         return message;
282     }
283 
284     /**
285      * Returns the PKCS#11 error code.
286      *
287      * @return The error code; e.g. 0x00000030.
288      * @preconditions
289      * @postconditions
290      */
291     public long getErrorCode() {
292         return errorCode_ ;
293     }
