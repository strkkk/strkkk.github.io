1   package org.infinispan.commands.tx;
2   
3   import org.infinispan.context.InvocationContext;
4   import org.infinispan.context.InvocationContextFactory;
5   import org.infinispan.context.impl.RemoteTxInvocationContext;
6   import org.infinispan.interceptors.InterceptorChain;
7   import org.infinispan.lifecycle.ComponentStatus;
8   import org.infinispan.remoting.transport.Address;
9   import org.infinispan.transaction.impl.RemoteTransaction;
10  import org.infinispan.transaction.impl.TransactionTable;
11  import org.infinispan.transaction.xa.GlobalTransaction;
12  import org.infinispan.util.logging.Log;
13  import org.infinispan.util.logging.LogFactory;
14  
15  /**
16   * An abstract transaction boundary command that holds a reference to a {@link org.infinispan.transaction.xa.GlobalTransaction}
17   *
18   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
19   * @author Mircea.Markus@jboss.com
20   * @since 4.0
21   */
22  public abstract class AbstractTransactionBoundaryCommand implements TransactionBoundaryCommand {
23  
24     private static final Log log = LogFactory.getLog(AbstractTransactionBoundaryCommand.class);
25     private static boolean trace = log.isTraceEnabled();
26  
27     protected GlobalTransaction globalTx;
28     protected final String cacheName;
29     protected InterceptorChain invoker;
30     protected InvocationContextFactory icf;
31     protected TransactionTable txTable;
32     private Address origin;
33     private int topologyId = -1;
34  
35     public AbstractTransactionBoundaryCommand(String cacheName) {
36        this.cacheName = cacheName;
37     }
38  
39     public void init(InterceptorChain chain, InvocationContextFactory icf, TransactionTable txTable) {
40        this.invoker = chain;
41        this.icf = icf;
42        this.txTable = txTable;
43     }
44  
45     @Override
46     public int getTopologyId() {
47        return topologyId;
48     }
49  
50     @Override
51     public void setTopologyId(int topologyId) {
52        this.topologyId = topologyId;
53     }
54  
55     @Override
56     public String getCacheName() {
57        return cacheName;
58     }
59  
60     @Override
61     public GlobalTransaction getGlobalTransaction() {
62        return globalTx;
63     }
64  
65     @Override
66     public void markTransactionAsRemote(boolean isRemote) {
67        globalTx.setRemote(isRemote);
68     }
69  
70     /**
71      * This is what is returned to remote callers when an invalid RemoteTransaction is encountered.  Can happen if a
72      * remote node propagates a transactional call to the current node, and the current node has no idea of the transaction
73      * in question.  Can happen during rehashing, when ownerships are reassigned during a transactions.
74      *
75      * Returning a null usually means the transactional command succeeded.
76      * @return return value to respond to a remote caller with if the transaction context is invalid.
77      */
78     protected Object invalidRemoteTxReturnValue() {
79        return null;
80     }
81  
82     @Override
83     public Object perform(InvocationContext ctx) throws Throwable {
84        if (ctx != null) throw new IllegalStateException("Expected null context!");
85        markGtxAsRemote();
86        RemoteTransaction transaction = getRemoteTransaction();
87        if (transaction == null) {
88           if (trace) log.tracef("Did not find a RemoteTransaction for %s", globalTx);
89           return invalidRemoteTxReturnValue();
90        }
91        visitRemoteTransaction(transaction);
92        RemoteTxInvocationContext ctxt = icf.createRemoteTxInvocationContext(
93              transaction, getOrigin());
94  
95        if (trace) log.tracef("About to execute tx command %s", this);
96        return invoker.invoke(ctxt, this);
97     }
98  
99     protected void visitRemoteTransaction(RemoteTransaction tx) {
100       // to be overridden
101    }
102 
103    protected RemoteTransaction getRemoteTransaction() {
104       return txTable.getRemoteTransaction(globalTx);
105    }
106 
107    @Override
108    public Object[] getParameters() {
109       return new Object[]{globalTx};
110    }
111 
112    @Override
113    public void setParameters(int commandId, Object[] args) {
114       globalTx = (GlobalTransaction) args[0];
115    }
116 
117    @Override
118    public boolean shouldInvoke(InvocationContext ctx) {
119       return true;
120    }
121 
122    @Override
123    public boolean ignoreCommandOnStatus(ComponentStatus status) {
124       return false;
125    }
126 
127    public boolean equals(Object o) {
128       if (this == o) return true;
129       if (o == null || getClass() != o.getClass()) return false;
130 
131       AbstractTransactionBoundaryCommand that = (AbstractTransactionBoundaryCommand) o;
132       return this.globalTx.equals(that.globalTx);
133    }
134 
135    public int hashCode() {
136       return globalTx.hashCode();
137    }
138 
139    @Override
140    public String toString() {
141       return "gtx=" + globalTx +
142             ", cacheName='" + cacheName + '\'' +
143             ", topologyId=" + topologyId +
144             '}';
145    }
146 
147    private void markGtxAsRemote() {
148       globalTx.setRemote(true);
149    }
150    
151    @Override
152    public Address getOrigin() {
153 	   return origin;
154    }
155    
156    @Override
157    public void setOrigin(Address origin) {
158 	   this.origin = origin;
159    }
160 
161    @Override
162    public boolean isReturnValueExpected() {
163       return true;
164    }
165 
166    @Override
167    public final boolean canBlock() {
168       //all tx commands must wait for the correct topology
169       return true;
170    }
171 }
