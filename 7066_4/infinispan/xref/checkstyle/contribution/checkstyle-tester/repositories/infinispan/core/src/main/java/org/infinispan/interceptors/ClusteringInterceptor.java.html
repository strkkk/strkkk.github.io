7   import org.infinispan.container.DataContainer;
8   import org.infinispan.container.EntryFactory;
9   import org.infinispan.container.entries.CacheEntry;
10  import org.infinispan.container.entries.InternalCacheEntry;
11  import org.infinispan.context.Flag;
12  import org.infinispan.context.InvocationContext;
13  import org.infinispan.distribution.ch.ConsistentHash;
14  import org.infinispan.factories.annotations.Inject;
15  import org.infinispan.factories.annotations.Start;
16  import org.infinispan.interceptors.base.BaseRpcInterceptor;
17  import org.infinispan.statetransfer.StateTransferManager;
18  import org.infinispan.util.concurrent.locks.LockManager;
19  
20  import static org.infinispan.commons.util.Util.toStr;

22  /**
23   * Base class for replication and distribution interceptors.
24   *
25   * @author anistor@redhat.com
26   * @since 5.2
27   */
28  public abstract class ClusteringInterceptor extends BaseRpcInterceptor {
29  
30     protected CommandsFactory cf;
31     protected EntryFactory entryFactory;
32     protected LockManager lockManager;
33     protected DataContainer dataContainer;
34     protected StateTransferManager stateTransferManager;
35     protected boolean needReliableReturnValues;
36     protected boolean isL1Enabled;
37  
38     @Inject
39     public void injectDependencies(CommandsFactory cf, EntryFactory entryFactory,
40                                    LockManager lockManager, DataContainer dataContainer,
41                                    StateTransferManager stateTransferManager) {
42        this.cf = cf;
43        this.entryFactory = entryFactory;
44        this.lockManager = lockManager;
45        this.dataContainer = dataContainer;
46        this.stateTransferManager = stateTransferManager;
47     }
48  
49     @Start
50     public void configure() {
51        needReliableReturnValues = !cacheConfiguration.unsafe().unreliableReturnValues();
52        isL1Enabled = cacheConfiguration.clustering().l1().enabled();
53     }
54  
55     protected boolean isNeedReliableReturnValues(FlagAffectedCommand command) {
56        return !command.hasFlag(Flag.SKIP_REMOTE_LOOKUP)
57              && !command.hasFlag(Flag.IGNORE_RETURN_VALUES) && needReliableReturnValues;
58     }
59  
60     protected boolean needsRemoteGet(InvocationContext ctx, AbstractDataCommand command) {
61        if (command.hasFlag(Flag.CACHE_MODE_LOCAL)
62              || command.hasFlag(Flag.SKIP_REMOTE_LOOKUP)
63              || command.hasFlag(Flag.IGNORE_RETURN_VALUES)) {
64           return false;
65        }
66        boolean shouldFetchFromRemote = false;
67        CacheEntry entry = ctx.lookupEntry(command.getKey());
68        if (entry == null || entry.isNull()) {
69           Object key = command.getKey();
70           ConsistentHash ch = stateTransferManager.getCacheTopology().getReadConsistentHash();
71           shouldFetchFromRemote = ctx.isOriginLocal() && !isValueAvailableLocally(ch, key);
72           if (!shouldFetchFromRemote && getLog().isTraceEnabled()) {
73              getLog().tracef("Not doing a remote get for key %s since entry is mapped to current node (%s) or is in L1. Owners are %s", toStr(key), rpcManager.getAddress(), ch.locateOwners(key));
74           }
75        }
76        return shouldFetchFromRemote;
77     }
78  
79     protected boolean isValueAvailableLocally(ConsistentHash consistentHash, Object key) {
80        final boolean isLocal = consistentHash.isKeyLocalToNode(rpcManager.getAddress(), key);
81        final InternalCacheEntry ice;
82        return isLocal || (isL1Enabled && (ice = dataContainer.get(key)) != null && ice.isL1Entry());
83     }
84  
85     protected InternalCacheEntry fetchValueLocallyIfAvailable(ConsistentHash consistentHash, Object key) {
86        if (consistentHash.isKeyLocalToNode(rpcManager.getAddress(), key)) {
87           return dataContainer.get(key);
88        } else if (isL1Enabled) {
89           InternalCacheEntry entry = dataContainer.get(key);
90           return entry != null && entry.isL1Entry() ? entry : null;
91        }
92        return null;
93     }
94  
95     /**
96      * For conditional operations (replace, remove, put if absent) Used only for optimistic transactional caches, to solve the following situation:
97      * <pre>
98      * - node A (owner, tx originator) does a successful replace
99      * - the actual value changes
100     * - tx commits. The value is applied on A (the check was performed at operation time) but is not applied on
101     *   B (check is performed at commit time).
102     * In such situations (optimistic caches) the remote conditional command should not re-check the old value.
103     * </pre>
104     */
105    protected boolean ignorePreviousValueOnBackup(WriteCommand command, InvocationContext ctx) {
106       return ctx.isOriginLocal() && command.isSuccessful();
107    }
108 
109    /**
110     * Retrieves a cache entry from a remote source.  Would typically involve an RPC call using a {@link org.infinispan.commands.remote.ClusteredGetCommand}
111     * and some form of quorum of responses if the responses returned are inconsistent - often the case if there is a
112     * rehash in progress, involving nodes that the key maps to.
113     *
114     * @param key key to look up
115     * @param isWrite {@code true} if this is triggered by a write operation
116     * @return an internal cache entry, or null if it cannot be located
117     */
118    protected abstract InternalCacheEntry retrieveFromRemoteSource(Object key, InvocationContext ctx, boolean acquireRemoteLock, FlagAffectedCommand command, boolean isWrite) throws Exception;
119 }
