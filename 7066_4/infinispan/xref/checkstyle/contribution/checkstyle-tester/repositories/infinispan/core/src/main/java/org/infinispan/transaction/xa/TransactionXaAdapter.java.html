19  /**
20   * This acts both as an local {@link org.infinispan.transaction.xa.CacheTransaction} and implementor of an {@link
21   * javax.transaction.xa.XAResource} that will be called by tx manager on various tx stages.
22   *
23   * @author Mircea.Markus@jboss.com
24   * @since 4.0
25   */
26  public class TransactionXaAdapter extends AbstractEnlistmentAdapter implements XAResource {
27  
28     private static final Log log = LogFactory.getLog(TransactionXaAdapter.class);
29     private static boolean trace = log.isTraceEnabled();
30  
31     /**
32      * It is really useful only if TM and client are in separate processes and TM fails. This is because a client might
33      * call tm.begin and then the TM (running separate process) crashes. In this scenario the TM won't ever call
34      * XAResource.rollback, so these resources would be held there forever. By knowing the timeout the RM can proceed
35      * releasing the resources associated with given tx.
36      */
37     private int txTimeout;
38  
39     private final XaTransactionTable txTable;
40  
41  
42     /**
43      * XAResource is associated with a transaction between enlistment (XAResource.start()) XAResource.end(). It's only the
44      * boundary methods (prepare, commit, rollback) that need to be "stateless".
45      * Reefer to section 3.4.4 from JTA spec v.1.1
46      */
47     private final LocalXaTransaction localTransaction;
48     private final RecoveryManager recoveryManager;
49     private volatile RecoveryManager.RecoveryIterator recoveryIterator;
50     private boolean recoveryEnabled;
51     private String cacheName;
52     private boolean onePhaseTotalOrder;
53  
54     public TransactionXaAdapter(LocalXaTransaction localTransaction, TransactionTable txTable,
55                                 RecoveryManager rm, TransactionCoordinator txCoordinator,
56                                 CommandsFactory commandsFactory, RpcManager rpcManager,
57                                 ClusteringDependentLogic clusteringDependentLogic,
58                                 Configuration configuration, String cacheName) {
59        super(localTransaction, commandsFactory, rpcManager, txTable, clusteringDependentLogic, configuration, txCoordinator);
60        this.localTransaction = localTransaction;
61        this.txTable = (XaTransactionTable) txTable;
62        this.recoveryManager = rm;
63        this.cacheName = cacheName;
64        recoveryEnabled = configuration.transaction().recovery().enabled();
65        //in distributed mode with write skew check, we only allow 2 phases!!
66        this.onePhaseTotalOrder = configuration.transaction().transactionProtocol().isTotalOrder() &&
67              !(configuration.clustering().cacheMode().isDistributed() && configuration.locking().writeSkewCheck());
68     }
69     public TransactionXaAdapter(TransactionTable txTable,
70                                 RecoveryManager rm, TransactionCoordinator txCoordinator,
71                                 CommandsFactory commandsFactory, RpcManager rpcManager,
72                                 ClusteringDependentLogic clusteringDependentLogic,
73                                 Configuration configuration, String cacheName) {
74        super(commandsFactory, rpcManager, txTable, clusteringDependentLogic, configuration, txCoordinator);
75        localTransaction = null;
76        this.txTable = (XaTransactionTable) txTable;
77        this.recoveryManager = rm;
78        this.cacheName = cacheName;
79        recoveryEnabled = configuration.transaction().recovery().enabled();
80        //in distributed mode, we only allow 2 phases!!
81        this.onePhaseTotalOrder = configuration.transaction().transactionProtocol().isTotalOrder() &&
82              !configuration.clustering().cacheMode().isDistributed();
83     }
84  
85     /**
86      * This can be call for any transaction object. See Section 3.4.6 (Resource Sharing) from JTA spec v1.1.
87      */
88     @Override
89     public int prepare(Xid externalXid) throws XAException {
90        Xid xid = convertXid(externalXid);
91        LocalXaTransaction localTransaction = getLocalTransactionAndValidate(xid);
92        return txCoordinator.prepare(localTransaction);
93     }
94  
95     /**
96      * Same comment as for {@link #prepare(javax.transaction.xa.Xid)} applies for commit.
97      */
98     @Override
99     public void commit(Xid externalXid, boolean isOnePhase) throws XAException {
100       Xid xid = convertXid(externalXid);
101       LocalXaTransaction localTransaction = getLocalTransactionAndValidate(xid);
102       boolean committedInOnePhase;
103       if (isOnePhase && onePhaseTotalOrder) {
104          committedInOnePhase = txCoordinator.commit(localTransaction, true);
105       } else if (isOnePhase) {
106          //isOnePhase being true means that we're the only participant in the distributed transaction and TM does the
107          //1PC optimization. We run a 2PC though, as running only 1PC has a high chance of leaving the cluster in
108          //inconsistent state.
109          txCoordinator.prepare(localTransaction);
110          committedInOnePhase = txCoordinator.commit(localTransaction, false);
111       } else {
112          committedInOnePhase = txCoordinator.commit(localTransaction, false);
113       }
114       forgetSuccessfullyCompletedTransaction(recoveryManager, xid, localTransaction, committedInOnePhase);
115    }
116 
117    /**
118     * Same comment as for {@link #prepare(javax.transaction.xa.Xid)} applies for commit.
119     */   
120    @Override
121    public void rollback(Xid externalXid) throws XAException {
122       Xid xid = convertXid(externalXid);
123       LocalXaTransaction localTransaction1 = getLocalTransactionAndValidateImpl(xid, txTable);
124       localTransaction.markForRollback(true); //ISPN-879 : make sure that locks are no longer associated to this transactions
125       txCoordinator.rollback(localTransaction1);
126    }
127 
128    @Override
129    public void start(Xid externalXid, int i) throws XAException {
130       Xid xid = convertXid(externalXid);
131       //transform in our internal format in order to be able to serialize
132       localTransaction.setXid(xid);
133       txTable.addLocalTransactionMapping(localTransaction);
134       if (trace) log.tracef("start called on tx %s", this.localTransaction.getGlobalTransaction());
135    }
136 
137    @Override
138    public void end(Xid externalXid, int i) throws XAException {
139       if (trace) log.tracef("end called on tx %s(%s)", this.localTransaction.getGlobalTransaction(), cacheName);
140    }
141 
142    @Override
143    public void forget(Xid externalXid) throws XAException {
144       Xid xid = convertXid(externalXid);
145       if (trace) log.tracef("forget called for xid %s", xid);
146       try {
147          if (recoveryEnabled) {
148             recoveryManager.removeRecoveryInformationFromCluster(null, xid, true, null);
149          } else {
150             if (trace) log.trace("Recovery not enabled");
151          }
152       } catch (Exception e) {
153          log.warnExceptionRemovingRecovery(e);
154          XAException xe = new XAException(XAException.XAER_RMERR);
155          xe.initCause(e);
156          throw xe;
157       }
158    }
159 
160    @Override
161    public int getTransactionTimeout() throws XAException {
162       if (trace) log.trace("start called");
163       return txTimeout;
164    }
165 
166    /**
167     * the only situation in which it returns true is when the other xa resource pertains to the same cache, on
168     * the same node.
169     */
170    @Override
171    public boolean isSameRM(XAResource xaResource) throws XAException {
172       if (!(xaResource instanceof TransactionXaAdapter)) {
173          return false;
174       }
175       TransactionXaAdapter other = (TransactionXaAdapter) xaResource;
176       //there is only one tx table per cache and this is more efficient that equals.
177       return this.txTable == other.txTable;
178    }
179 
180    @Override
181    public Xid[] recover(int flag) throws XAException {
182       if (!recoveryEnabled) {
183          log.recoveryIgnored();
184          return RecoveryManager.RecoveryIterator.NOTHING;
185       }
186       if (trace) log.trace("recover called: " + flag);
187 
188       if (isFlag(flag, TMSTARTRSCAN)) {
189          recoveryIterator = recoveryManager.getPreparedTransactionsFromCluster();
190          if (trace) log.tracef("Fetched a new recovery iterator: %s" , recoveryIterator);
191       }
192       if (isFlag(flag, TMENDRSCAN)) {
193          if (trace) log.trace("Flushing the iterator");
194          return recoveryIterator.all();
195       } else {
196          //as per the spec: "TMNOFLAGS this flag must be used when no other flags are specified."
197          if (!isFlag(flag, TMSTARTRSCAN) && !isFlag(flag, TMNOFLAGS))
198             throw new IllegalArgumentException("TMNOFLAGS this flag must be used when no other flags are specified." +
199                                                      " Received " + flag);
200          return recoveryIterator.hasNext() ? recoveryIterator.next() : RecoveryManager.RecoveryIterator.NOTHING;
201       }
202    }
203 
204    private boolean isFlag(int value, int flag) {
205       return (value & flag) != 0;
206    }
207 
208    @Override
209    public boolean setTransactionTimeout(int i) throws XAException {
210       this.txTimeout = i;
211       return true;
212    }

214    @Override
215    public String toString() {
216       return "TransactionXaAdapter{" +
217             "localTransaction=" + localTransaction +
218             '}';
219    }

221    private void forgetSuccessfullyCompletedTransaction(RecoveryManager recoveryManager, Xid xid, LocalXaTransaction localTransaction, boolean committedInOnePhase) {
222       final GlobalTransaction gtx = localTransaction.getGlobalTransaction();
223       if (recoveryEnabled) {
224          recoveryManager.removeRecoveryInformationFromCluster(localTransaction.getRemoteLocksAcquired(), xid, false, gtx);
225          txTable.removeLocalTransaction(localTransaction);
226       } else {
227          releaseLocksForCompletedTransaction(localTransaction, committedInOnePhase);
228       }
229    }
230 
231    private LocalXaTransaction getLocalTransactionAndValidate(Xid xid) throws XAException {
232       return getLocalTransactionAndValidateImpl(xid, txTable);
233    }
234 
235    private static LocalXaTransaction getLocalTransactionAndValidateImpl(Xid xid, XaTransactionTable txTable) throws XAException {
236       LocalXaTransaction localTransaction = txTable.getLocalTransaction(xid);
237       if  (localTransaction == null) {
238          if (trace) log.tracef("no tx found for %s", xid);
239          throw new XAException(XAException.XAER_NOTA);
240       }
241       return localTransaction;
242    }
243 
244    public LocalXaTransaction getLocalTransaction() {
245       return localTransaction;
246    }
247 
248    /**
249     * Only does the conversion if recovery is enabled.
250     */
251    private Xid convertXid(Xid externalXid) {
252       if (recoveryEnabled && (!(externalXid instanceof SerializableXid))) {
253          return new SerializableXid(externalXid);
254       } else {
255          return externalXid;
256       }
257    }
258 
259    @Override
260    public boolean equals(Object o) {
261       if (this == o) return true;
262       if (o == null || getClass() != o.getClass()) return false;
263 
264       TransactionXaAdapter that = (TransactionXaAdapter) o;
265 
266       if (localTransaction != null ? !localTransaction.equals(that.localTransaction) : that.localTransaction != null)
267          return false;
268       //also include the name of the cache in comparison - needed when same tx spans multiple caches.
269       return cacheName != null ?
270             cacheName.equals(that.cacheName) : that.cacheName == null;
271    }
