86     @Override
87     public RecoveryIterator getPreparedTransactionsFromCluster() {
88        PreparedTxIterator iterator = new PreparedTxIterator();
89  
90        //1. get local transactions first
91        //add the locally prepared transactions. The list of prepared transactions (even if they are not in-doubt)
92        //is mandated by the recovery process according to the JTA spec: "The transaction manager calls this [i.e. recover]
93        // method during recovery to obtain the list of transaction branches that are currently in prepared or heuristically
94        // completed states."
95        iterator.add(txTable.getLocalPreparedXids());
96  
97        //2. now also add the in-doubt transactions.
98        iterator.add(getInDoubtTransactions());
99  
100       //3. then the remote ones
101       if (notOnlyMeInTheCluster() && broadcastForPreparedTx) {
102          boolean success = true;
103          Map<Address, Response> responses = getAllPreparedTxFromCluster();
104          for (Map.Entry<Address, Response> rEntry : responses.entrySet()) {
105             Response thisResponse = rEntry.getValue();
106             if (isSuccessful(thisResponse)) {
107                List<Xid> responseValue = (List<Xid>) ((SuccessfulResponse) thisResponse).getResponseValue();
108                if (log.isTraceEnabled()) {
109                   log.tracef("Received Xid lists %s from node %s", responseValue, rEntry.getKey());
110                }
111                iterator.add(responseValue);
112             } else {
113                log.missingListPreparedTransactions(rEntry.getKey(), rEntry.getValue());
114                success = false;
115             }
116          }
117          //this makes sure that the broadcast only happens once!
118          this.broadcastForPreparedTx = !success;
119          if (!broadcastForPreparedTx)
120             log.debug("Finished broadcasting for remote prepared transactions. Returning only local values from now on.");
121       }
122       return iterator;
123    }

145    @Override
146    public RecoveryAwareTransaction removeRecoveryInformation(Xid xid) {
147       RecoveryAwareTransaction remove = inDoubtTransactions.remove(new RecoveryInfoKey(xid, cacheName));
148       log.tracef("removed in doubt xid: %s", xid);
149       if (remove == null) {
150          return (RecoveryAwareTransaction) txTable.removeRemoteTransaction(xid);
151       }
152       return remove;
153    }

155    @Override
156    public RecoveryAwareTransaction removeRecoveryInformation(Long internalId) {
157       Xid remoteTransactionXid = txTable.getRemoteTransactionXid(internalId);
158       if (remoteTransactionXid != null) {
159          return removeRecoveryInformation(remoteTransactionXid);
160       } else {
161          for (RecoveryAwareRemoteTransaction raRemoteTx : inDoubtTransactions.values()) {
162             RecoverableTransactionIdentifier globalTransaction = (RecoverableTransactionIdentifier) raRemoteTx.getGlobalTransaction();
163             if (internalId.equals(globalTransaction.getInternalId())) {
164                Xid xid = globalTransaction.getXid();
165                log.tracef("Found transaction xid %s that maps internal id %s", xid, internalId);
166                removeRecoveryInformation(xid);
167                return raRemoteTx;
168             }
169          }
170       }
171       log.tracef("Could not find tx to map to internal id %s", internalId);
172       return null;
173    }

175    @Override
176    public List<Xid> getInDoubtTransactions() {
177       List<Xid> result = new ArrayList<Xid>();
178       Set<RecoveryInfoKey> recoveryInfoKeys = inDoubtTransactions.keySet();
179       for (RecoveryInfoKey key : recoveryInfoKeys) {
180          if (key.cacheName.equals(cacheName))
181             result.add(key.xid);
182       }
183       log.tracef("Returning %s ", result);
184       return result;
185    }

187    @Override
188    public Set<InDoubtTxInfo> getInDoubtTransactionInfo() {
189       List<Xid> txs = getInDoubtTransactions();
190       Set<RecoveryAwareLocalTransaction> localTxs = txTable.getLocalTxThatFailedToComplete();
191       log.tracef("Local transactions that failed to complete is %s", localTxs);
192       Set<InDoubtTxInfo> result = new HashSet<InDoubtTxInfo>();
193       for (RecoveryAwareLocalTransaction r : localTxs) {
194          long internalId = ((RecoverableTransactionIdentifier) r.getGlobalTransaction()).getInternalId();
195          result.add(new InDoubtTxInfoImpl(r.getXid(), internalId));
196       }
197       for (Xid xid : txs) {
198          RecoveryAwareRemoteTransaction pTx = getPreparedTransaction(xid);
199          if (pTx == null) continue; //might be removed concurrently, 2check for null
200          RecoverableTransactionIdentifier gtx = (RecoverableTransactionIdentifier) pTx.getGlobalTransaction();
201          InDoubtTxInfoImpl infoInDoubt = new InDoubtTxInfoImpl(xid, gtx.getInternalId(), pTx.getStatus());
202          result.add(infoInDoubt);
203       }
204       log.tracef("The set of in-doubt txs from this node is %s", result);
205       return result;
206    }

208    @Override
209    public Set<InDoubtTxInfo> getInDoubtTransactionInfoFromCluster() {
210       Map<Xid, InDoubtTxInfoImpl> result = new HashMap<Xid, InDoubtTxInfoImpl>();
211       if (rpcManager != null) {
212          GetInDoubtTxInfoCommand inDoubtTxInfoCommand = commandFactory.buildGetInDoubtTxInfoCommand();
213          Map<Address, Response> addressResponseMap = rpcManager.invokeRemotely(null, inDoubtTxInfoCommand,
214                                                                                rpcManager.getDefaultRpcOptions(true));
215          for (Map.Entry<Address, Response> re : addressResponseMap.entrySet()) {
216             Response r = re.getValue();
217             if (!isSuccessful(r)) {
218                throw new CacheException("Could not fetch in doubt transactions: " + r);
219             }
220             Set<InDoubtTxInfoImpl> infoInDoubtSet = (Set<InDoubtTxInfoImpl>) ((SuccessfulResponse) r).getResponseValue();
221             for (InDoubtTxInfoImpl infoInDoubt : infoInDoubtSet) {
222                InDoubtTxInfoImpl inDoubtTxInfo = result.get(infoInDoubt.getXid());
223                if (inDoubtTxInfo == null) {
224                   inDoubtTxInfo = infoInDoubt;
225                   result.put(infoInDoubt.getXid(), inDoubtTxInfo);
226                } else {
227                   inDoubtTxInfo.addStatus(infoInDoubt.getStatus());
228                }
229                inDoubtTxInfo.addOwner(re.getKey());
230             }
231          }
232       }
233       Set<InDoubtTxInfo> onThisNode = getInDoubtTransactionInfo();
234       Iterator<InDoubtTxInfo> iterator = onThisNode.iterator();
235       while (iterator.hasNext()) {
236          InDoubtTxInfo info = iterator.next();
237          InDoubtTxInfoImpl inDoubtTxInfo = result.get(info.getXid());
238          if (inDoubtTxInfo != null) {
239             inDoubtTxInfo.setLocal(true);
240             iterator.remove();
241          } else {
242             ((InDoubtTxInfoImpl)info).setLocal(true);
243          }
244       }
245       HashSet<InDoubtTxInfo> value = new HashSet<InDoubtTxInfo>(result.values());
246       value.addAll(onThisNode);
247       return value;
248    }

250    public void registerInDoubtTransaction(RecoveryAwareRemoteTransaction remoteTransaction) {
251       Xid xid = ((RecoverableTransactionIdentifier)remoteTransaction.getGlobalTransaction()).getXid();
252       RecoveryAwareTransaction previous = inDoubtTransactions.put(new RecoveryInfoKey(xid, cacheName), remoteTransaction);
253       if (previous != null) {
254          log.preparedTxAlreadyExists(previous, remoteTransaction);
255          throw new IllegalStateException("Are there two different transactions having same Xid in the cluster?");
256       }
257    }

265    @Override
266    public String forceTransactionCompletion(Xid xid, boolean commit) {
267       //this means that we have this as a local transaction that originated here
268       LocalXaTransaction localTransaction = txTable.getLocalTransaction(xid);
269       if (localTransaction != null) {
270          localTransaction.clearRemoteLocksAcquired();
271          return completeTransaction(localTransaction, commit, xid);
272       } else {
273          RecoveryAwareRemoteTransaction tx = getPreparedTransaction(xid);
274          if (tx == null) return "Could not find transaction " + xid;
275          GlobalTransaction globalTransaction = tx.getGlobalTransaction();
276          globalTransaction.setAddress(rpcManager.getAddress());
277          globalTransaction.setRemote(false);
278          RecoveryAwareLocalTransaction localTx = (RecoveryAwareLocalTransaction) txFactory.newLocalTransaction(null, globalTransaction, false, tx.getTopologyId());
279          localTx.setModifications(tx.getModifications());
280          localTx.setXid(xid);
281          localTx.addAllAffectedKeys(tx.getAffectedKeys());
282          for (Object lk : tx.getLockedKeys()) localTx.registerLockedKey(lk);
283          return completeTransaction(localTx, commit, xid);
284       }
285    }

287    private String completeTransaction(LocalTransaction localTx, boolean commit, Xid xid) {
288       if (commit) {
289          try {
290             localTx.clearLookedUpEntries();
291             txCoordinator.prepare(localTx, true);
292             txCoordinator.commit(localTx, false);
293          } catch (XAException e) {
294             log.warnCouldNotCommitLocalTx(localTx, e);
295             return "Could not commit transaction " + xid + " : " + e.getMessage();
296          }
297       } else {
298          try {
299             txCoordinator.rollback(localTx);
300          } catch (XAException e) {
301             log.warnCouldNotRollbackLocalTx(localTx, e);
302             return "Could not commit transaction " + xid + " : " + e.getMessage();
303          }
304       }
305       removeRecoveryInformationFromCluster(null, xid, false, localTx.getGlobalTransaction());
306       return commit ? "Commit successful!" : "Rollback successful";
307    }

309    @Override
310    public String forceTransactionCompletionFromCluster(Xid xid, Address where, boolean commit) {
311       CompleteTransactionCommand ctc = commandFactory.buildCompleteTransactionCommand(xid, commit);
312       Map<Address, Response> responseMap = rpcManager.invokeRemotely(Collections.singleton(where), ctc, rpcManager.getDefaultRpcOptions(true));
313       if (responseMap.size() != 1 || responseMap.get(where) == null) {
314          log.expectedJustOneResponse(responseMap);
315          throw new CacheException("Expected response size is 1, received " + responseMap);
316       }
317       return (String) ((SuccessfulResponse) responseMap.get(where)).getResponseValue();
318    }

320    @Override
321    public boolean isTransactionPrepared(GlobalTransaction globalTx) {
322       Xid xid = ((RecoverableTransactionIdentifier) globalTx).getXid();
323       RecoveryAwareRemoteTransaction remoteTransaction = (RecoveryAwareRemoteTransaction) txTable.getRemoteTransaction(globalTx);
324       boolean remotePrepared = remoteTransaction != null && remoteTransaction.isPrepared();
325       boolean result = inDoubtTransactions.get(new RecoveryInfoKey(xid, cacheName)) != null //if it is in doubt must be prepared
326             || txTable.getLocalPreparedXids().contains(xid) || remotePrepared;
327       if (log.isTraceEnabled()) log.tracef("Is tx %s prepared? %s", xid, result);
328       return result;
329    }

331    private boolean isSuccessful(Response thisResponse) {
332       return thisResponse != null && thisResponse.isValid() && thisResponse.isSuccessful();
333    }

335    private boolean notOnlyMeInTheCluster() {
336       return rpcManager != null && rpcManager.getTransport().getMembers().size() > 1;
337    }

339    private Map<Address, Response> getAllPreparedTxFromCluster() {
340       GetInDoubtTransactionsCommand command = commandFactory.buildGetInDoubtTransactionsCommand();
341       Map<Address, Response> addressResponseMap = rpcManager.invokeRemotely(null, command, rpcManager.getDefaultRpcOptions(true));
342       if (log.isTraceEnabled()) log.tracef("getAllPreparedTxFromCluster received from cluster: %s", addressResponseMap);
343       return addressResponseMap;
344    }
