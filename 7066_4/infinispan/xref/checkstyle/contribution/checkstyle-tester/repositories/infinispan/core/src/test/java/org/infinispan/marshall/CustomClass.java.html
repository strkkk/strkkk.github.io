1   package org.infinispan.marshall;
2

12     public String getVal() {
13        return val;
14     }
15  
16     @Override
17     public String toString() {
18        return "CustomClass{" +
19              "val='" + val + '\'' +
20              '}';
21     }
22  
23     @Override
24     public boolean equals(Object o) {
25        if (this == o) return true;
26        if (o == null || getClass() != o.getClass()) return false;
27  
28        CustomClass that = (CustomClass) o;
29  
30        if (val != null ? !val.equals(that.val) : that.val != null) return false;
31  
32        return true;
33     }
34  
35     @Override
36     public int hashCode() {
37        return val != null ? val.hashCode() : 0;
38     }
39  }
