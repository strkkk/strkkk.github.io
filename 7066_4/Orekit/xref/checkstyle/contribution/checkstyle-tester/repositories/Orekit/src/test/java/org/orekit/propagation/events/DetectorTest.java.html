55      @Test
56      public void testBasicScheduling() throws OrekitException {
57  
58          final TimeScale utc = TimeScalesFactory.getUTC();
59          final Vector3D position = new Vector3D(-6142438.668, 3492467.56, -25767.257);
60          final Vector3D velocity = new Vector3D(505.848, 942.781, 7435.922);
61          final AbsoluteDate date = new AbsoluteDate(2003, 9, 16, utc);
62          final Orbit orbit = new CircularOrbit(new PVCoordinates(position,  velocity),
63                                                FramesFactory.getEME2000(), date, mu);
64  
65          Propagator propagator = new KeplerianPropagator(orbit);
66          double stepSize = 60.0;
67          OutOfOrderChecker checker = new OutOfOrderChecker(stepSize);
68          propagator.addEventDetector(new DateDetector(date.shiftedBy(5.25 * stepSize)).withHandler(checker));
69          propagator.setMasterMode(stepSize, checker);
70          propagator.propagate(date.shiftedBy(10 * stepSize));
71          Assert.assertTrue(checker.outOfOrderCallDetected());
72  
73      }

75      private static class OutOfOrderChecker implements EventHandler<DateDetector>, OrekitFixedStepHandler {
76  
77          private AbsoluteDate triggerDate;
78          private boolean outOfOrderCallDetected;
79          private double stepSize;
80  
81          public OutOfOrderChecker(final double stepSize) {
82              triggerDate = null;
83              outOfOrderCallDetected = false;
84              this.stepSize = stepSize;
85          }
86  
87          public Action eventOccurred(SpacecraftState s, DateDetector detector, boolean increasing) {
88              triggerDate = s.getDate();
89              return Action.CONTINUE;
90          }
91  
92          public void handleStep(SpacecraftState currentState, boolean isLast) {
93              // step handling and event occurrences may be out of order up to one step
94              // with variable steps, and two steps with fixed steps (due to the delay
95              // induced by StepNormalizer)
96              if (triggerDate != null) {
97                  double dt = currentState.getDate().durationFrom(triggerDate);
98                  if (dt < 0) {
99                      outOfOrderCallDetected = true;
100                     Assert.assertTrue(FastMath.abs(dt) < (2 * stepSize));
101                 }
102             }
103         }
104 
105         public boolean outOfOrderCallDetected() {
106             return outOfOrderCallDetected;
107         }
108 
109         @Override
110         public void init(SpacecraftState initialState, AbsoluteDate target) {
111             EventHandler.super.init(initialState, target);
112         }
113 
114     }

116     @Test
117     public void testIssue108Numerical() throws OrekitException {
118         final TimeScale utc = TimeScalesFactory.getUTC();
119         final Vector3D position = new Vector3D(-6142438.668, 3492467.56, -25767.257);
120         final Vector3D velocity = new Vector3D(505.848, 942.781, 7435.922);

129         propagator.addEventDetector(counter);
130         propagator.propagate(date.shiftedBy(n * step));
131         Assert.assertEquals(n + 1, counter.getCount());

184     @Test
185     public void testNoisyGFunction() throws OrekitException {
186 
187         // initial conditions
188         Frame eme2000 = FramesFactory.getEME2000();
189         TimeScale utc = TimeScalesFactory.getUTC();
190         AbsoluteDate initialDate   = new AbsoluteDate(2011, 5, 11, utc);
191         AbsoluteDate startDate     = new AbsoluteDate(2032, 10, 17, utc);
192         AbsoluteDate interruptDate = new AbsoluteDate(2032, 10, 18, utc);
193         AbsoluteDate targetDate    = new AbsoluteDate(2211, 5, 11, utc);
194         KeplerianPropagator k1 =
195                 new KeplerianPropagator(new EquinoctialOrbit(new PVCoordinates(new Vector3D(4008462.4706055815, -3155502.5373837613, -5044275.9880020910),
196                                                                                new Vector3D(-5012.9298276860990, 1920.3567095973078, -5172.7403501801580)),
197                                                              eme2000, initialDate, Constants.WGS84_EARTH_MU));
198         KeplerianPropagator k2 =
199                 new KeplerianPropagator(new EquinoctialOrbit(new PVCoordinates(new Vector3D(4008912.4039522274, -3155453.3125615157, -5044297.6484738905),
200                                                                                new Vector3D(-5012.5883854112530, 1920.6332221785074, -5172.2177085540500)),
201                                                              eme2000, initialDate, Constants.WGS84_EARTH_MU));
202         k2.addEventDetector(new CloseApproachDetector(2015.243454166727, 0.0001, 100,
203                                                       new ContinueOnEvent<CloseApproachDetector>(),
204                                                       k1));
205         k2.addEventDetector(new DateDetector(Constants.JULIAN_DAY, 1.0e-6, interruptDate));
206         SpacecraftState s = k2.propagate(startDate, targetDate);
207         Assert.assertEquals(0.0, interruptDate.durationFrom(s.getDate()), 1.1e-6);
208     }

210     private static class CloseApproachDetector extends AbstractDetector<CloseApproachDetector> {
211 
212         private static final long serialVersionUID = 1L;
213         private final PVCoordinatesProvider provider;
214 
215         public CloseApproachDetector(double maxCheck, double threshold,
216                                      final int maxIter, final EventHandler<? super CloseApproachDetector> handler,
217                                      PVCoordinatesProvider provider) {
218             super(maxCheck, threshold, maxIter, handler);
219             this.provider = provider;
220         }
221 
222         public double g(final SpacecraftState s) throws OrekitException {
223             PVCoordinates pv1     = provider.getPVCoordinates(s.getDate(), s.getFrame());
224             PVCoordinates pv2     = s.getPVCoordinates();
225             Vector3D deltaP       = pv1.getPosition().subtract(pv2.getPosition());
226             Vector3D deltaV       = pv1.getVelocity().subtract(pv2.getVelocity());
227             double radialVelocity = Vector3D.dotProduct(deltaP.normalize(), deltaV);
228             return radialVelocity;
229         }
230 
231         protected CloseApproachDetector create(final double newMaxCheck, final double newThreshold,
232                                                final int newMaxIter,
233                                                final EventHandler<? super CloseApproachDetector> newHandler) {
234             return new CloseApproachDetector(newMaxCheck, newThreshold, newMaxIter, newHandler,
235                                              provider);
236         }
237 
238     }

240     @Test
241     public void testWrappedException() throws OrekitException {
242         final Throwable dummyCause = new RuntimeException();
243         try {
244             // initial conditions
245             Frame eme2000 = FramesFactory.getEME2000();
246             TimeScale utc = TimeScalesFactory.getUTC();
247             final AbsoluteDate initialDate   = new AbsoluteDate(2011, 5, 11, utc);
248             final AbsoluteDate exceptionDate = initialDate.shiftedBy(3600.0);
249             KeplerianPropagator k =
250                     new KeplerianPropagator(new EquinoctialOrbit(new PVCoordinates(new Vector3D(4008462.4706055815, -3155502.5373837613, -5044275.9880020910),
251                                                                                    new Vector3D(-5012.9298276860990, 1920.3567095973078, -5172.7403501801580)),
252                                                                  eme2000, initialDate, Constants.WGS84_EARTH_MU));
253             k.addEventDetector(new DateDetector(initialDate.shiftedBy(Constants.JULIAN_DAY)) {
254                 private static final long serialVersionUID = 1L;
255                 @Override
256                 public double g(final SpacecraftState s) throws OrekitException {
257                     final double dt = s.getDate().durationFrom(exceptionDate);
258                     if (FastMath.abs(dt) < 1.0) {
259                         throw new OrekitException(dummyCause, LocalizedCoreFormats.SIMPLE_MESSAGE, "dummy");
260                     }
261                     return dt;
262                 }
263             });
264             k.propagate(initialDate.shiftedBy(Constants.JULIAN_YEAR));
265             Assert.fail("an exception should have been thrown");
266         } catch (OrekitException oe) {
267             Assert.assertSame(dummyCause, oe.getCause());
268         }
269     }
