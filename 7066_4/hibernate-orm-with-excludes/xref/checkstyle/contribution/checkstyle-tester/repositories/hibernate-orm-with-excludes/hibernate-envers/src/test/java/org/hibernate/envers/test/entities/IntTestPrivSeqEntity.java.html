1   package org.hibernate.envers.test.entities;
2   
3   import javax.persistence.Column;
4   import javax.persistence.Entity;
5   import javax.persistence.GeneratedValue;
6   import javax.persistence.GenerationType;
7   import javax.persistence.Id;
8   import javax.persistence.SequenceGenerator;
9   
10  import org.hibernate.envers.Audited;
11  
12  /**
13   * Duplicate of {@link IntTestEntity} but with private sequence generator.
14   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
15   */
16  @Entity
17  public class IntTestPrivSeqEntity {
18      @Id
19      @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "IntTestPrivSeq")
20      @SequenceGenerator(name = "IntTestPrivSeq", sequenceName="INTTESTPRIV_SEQ",
21                         allocationSize = 1, initialValue = 1)
22      private Integer id;
23  
24      @Audited
25      @Column(name = "NUMERIC_VALUE")
26      private Integer number;
27  
28      public IntTestPrivSeqEntity() {
29      }
30  
31      public IntTestPrivSeqEntity(Integer number, Integer id) {
32          this.id = id;
33          this.number = number;
34      }
35  
36      public IntTestPrivSeqEntity(Integer number) {
37          this.number = number;
38      }
39  
40      public Integer getId() {
41          return id;
42      }
43  
44      public void setId(Integer id) {
45          this.id = id;
46      }
47  
48      public Integer getNumber() {
49          return number;
50      }
51  
52      public void setNumber(Integer number) {
53          this.number = number;
54      }
55  
56      public boolean equals(Object o) {
57          if (this == o) return true;
58          if (!(o instanceof IntTestPrivSeqEntity)) return false;
59  
60          IntTestPrivSeqEntity that = (IntTestPrivSeqEntity) o;
61  
62          if (id != null ? !id.equals(that.id) : that.id != null) return false;
63          if (number != null ? !number.equals(that.number) : that.number != null) return false;
64  
65          return true;
66      }
67  
68      public int hashCode() {
69          int result = (id != null ? id.hashCode() : 0);
70          result = 31 * result + (number != null ? number.hashCode() : 0);
71          return result;
72      }
73  
74      public String toString() {
75          return "ITPSE(id = " + id + ", number = " + number + ")";
76      }
77  }
