2   package org.hibernate.test.annotations.cid;
3   import java.io.Serializable;
4   import javax.persistence.Embeddable;
5   import javax.persistence.JoinColumn;
6   import javax.persistence.JoinColumns;
7   import javax.persistence.ManyToOne;
8   
9   /**
10   * Child Pk with many to one inside
11   *
12   * @author Emmanuel Bernard
13   */
14  @Embeddable
15  public class ChildPk implements Serializable {
16  	public int nthChild;
17  	@ManyToOne()
18  	@JoinColumns({
19  	@JoinColumn(name = "parentLastName", referencedColumnName = "p_lname", nullable = false),
20  	@JoinColumn(name = "parentFirstName", referencedColumnName = "firstName", nullable = false)
21  			})
22  	public Parent parent;
23  
24  	public boolean equals(Object o) {
25  		if ( this == o ) return true;
26  		if ( !( o instanceof ChildPk ) ) return false;
27  
28  		final ChildPk childPk = (ChildPk) o;
29  
30  		if ( nthChild != childPk.nthChild ) return false;
31  		if ( !parent.equals( childPk.parent ) ) return false;
32  
33  		return true;
34  	}
35  
36  	public int hashCode() {
37  		int result;
38  		result = nthChild;
39  		result = 29 * result + parent.hashCode();
40  		return result;
41  	}
42  }
