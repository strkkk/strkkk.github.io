30  import java.util.Map;

34  import org.w3c.dom.NodeList;

36  import com.apple.internal.jobjc.generator.RestrictedKeywords;
37  import com.apple.internal.jobjc.generator.classes.JObjCClassClassFile;
38  import com.apple.internal.jobjc.generator.classes.JObjCClassFile;

40  import com.apple.jobjc.MacOSXFramework;
41  import com.apple.jobjc.SuperClassExtractor;
42  import com.apple.jobjc.UnsafeRuntimeAccess;
43  import com.apple.jobjc.NSClass.NSClassNotFoundException;
44  import com.apple.jobjc.SEL;
45  
46  public class Clazz extends Element<Framework> implements OutputFileGenerator {
47      private final Map<String, Method> instanceMethodsByName = new HashMap<String, Method>();
48      private final Map<String, Method> classMethodsByName = new HashMap<String, Method>();
49  
50      public final List<Method> classMethods;
51      public final List<Method> instanceMethods;
52  
53      public final List<Clazz> subClassers = new ArrayList<Clazz>(0);
54      public Clazz superClass;
55  
56      public Clazz(String name, List<Method> classMethods, List<Method> instanceMethods, Clazz superClass, Framework parent){
57          super(name, parent);
58          this.classMethods = classMethods;
59          this.instanceMethods = instanceMethods;
60          this.superClass = superClass;
61      }
62  
63      public Clazz(final Node classNode, final Framework parent) {
64          super(classNode, parent);
65          this.classMethods = new ArrayList<Method>();
66          this.instanceMethods = new ArrayList<Method>();
67  
68          final NodeList methodNodes = classNode.getChildNodes();
69          for (int i = 0; i < methodNodes.getLength(); i++) {
70              final Node node = methodNodes.item(i);
71              if (!"method".equals(node.getLocalName())) continue;
72  
73              final String selName = Element.getAttr(node, "selector");
74              if(selName == null || !SEL.validName(selName)){
75                  System.err.format("Warning: Discarding method %1$s:%2$s:%3$s"
76                          + " -- Invalid selector name. Verify.\n",
77                          parent.name, name, selName);
78                  continue;
79              }
80  
81              final Method method = new Method(node, parent);
82              if (method.isClassMethod) {
83                  classMethods.add(method);
84              } else {
85                  instanceMethods.add(method);
86              }
87          }
88      }
89  
90      public String getPackage() {
91          return parent.pkg;
92      }
93  
94      public String getFullPath(){
95          return parent.pkg + "." + name;
96      }
97  
98      @Override
99      public String toString() {
100         return super.toString() + " " + classMethods + " " + instanceMethods;
101     }
102 
103     public boolean doesActuallyExist(){
104         try{
105             UnsafeRuntimeAccess.getNSClass(parent.load(), name);
106         }catch(NSClassNotFoundException x){
107             return false;
108         }
109         return true;
110     }
111 
112     void resolveSuperClass(final MacOSXFramework nativeFramework, final Map<String, Clazz> allClasses) throws Throwable {
113         superClass = SuperClassExtractor.getSuperClassFor(name, nativeFramework, allClasses);
114     }
115 
116     public void disambiguateMethods() {
117         disambiguateMethods(instanceMethods, instanceMethodsByName);
118         disambiguateMethods(classMethods, classMethodsByName);
119     }
120 
121     private void disambiguateMethods(final List<Method> methods, final Map<String, Method> methodMap) {
122         final Set<String> existingMethodNames = RestrictedKeywords.getNewRestrictedSet();
123         for (final Method method : methods) {
124             method.disambiguateNameAndArgs(this, existingMethodNames);
125             methodMap.put(method.javaName, method);
126         }
127     }
128 
129     public void generateClasses(final List<OutputFile> generatedClassFiles) {
130         generatedClassFiles.add(new JObjCClassClassFile(this));
131         generatedClassFiles.add(new JObjCClassFile(this));
132     }
133 
134     Method getParentMethodMatchingName(final String methodName) {
135         if(superClass == null) return null;
136         Method m = superClass.getMethodNamed(methodName);
137         if(m != null) return m;
138         return superClass.getParentMethodMatchingName(methodName);
139     }
140 
141     private Method getMethodNamed(final String methodName) {
142         final Method instanceMethod = instanceMethodsByName.get(methodName);
143         if (instanceMethod != null) return instanceMethod;
144 
145         final Method classMethod = classMethodsByName.get(methodName);
146         if (classMethod != null) return classMethod;
147 
148         return null;
149     }
150 
151     int compareTo(Clazz o) {
152       return toString().compareTo(o.toString());
153     }
154 }
