25  package org.hibernate.engine.query.spi;
26  import org.hibernate.QueryException;
27  import org.hibernate.hql.internal.classic.ParserHelper;
28  import org.hibernate.internal.util.StringHelper;
29  
30  /**
31   * The single available method {@link #parse} is responsible for parsing a
32   * query string and recognizing tokens in relation to parameters (either
33   * named, JPA-style, or ordinal) and providing callbacks about such
34   * recognitions.
35   *
36   * @author Steve Ebersole
37   */
38  public class ParameterParser {
39  
40  	public static interface Recognizer {
41  		public void outParameter(int position);
42  		public void ordinalParameter(int position);
43  		public void namedParameter(String name, int position);
44  		public void jpaPositionalParameter(String name, int position);
45  		public void other(char character);
46  	}
47  
48  	/**
49  	 * Direct instantiation of ParameterParser disallowed.
50  	 */
51  	private ParameterParser() {
52  	}
53  
54  	/**
55  	 * Performs the actual parsing and tokenizing of the query string making appropriate
56  	 * callbacks to the given recognizer upon recognition of the various tokens.
57  	 * <p/>
58  	 * Note that currently, this only knows how to deal with a single output
59  	 * parameter (for callable statements).  If we later add support for
60  	 * multiple output params, this, obviously, needs to change.
61  	 *
62  	 * @param sqlString The string to be parsed/tokenized.
63  	 * @param recognizer The thing which handles recognition events.
64  	 * @throws QueryException Indicates unexpected parameter conditions.
65  	 */
66  	public static void parse(String sqlString, Recognizer recognizer) throws QueryException {
67  		boolean hasMainOutputParameter = startsWithEscapeCallTemplate( sqlString );
68  		boolean foundMainOutputParam = false;
69  
70  		int stringLength = sqlString.length();
71  		boolean inQuote = false;
72  		for ( int indx = 0; indx < stringLength; indx++ ) {
73  			char c = sqlString.charAt( indx );
74  			if ( inQuote ) {
75  				if ( '\'' == c ) {
76  					inQuote = false;
77  				}
78  				recognizer.other( c );
79  			}
80  			else if ( '\'' == c ) {
81  				inQuote = true;
82  				recognizer.other( c );
83  			}
84  			else if ( '\\' == c ) {
85  				// skip sending the backslash and instead send then next character, treating is as a literal
86  				recognizer.other( sqlString.charAt( ++indx ) );
87  			}
88  			else {
89  				if ( c == ':' ) {
90  					// named parameter
91  					int right = StringHelper.firstIndexOfChar( sqlString, ParserHelper.HQL_SEPARATORS_BITSET, indx + 1 );
92  					int chopLocation = right < 0 ? sqlString.length() : right;
93  					String param = sqlString.substring( indx + 1, chopLocation );
94  					if ( StringHelper.isEmpty( param ) ) {
95  						throw new QueryException(
96  								"Space is not allowed after parameter prefix ':' [" + sqlString + "]"
97  						);
98  					}
99  					recognizer.namedParameter( param, indx );
100 					indx = chopLocation - 1;
101 				}
102 				else if ( c == '?' ) {
103 					// could be either an ordinal or JPA-positional parameter
104 					if ( indx < stringLength - 1 && Character.isDigit( sqlString.charAt( indx + 1 ) ) ) {
105 						// a peek ahead showed this as an JPA-positional parameter
106 						int right = StringHelper.firstIndexOfChar( sqlString, ParserHelper.HQL_SEPARATORS, indx + 1 );
107 						int chopLocation = right < 0 ? sqlString.length() : right;
108 						String param = sqlString.substring( indx + 1, chopLocation );
109 						// make sure this "name" is an integral
110 						try {
111                             Integer.valueOf( param );
112 						}
113 						catch( NumberFormatException e ) {
114 							throw new QueryException( "JPA-style positional param was not an integral ordinal" );
115 						}
116 						recognizer.jpaPositionalParameter( param, indx );
117 						indx = chopLocation - 1;
118 					}
119 					else {
120 						if ( hasMainOutputParameter && !foundMainOutputParam ) {
121 							foundMainOutputParam = true;
122 							recognizer.outParameter( indx );
123 						}
124 						else {
125 							recognizer.ordinalParameter( indx );
126 						}
127 					}
128 				}
129 				else {
130 					recognizer.other( c );
131 				}
132 			}
133 		}
134 	}
135 
136 	public static boolean startsWithEscapeCallTemplate(String sqlString) {
137 		if ( ! ( sqlString.startsWith( "{" ) && sqlString.endsWith( "}" ) ) ) {
138 			return false;
139 		}
140 
141 		int chopLocation = sqlString.indexOf( "call" );
142 		if ( chopLocation <= 0 ) {
143 			return false;
144 		}
145 
146 		final String checkString = sqlString.substring( 1, chopLocation + 4 );
147 		final String fixture = "?=call";
148 		int fixturePosition = 0;
149 		boolean matches = true;
150 		for ( int i = 0, max = checkString.length(); i < max; i++ ) {
151 			final char c = Character.toLowerCase( checkString.charAt( i ) );
152 			if ( Character.isWhitespace( c ) ) {
153 				continue;
154 			}
155 			if ( c == fixture.charAt( fixturePosition ) ) {
156 				fixturePosition++;
157 				continue;
158 			}
159 			matches = false;
160 			break;
161 		}
162 
163 		return matches;
164 	}
