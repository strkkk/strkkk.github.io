50     private static class FieldPropertyAccessor extends BasePropertyAccessor {
51  
52        protected final Field field;
53  
54        FieldPropertyAccessor(Field field) {
55           this.field = field;
56        }
57  
58        @Override
59        public Class<?> getPropertyType() {
60           return field.getType();
61        }
62  
63        public boolean isMultiple() {
64           return false;
65        }

68           try {
69              return field.get(instance);
70           } catch (IllegalAccessException e) {
71              throw new RuntimeException(e);
72           }
73        }
74  
75        public Iterator<Object> getValueIterator(Object instance) {
76           throw new UnsupportedOperationException("This property cannot be iterated");
77        }

241    public static PropertyAccessor getAccessor(Class<?> clazz, String propertyName) {
242       if (propertyName == null || propertyName.length() == 0) {
243          throw new IllegalArgumentException("Property name cannot be null or empty");
244       }
245       if (propertyName.contains(".")) {
246          throw new IllegalArgumentException("The argument cannot be a nested property");
247       }
248 
249       // try getter method access
250       // we need to find a no-arg public "getXyz" or "isXyz" method which has a suitable return type
251       String propertyNameSuffix = Character.toUpperCase(propertyName.charAt(0)) + propertyName.substring(1);
252       try {
253          Method m = clazz.getDeclaredMethod("get" + propertyNameSuffix);
254          if (Modifier.isPublic(m.getModifiers()) && !m.isSynthetic() && !m.isBridge()
255                && !m.getReturnType().equals(Void.class)) {
256             return getMethodAccessor(m);
257          }
258       } catch (NoSuchMethodException e) {
259          try {
260             Method m = clazz.getDeclaredMethod("is" + propertyNameSuffix);
261             if (Modifier.isPublic(m.getModifiers()) && !m.isSynthetic() && !m.isBridge()
262                   && (m.getReturnType().equals(boolean.class) || m.getReturnType().equals(Boolean.class))) {
263                return getMethodAccessor(m);
264             }
265          } catch (NoSuchMethodException e1) {
266             // ignored, continue
267          }
268       }
269 
270       // try field access
271       try {
272          Field f = clazz.getDeclaredField(propertyName);
273          if (f != null && !f.isSynthetic()) {
274             return getFieldAccessor(f);
275          }
276       } catch (NoSuchFieldException e) {
277          // ignored, continue
278       }
279 
280       return null;  //todo throw property not found exception
281    }

283    private static PropertyAccessor getFieldAccessor(Field f) {
284       f.setAccessible(true);
285       Class<?> fieldClass = f.getType();
286       if (fieldClass.isArray()) {
287          return new ArrayFieldPropertyAccessor(f);
288       } else if (Collection.class.isAssignableFrom(fieldClass)) {
289          return new CollectionFieldPropertyAccessor(f);
290       } else if (Map.class.isAssignableFrom(fieldClass)) {
291          return new MapFieldPropertyAccessor(f);
292       }
293       return new FieldPropertyAccessor(f);
294    }

296    private static PropertyAccessor getMethodAccessor(Method m) {
297       Class<?> fieldClass = m.getReturnType();
298       if (fieldClass.isArray()) {
299          return new ArrayMethodPropertyAccessor(m);
300       } else if (Collection.class.isAssignableFrom(fieldClass)) {
301          return new CollectionMethodPropertyAccessor(m);
302       } else if (Map.class.isAssignableFrom(fieldClass)) {
303          return new MapMethodPropertyAccessor(m);
304       }
305       return new MethodPropertyAccessor(m);
306    }

308    private static Class determineElementType(Class<?> type, Type genericType) {
309       if (type.isArray()) {
310          if (genericType instanceof Class) {
311             return type.getComponentType();
312          }
313          GenericArrayType genericArrayType = (GenericArrayType) genericType;
314          Type genericComponentType = genericArrayType.getGenericComponentType();
315          if (genericComponentType instanceof ParameterizedType) {
316             return (Class) ((ParameterizedType) genericComponentType).getRawType();
317          } else {
318             return (Class) ((TypeVariable) genericComponentType).getBounds()[0];
319          }
320       } else if (Collection.class.isAssignableFrom(type)) {
321          return determineCollectionElementType(genericType);
322       } else if (Map.class.isAssignableFrom(type)) {
323          return determineMapValueTypeParam(genericType);
324       }
325       return null;
326    }

328    private static Class determineMapValueTypeParam(Type genericType) {
329       if (genericType instanceof ParameterizedType) {
330          ParameterizedType type = (ParameterizedType) genericType;
331          Type fieldArgType = type.getActualTypeArguments()[1];
332          if (fieldArgType instanceof TypeVariable) {
333             TypeVariable genericComponentType = (TypeVariable) fieldArgType;
334             return (Class) genericComponentType.getBounds()[0];
335          } else {
336             return (Class) fieldArgType;
337          }
338       } else if (genericType instanceof Class) {
339          Class c = (Class) genericType;
340          if (c.getGenericSuperclass() != null && Map.class.isAssignableFrom(c.getSuperclass())) {
341             Class x = determineMapValueTypeParam(c.getGenericSuperclass());
342             if (x != null) {
343                return x;
344             }
345          }
346          for (Type t : c.getGenericInterfaces()) {
347             if (t instanceof Class && Map.class.isAssignableFrom((Class<?>) t)
348                   || t instanceof ParameterizedType && Map.class.isAssignableFrom((Class) ((ParameterizedType) t).getRawType())) {
349                Class x = determineMapValueTypeParam(t);
350                if (x != null) {
351                   return x;
352                }
353             }
354          }
355       }
356       return null;
357    }

359    private static Class determineCollectionElementType(Type genericType) {
360       if (genericType instanceof ParameterizedType) {
361          ParameterizedType type = (ParameterizedType) genericType;
362          Type fieldArgType = type.getActualTypeArguments()[0];
363          if (fieldArgType instanceof Class) {
364             return (Class) fieldArgType;
365          }
366          return (Class) ((ParameterizedType) fieldArgType).getRawType();
367       } else if (genericType instanceof Class) {
368          Class c = (Class) genericType;
369          if (c.getGenericSuperclass() != null && Collection.class.isAssignableFrom(c.getSuperclass())) {
370             Class x = determineCollectionElementType(c.getGenericSuperclass());
371             if (x != null) {
372                return x;
373             }
374          }
375          for (Type t : c.getGenericInterfaces()) {
376             if (t instanceof Class && Map.class.isAssignableFrom((Class<?>) t)
377                   || t instanceof ParameterizedType && Collection.class.isAssignableFrom((Class) ((ParameterizedType) t).getRawType())) {
378                Class x = determineCollectionElementType(t);
379                if (x != null) {
380                   return x;
381                }
382             }
383          }
384       }
385       return null;
386    }
