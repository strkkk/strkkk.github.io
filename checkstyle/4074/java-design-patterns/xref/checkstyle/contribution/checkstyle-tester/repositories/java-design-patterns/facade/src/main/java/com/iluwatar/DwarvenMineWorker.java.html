1   package com.iluwatar;
2   
3   /**
4    * 
5    * DwarvenMineWorker is one of the goldmine subsystems.
6    *
7    */
8   public abstract class DwarvenMineWorker {
9   
10  	public void goToSleep() {
11  		System.out.println(name() + " goes to sleep.");
12  	}
13  
14  	public void wakeUp() {
15  		System.out.println(name() + " wakes up.");
16  	}
17  
18  	public void goHome() {
19  		System.out.println(name() + " goes home.");
20  	}
21  
22  	public void goToMine() {
23  		System.out.println(name() + " goes to the mine.");
24  	}
25  
26      private void action(Action action) {
27          switch (action) {
28              case GO_TO_SLEEP:
29                  goToSleep();
30                  break;
31              case WAKE_UP:
32                  wakeUp();
33                  break;
34              case GO_HOME:
35                  goHome();
36                  break;
37              case GO_TO_MINE:
38                  goToMine();
39                  break;
40              case WORK:
41                  work();
42                  break;
43              default:
44                  System.out.println("Undefined action");
45                  break;
46          }
47      }
48  
49      public void action(Action... actions) {
50          for (Action action : actions) {
51              action(action);
52          }
53      }
54  
55  	public abstract void work();
56  
57  	public abstract String name();
58  
59      static enum Action {
60          GO_TO_SLEEP, WAKE_UP, GO_HOME, GO_TO_MINE, WORK
61      }
62  }
