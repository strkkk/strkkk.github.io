19  package de.tobject.findbugs.view.test;
20  
21  import static org.junit.Assert.assertNotNull;
22  
23  import org.eclipse.core.runtime.CoreException;
24  import org.eclipse.jface.viewers.ITreeContentProvider;
25  import org.junit.AfterClass;
26  import org.junit.BeforeClass;
27  import org.junit.Test;
28  
29  import de.tobject.findbugs.test.TestScenario;
30  import de.tobject.findbugs.view.explorer.GroupType;
31  
32  /**
33   * This class tests the BugExplorerView and its related classes. This tests a
34   * different scenario with two source folders.
35   *
36   * @author Tomás Pollak
37   */
38  public class BugExplorerViewTwoSrcFoldersTest extends AbstractBugExplorerViewTest {
39      @BeforeClass
40      public static void setUpClass() throws Exception {
41          setUpTestProject(TestScenario.TWO_SRC_FOLDERS);
42      }
43  
44      @AfterClass
45      public static void tearDownClass() throws CoreException {
46          tearDownTestProject();
47      }
48  
49      @Test
50      public void testViewContents_runTwice_onceOnSrcFolder() throws CoreException {
51          // Run Findbugs on the entire project
52          work(createFindBugsWorker());
53  
54          // Get the content provider
55          ITreeContentProvider contentProvider = getNavigatorContentProvider();
56          assertNotNull(contentProvider);
57          setGroupingInBugContentProvider(getDefaultGrouping());
58  
59          // Assert expected view contents
60          Object projectBugGroup = getSingleElement(contentProvider);
61          ExpectedViewElement expectedProject = createProjectPatternMarkerExpectedContents();
62          expectedProject.assertEquals(projectBugGroup, contentProvider);
63  
64          // Run Findbugs on the first source folder
65          work(createFindBugsWorker(), getDefaultPackageInSrc());
66  
67          // Assert expected view contents
68          projectBugGroup = getSingleElement(contentProvider);
69          expectedProject = createProjectPatternMarkerExpectedContents();
70          expectedProject.assertEquals(projectBugGroup, contentProvider);
71      }
72  
73      @Override
74      protected TestScenario getTestScenario() {
75          return TestScenario.TWO_SRC_FOLDERS;
76      }
77  
78      private ExpectedViewElement createProjectPatternMarkerExpectedContents() {
79          // Creates the expected elements tree
80          // Project
81          // -- Pattern
82          // ---- Marker
83          // -- Pattern
84          // ---- Marker
85          // -- Pattern
86          // ---- Marker
87          ExpectedViewElement expectedMarker1 = new ExpectedViewMarker("DM_STRING_CTOR");
88          ExpectedViewElement expectedMarker2 = new ExpectedViewMarker("URF_UNREAD_FIELD");
89          ExpectedViewElement expectedMarker3 = new ExpectedViewMarker("DM_NUMBER_CTOR");
90          ExpectedViewElement expectedPattern1 = new ExpectedViewBugGroup(GroupType.Pattern, null, setOf(expectedMarker1),
91                  setOf(expectedMarker1));
92          ExpectedViewElement expectedPattern2 = new ExpectedViewBugGroup(GroupType.Pattern, null, setOf(expectedMarker2),
93                  setOf(expectedMarker2));
94          ExpectedViewElement expectedPattern3 = new ExpectedViewBugGroup(GroupType.Pattern, null, setOf(expectedMarker3),
95                  setOf(expectedMarker3));
96          ExpectedViewElement expectedProject = new ExpectedViewBugGroup(GroupType.Project, getProject(), setOf(expectedPattern1,
97                  expectedPattern2, expectedPattern3), setOf(expectedMarker1, expectedMarker2, expectedMarker3));
98          return expectedProject;
99      }
