98  	public void testObtainConnectionIds(AbstractServerConnectionFactory serverFactory) throws Exception {
99  		final List<IpIntegrationEvent> events =
100 				Collections.synchronizedList(new ArrayList<IpIntegrationEvent>());
101 		int expectedEvents = serverFactory instanceof TcpNetServerConnectionFactory
102 				? 7  // Listening, + OPEN, CLOSE, EXCEPTION for each side
103 				: 5; // Listening, + OPEN, CLOSE (but we *might* get exceptions, depending on timing).
104 		final CountDownLatch serverListeningLatch = new CountDownLatch(1);
105 		final CountDownLatch eventLatch = new CountDownLatch(expectedEvents);
106 		ApplicationEventPublisher publisher = event -> {
107 			LogFactory.getLog(this.getClass()).trace("Received: " + event);
108 			events.add((IpIntegrationEvent) event);
109 			if (event instanceof TcpConnectionServerListeningEvent) {
110 				serverListeningLatch.countDown();
111 			}
112 			eventLatch.countDown();
113 		};
114 		serverFactory.setBeanName("serverFactory");
115 		serverFactory.setApplicationEventPublisher(publisher);
116 		serverFactory = spy(serverFactory);
117 		final CountDownLatch serverConnectionInitLatch = new CountDownLatch(1);
118 		doAnswer(invocation -> {
119 			Object result = invocation.callRealMethod();
120 			serverConnectionInitLatch.countDown();
121 			return result;
122 		}).when(serverFactory).wrapConnection(any(TcpConnectionSupport.class));
123 		ThreadPoolTaskScheduler scheduler = new ThreadPoolTaskScheduler();
124 		scheduler.setPoolSize(10);
125 		scheduler.afterPropertiesSet();
126 		BeanFactory bf = mock(BeanFactory.class);
127 		when(bf.containsBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME)).thenReturn(true);
128 		when(bf.getBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME, TaskScheduler.class)).thenReturn(scheduler);
129 		serverFactory.setBeanFactory(bf);
130 		TcpReceivingChannelAdapter adapter = new TcpReceivingChannelAdapter();
131 		adapter.setOutputChannel(new NullChannel());
132 		adapter.setConnectionFactory(serverFactory);
133 		adapter.start();
134 		assertThat(serverListeningLatch.await(10, TimeUnit.SECONDS)).as("Listening event not received").isTrue();
135 		assertThat(events.get(0)).isInstanceOf(TcpConnectionServerListeningEvent.class);
136 		assertThat(((TcpConnectionServerListeningEvent) events.get(0)).getPort()).isEqualTo(serverFactory.getPort());
137 		int port = serverFactory.getPort();
138 		TcpNetClientConnectionFactory clientFactory = new TcpNetClientConnectionFactory("localhost", port);
139 		clientFactory.registerListener(message -> false);
140 		clientFactory.setBeanName("clientFactory");
141 		clientFactory.setApplicationEventPublisher(publisher);
142 		clientFactory.start();
143 		TcpConnectionSupport client = clientFactory.getConnection();
144 		List<String> clients = clientFactory.getOpenConnectionIds();
145 		assertThat(clients.size()).isEqualTo(1);
146 		assertThat(clients.contains(client.getConnectionId())).isTrue();
147 		assertThat(serverConnectionInitLatch.await(10, TimeUnit.SECONDS)).as("Server connection failed to register")
148 				.isTrue();
149 		List<String> servers = serverFactory.getOpenConnectionIds();
150 		assertThat(servers.size()).isEqualTo(1);
151 		assertThat(serverFactory.closeConnection(servers.get(0))).isTrue();
152 		servers = serverFactory.getOpenConnectionIds();
153 		assertThat(servers.size()).isEqualTo(0);
154 		int n = 0;
155 		clients = clientFactory.getOpenConnectionIds();
156 		while (n++ < 100 && clients.size() > 0) {
157 			Thread.sleep(100);
158 			clients = clientFactory.getOpenConnectionIds();
159 		}
160 		assertThat(clients.size()).isEqualTo(0);
161 		assertThat(eventLatch.await(10, TimeUnit.SECONDS)).isTrue();
162 		assertThat(events.size())
163 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
164 				.isGreaterThanOrEqualTo(expectedEvents);
165 
166 		FooEvent event = new FooEvent(client, "foo");
167 		client.publishEvent(event);
168 		assertThat(events.size())
169 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
170 				.isGreaterThanOrEqualTo(expectedEvents + 1);
171 
172 		try {
173 			event = new FooEvent(mock(TcpConnectionSupport.class), "foo");
174 			client.publishEvent(event);
175 			fail("Expected exception");
176 		}
177 		catch (IllegalArgumentException e) {
178 			assertThat("Can only publish events with this as the source".equals(e.getMessage())).isTrue();
179 		}
180 
181 		SocketAddress address = serverFactory.getServerSocketAddress();
182 		if (address instanceof InetSocketAddress) {
183 			InetSocketAddress inetAddress = (InetSocketAddress) address;
184 			assertThat(inetAddress.getPort()).isEqualTo(port);
185 		}
186 		serverFactory.stop();
187 		scheduler.shutdown();
188 	}

202 	private void testEarlyClose(final AbstractServerConnectionFactory factory, String property,
203 			String message) throws Exception {
204 		factory.setApplicationEventPublisher(mock(ApplicationEventPublisher.class));
205 		factory.setBeanName("foo");
206 		factory.registerListener(mock(TcpListener.class));
207 		factory.afterPropertiesSet();
208 		Log logger = spy(TestUtils.getPropertyValue(factory, "logger", Log.class));
209 		new DirectFieldAccessor(factory).setPropertyValue("logger", logger);
210 		final CountDownLatch latch1 = new CountDownLatch(1);
211 		final CountDownLatch latch2 = new CountDownLatch(1);
212 		final CountDownLatch latch3 = new CountDownLatch(1);
213 		when(logger.isInfoEnabled()).thenReturn(true);
214 		when(logger.isDebugEnabled()).thenReturn(true);
215 		doAnswer(invocation -> {
216 			latch1.countDown();
217 			// wait until the stop nulls the channel
218 			latch2.await(10, TimeUnit.SECONDS);
219 			return null;
220 		}).when(logger).info(contains("Listening"));
221 		doAnswer(invocation -> {
222 			latch3.countDown();
223 			return null;
224 		}).when(logger).debug(contains(message));
225 		factory.start();
226 		assertThat(latch1.await(10, TimeUnit.SECONDS)).as("missing info log").isTrue();
227 		// stop on a different thread because it waits for the executor
228 		new SimpleAsyncTaskExecutor()
229 				.execute(factory::stop);
230 		int n = 0;
231 		DirectFieldAccessor accessor = new DirectFieldAccessor(factory);
232 		while (n++ < 200 && accessor.getPropertyValue(property) != null) {
233 			Thread.sleep(100);
234 		}
235 		assertThat(n < 200).as("Stop was not invoked in time").isTrue();
236 		latch2.countDown();
237 		assertThat(latch3.await(10, TimeUnit.SECONDS)).as("missing debug log").isTrue();
238 		String expected = "bean 'foo', port=" + factory.getPort() + message;
239 		ArgumentCaptor<String> captor = ArgumentCaptor.forClass(String.class);
240 		verify(logger, atLeast(1)).debug(captor.capture());
241 		assertThat(captor.getAllValues()).contains(expected);
242 		factory.stop();
243 	}

265 	private void healthCheckSuccess(AbstractServerConnectionFactory server, boolean fail) throws InterruptedException {
266 		CountDownLatch serverUp = new CountDownLatch(1);
267 		server.setApplicationEventPublisher(event -> {
268 			if (event instanceof TcpConnectionServerListeningEvent) {
269 				serverUp.countDown();
270 			}
271 		});
272 		server.setBeanFactory(mock(BeanFactory.class));
273 		AtomicReference<TcpConnection> connection = new AtomicReference<>();
274 		server.registerSender(conn -> {
275 			connection.set(conn);
276 		});
277 		AtomicInteger tested = new AtomicInteger();
278 		server.registerListener(msg -> {
279 			if (!(msg instanceof ErrorMessage)) {
280 				String payload = new String((byte[]) msg.getPayload());
281 				if (payload.equals("PING")) {
282 					tested.incrementAndGet();
283 					connection.get().send(new GenericMessage<>(fail ? "PANG" : "PONG"));
284 				}
285 				else {
286 					connection.get().send(msg);
287 				}
288 			}
289 			return false;
290 		});
291 		server.start();
292 		assertThat(serverUp.await(10, TimeUnit.SECONDS)).isTrue();
293 		TcpNetClientConnectionFactory clientFactory = new TcpNetClientConnectionFactory("localhost", server.getPort());
294 		clientFactory.setApplicationEventPublisher(event -> { });
295 		clientFactory.setConnectionTest(conn -> {
296 			CountDownLatch latch = new CountDownLatch(1);
297 			AtomicBoolean result = new AtomicBoolean();
298 			conn.registerTestListener(msg -> {
299 				if (Arrays.equals("PONG".getBytes(), (byte[]) msg.getPayload())) {
300 					result.set(true);
301 				}
302 				latch.countDown();
303 				return false;
304 			});
305 			conn.send(new GenericMessage<>("PING"));
306 			try {
307 				latch.await(10, TimeUnit.SECONDS);
308 			}
309 			catch (InterruptedException e) {
310 				Thread.currentThread().interrupt();
311 			}
312 			return result.get();
313 		});
314 		TcpOutboundGateway gateway = new TcpOutboundGateway();
315 		gateway.setRemoteTimeout(60000);
316 		gateway.setConnectionFactory(clientFactory);
317 		QueueChannel outputChannel = new QueueChannel();
318 		gateway.setOutputChannel(outputChannel);
319 		gateway.start();
320 		if (fail) {
321 			assertThatExceptionOfType(MessagingException.class).isThrownBy(() ->
322 				gateway.handleMessage(new GenericMessage<>("test1")))
323 					.withMessageContaining("Connection test failed for");
324 		}
325 		else {
326 			gateway.handleMessage(new GenericMessage<>("test1"));
327 			Message<?> received = outputChannel.receive(0);
328 			assertThat(received).isNotNull();
329 			assertThat(received.getPayload()).isEqualTo("test1".getBytes());
330 			gateway.handleMessage(new GenericMessage<>("test2"));
331 			received = outputChannel.receive(0);
332 			assertThat(received).isNotNull();
333 			assertThat(received.getPayload()).isEqualTo("test2".getBytes());
334 			assertThat(tested.get()).isEqualTo(1);
335 		}
336 		gateway.stop();
337 		server.stop();
338 	}
