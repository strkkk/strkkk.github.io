20      /**
21       * Constructs an infinite (although in practice bounded by Long.MAX_VALUE) stream of longs 0, 1, 2, 3...
22       * for use as indices.
23       * @return A stream of longs.
24       */
25      public static LongStream indices() {
26          return LongStream.iterate(0L, l -> l + 1);
27      }

29      /**
30       * Zip the source stream together with the stream of indices() to provide a stream of indexed values.
31       * @param source  The source stream.
32       * @param <T> The type over which the source stream streams.
33       * @return A stream of indexed values.
34       */
35      public static <T> Stream<Indexed<T>> zipWithIndex(Stream<T> source) {
36          return zip(indices().mapToObj(Long::valueOf), source, Indexed::index);
37      }

39      /**
40       * Zip together the "left" and "right" streams until either runs out of values.
41       * Each pair of values is combined into a single value using the supplied combiner function.
42       * @param lefts The "left" stream to zip.
43       * @param rights The "right" stream to zip.
44       * @param combiner The function to combine "left" and "right" values.
45       * @param <L> The type over which the "left" stream streams.
46       * @param <R> The type over which the "right" stream streams.
47       * @param <O> The type created by the combiner out of pairs of "left" and "right" values, over which the resulting
48       *           stream streams.
49       * @return A stream of zipped values.
50       */
51      public static <L, R, O> Stream<O> zip(Stream<L> lefts, Stream<R> rights, BiFunction<L, R, O> combiner) {
52          return StreamSupport.stream(ZippingSpliterator.zipping(lefts.spliterator(), rights.spliterator(), combiner), false);
53      }

55      /**
56       * Zip together the "left", "middle" and "right" streams until any stream runs out of values.
57       * Each triple of values is combined into a single value using the supplied combiner function.
58       * @param lefts The "left" stream to zip.
59       * @param middles The "middle" stream to zip.
60       * @param rights The "right" stream to zip.
61       * @param combiner The function to combine "left", "middle" and "right" values.
62       * @param <L> The type over which the "left" stream streams.
63       * @param <M> The type over which the "middle" stream streams.
64       * @param <R> The type over which the "right" stream streams.
65       * @param <O> The type created by the combiner out of triples of "left", "middle" and "right" values, over which the resulting
66       *           stream streams.
67       * @return A stream of zipped values.
68       */
69      public static <L, M, R, O> Stream<O> zip(Stream<L> lefts, Stream<M> middles, Stream<R> rights, TriFunction<L, M, R, O> combiner) {
70          return StreamSupport.stream(TriZippingSpliterator.zipping(
71                  lefts.spliterator(),
72                  middles.spliterator(),
73                  rights.spliterator(),
74                  combiner), false);
75      }

77      private static boolean isSized(int characteristics) {
78          return (characteristics & Spliterator.SIZED) != 0;
79      }

81      /**
82       * Construct a stream which takes values from the source stream for as long as they meet the supplied condition, and stops
83       * as soon as a value is encountered which does not meet the condition.
84       * @param source The source stream.
85       * @param condition The condition to apply to elements of the source stream.
86       * @param <T> The type over which the stream streams.
87       * @return A condition-bounded stream.
88       */
89      public static <T> Stream<T> takeWhile(Stream<T> source, Predicate<T> condition) {
90          return StreamSupport.stream(TakeWhileSpliterator.over(source.spliterator(), condition), false);
91      }

93      /**
94       * Construct a stream which takes values from the source stream until one of them meets the supplied condition,
95       * and then stops.
96       * @param source The source stream.
97       * @param condition The condition to apply to elements of the source stream.
98       * @param <T> The type over which the stream streams.
99       * @return A condition-bounded stream.
100      */
101     public static <T> Stream<T> takeUntil(Stream<T> source, Predicate<T> condition) {
102         return takeWhile(source, condition.negate());
103     }

167      * @param <T> The type over which to stream
168      * @return A stream of lists representing the windows
169      */
170     public static <T> Stream<List<T>> windowed(Stream<T> source, int windowSize, int skip){
171         return StreamSupport.stream(WindowedSpliterator.over(source.spliterator(), windowSize, skip), false);

201     /**
202      * Construct a stream which interleaves the supplied streams, picking items using the supplied selector function.
203      *
204      * The selector function will be passed an array containing one value from each stream, or null if that stream
205      * has no more values, and must return the integer index of the value to accept. That value will become part of the
206      * interleaved stream, and the source stream at that index will advance to the next value.
207      *
208      * See the {@link com.codepoetics.protonpack.selectors.Selectors} class for ready-made selectors for round-robin and sorted
209      * item selection.
210      * @param selector The selector function to use.
211      * @param streams The streams to interleave.
212      * @param <T> The type over which the interleaved streams stream.
213      * @return An interleaved stream.
214      */
215     public static <T> Stream<T> interleave(Function<T[], Integer> selector, Stream<T>... streams) {
216         Spliterator<T>[] spliterators = (Spliterator<T>[]) Stream.of(streams).map(s -> s.spliterator()).toArray(Spliterator[]::new);
217         return StreamSupport.stream(InterleavingSpliterator.interleaving(spliterators, selector), false);
218     }

221      * Construct a stream which interleaves the supplied streams, picking items using the supplied selector function.
222      *
223      * The selector function will be passed an array containing one value from each stream, or null if that stream
224      * has no more values, and must return the integer index of the value to accept. That value will become part of the
225      * interleaved stream, and the source stream at that index will advance to the next value.
226      *
227      * See the {@link com.codepoetics.protonpack.selectors.Selectors} class for ready-made selectors for round-robin and sorted
228      * item selection.
229      * @param selector The selector function to use.
230      * @param streams The streams to interleave.
231      * @param <T> The type over which the interleaved streams stream.
232      * @return An interleaved stream.

234     public static <T> Stream<T> interleave(Function<T[], Integer> selector, List<Stream<T>> streams) {

239     /**
240      * Construct a stream which merges together values from the supplied streams, somewhat in the manner of the
241      * stream constructed by {@link com.codepoetics.protonpack.StreamUtils#zip(java.util.stream.Stream, java.util.stream.Stream, java.util.function.BiFunction)},
242      * but for an arbitrary number of streams and using a merger to merge the values from multiple streams
243      * into an accumulator.
244      *
245      * @param unitSupplier Supplies the initial "zero" or "unit" value for the accumulator.
246      * @param merger Merges each item from the collection of values taken from the source streams into the accumulator value.
247      * @param streams The streams to merge.
248      * @param <T> The type over which the merged streams stream.
249      * @param <O> The type of the accumulator, over which the constructed stream streams.
250      * @return A merging stream.
251      */
252     public static <T, O> Stream<O> merge(Supplier<O> unitSupplier, BiFunction<O, T, O> merger, Stream<T>...streams) {
253         Spliterator<T>[] spliterators = (Spliterator<T>[]) Stream.of(streams).map(s -> s.spliterator()).toArray(Spliterator[]::new);
254         return StreamSupport.stream(MergingSpliterator.merging(spliterators, unitSupplier, merger), false);
255     }

257     /**
258      * Construct a stream which merges together values from the supplied streams into lists of values, somewhat in the manner of the
259      * stream constructed by {@link com.codepoetics.protonpack.StreamUtils#zip(java.util.stream.Stream, java.util.stream.Stream, java.util.function.BiFunction)},
260      * but for an arbitrary number of streams.
261      *
262      * @param streams The streams to merge.
263      * @param <T> The type over which the merged streams stream.
264      * @return A merging stream of lists of T.
265      */
266     public static <T> Stream<List<T>> mergeToList(Stream<T>...streams) {
267         return merge(ArrayList::new, (l, x) -> {
268             l.add(x);
269             return l;
270         }, streams);
271     }

275      *
276      * @param source The source stream.
277      * @param predicate The filter condition.
278      * @param <T> The type over which the stream streams.
279      * @return A rejecting stream.

281     public static <T> Stream<T> reject(Stream<T> source, Predicate<? super T> predicate) {

285     /**
286      * Aggregates items from source stream into list of items while supplied predicate is true when evaluated on previous and current item.
287      * Can by seen as streaming alternative to Collectors.groupingBy when source stream is sorted by key. 
288      * @param source - source stream
289      * @param predicate - predicate specifying boundary between groups of items
290      * @param <T> The type over which the stream streams.
291      * @return Stream of List&lt;T&gt; aggregated according to predicate
292      */
293     public static <T> Stream<List<T>> aggregate(Stream<T> source, BiPredicate<T, T> predicate) {
294         return StreamSupport.stream(new AggregatingSpliterator<T>(source.spliterator(), 
295                 (a, e) -> a.isEmpty() || predicate.test(a.get(a.size() - 1), e)), false);
296     }

298     /**
299      * Aggregates items from source stream into list of items with fixed size
300      * @param source - source stream
301      * @param size - size of the aggregated list
302      * @param <T> The type over which the stream streams.
303      * @return Stream of List&lt;T&gt; with all list of size @size with possible exception of last List&lt;T&gt;
304      */
305     public static <T> Stream<List<T>> aggregate(Stream<T> source, int size) {
306         if (size <= 0) throw new IllegalArgumentException("Positive size expected, was: "+size);
307         return StreamSupport.stream(new AggregatingSpliterator<T>(source.spliterator(), (a, e) -> a.size() < size), false);
308     }

322     /**
323      * Converts nulls into an empty stream, and non-null values into a stream with one element.
324      * @param nullable The nullable value to convert.
325      * @param <T> The type of the value.
326      * @return A stream of zero or one values.
327      */
328     public static <T> Stream<T> streamNullable(T nullable) {
329         return null == nullable ? Stream.empty() : Stream.of(nullable);
330     }

332     /**
333      * Converts an Optional value to a stream of 0..1 values
334      * @param optional source optional value
335      * @param <T> The type of the optional value
336      * @return Stream of a single item of type T or an empty stream
337      */
338     public static <T> Stream<T> stream(Optional<T> optional) {
339         return optional.map(Stream::of).orElseGet(Stream::empty);
340     }
