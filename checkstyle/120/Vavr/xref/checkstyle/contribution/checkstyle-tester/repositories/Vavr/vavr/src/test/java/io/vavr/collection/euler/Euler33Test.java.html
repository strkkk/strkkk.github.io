7   package io.vavr.collection.euler;

9   import io.vavr.Tuple;
10  import io.vavr.collection.CharSeq;
11  import io.vavr.Tuple2;
12  import io.vavr.collection.List;
13  import org.junit.Test;
14  
15  import static io.vavr.collection.euler.Utils.factors;
16  import static org.assertj.core.api.Assertions.assertThat;
17  
18  public class Euler33Test {
19  
20      /**
21       * <strong>Problem 33: Digit cancelling fractions</strong>
22       * <p>
23       * The fraction 49/98 is a curious fraction, as an inexperienced
24       * mathematician in attempting to simplify it may incorrectly believe that
25       * 49/98 = 4/8, which is correct, is obtained by cancelling the 9s.
26       * <p>
27       * We shall consider fractions like, 30/50 = 3/5, to be trivial examples.
28       * <p>
29       * There are exactly four non-trivial examples of this type of fraction,
30       * less than one in value, and containing two digits in the numerator and
31       * denominator.
32       * <p>
33       * If the product of these four fractions is given in its lowest common
34       * terms, find the value of the denominator.
35       * <p>
36       * See also <a href="https://projecteuler.net/problem=33">projecteuler.net
37       * problem 33</a>.
38       */
39      @Test
40      public void shouldSolveProblem33() {
41          assertThat(isNonTrivialDigitCancellingFraction(Tuple.of(49, 98))).isTrue();
42          assertThat(isNonTrivialDigitCancellingFraction(Tuple.of(30, 50))).isFalse();
43          assertThat(isNonTrivialDigitCancellingFraction(Tuple.of(21, 22))).isFalse();
44          assertThat(lowestCommonDenominatorOfProductOfTheFourNonTrivialdigitCancellingFractions()).isEqualTo(100);
45      }
46  
47      private static int lowestCommonDenominatorOfProductOfTheFourNonTrivialdigitCancellingFractions() {
48          return List.rangeClosed(10, 98)
49                  .flatMap(n -> List.rangeClosed(n + 1, 99).map(d -> Tuple.of(n, d)))
50                  .filter(Euler33Test::isNonTrivialDigitCancellingFraction)
51                  .fold(Tuple.of(1, 1), Euler33Test::multiplyFractions)
52                  .apply(Euler33Test::simplifyFraction)._2;
53      }
54  
55      private static boolean isNonTrivialDigitCancellingFraction(Tuple2<Integer, Integer> fraction) {
56          return CharSeq.of(fraction._1.toString())
57                  .filter(d -> d != '0')
58                  .find(d -> CharSeq.of(fraction._2.toString()).contains(d))
59                  .map(d -> fractionCanBeSimplifiedByCancellingDigit(fraction, d))
60                  .getOrElse(false);
61      }
62  
63      private static boolean fractionCanBeSimplifiedByCancellingDigit(Tuple2<Integer, Integer> fraction, char d) {
64          return Tuple.of(CharSeq.of(fraction._1.toString()).remove(d), CharSeq.of(fraction._2.toString()).remove(d))
65                  .map(CharSeq::mkString, CharSeq::mkString)
66                  .map(Double::valueOf, Double::valueOf)
67                  .apply((d1, d2) -> fraction._1 / d1 == fraction._2 / d2);
68      }
69  
70      private static Tuple2<Integer, Integer> multiplyFractions(Tuple2<Integer, Integer> f1, Tuple2<Integer, Integer> f2) {
71          return Tuple.of(f1._1 * f2._1, f1._2 * f2._2);
72      }
73  
74      private static Tuple2<Integer, Integer> simplifyFraction(int numerator, int denominator) {
75          return factors(numerator)
76                  .map(Long::intValue)
77                  .filter(f -> f != 1)
78                  .sorted()
79                  .findLast(f -> denominator % f == 0)
80                  .map(f -> simplifyFraction(numerator / f, denominator / f))
81                  .getOrElse(Tuple.of(numerator, denominator));
82      }
