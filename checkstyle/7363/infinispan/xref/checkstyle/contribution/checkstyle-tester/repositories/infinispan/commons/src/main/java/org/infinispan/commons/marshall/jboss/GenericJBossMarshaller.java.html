1   package org.infinispan.commons.marshall.jboss;
2   
3   /**
4    * A marshaller that makes use of <a href="http://www.jboss.org/jbossmarshalling">JBoss Marshalling</a>
5    * to serialize and deserialize objects. This marshaller is oriented at external,
6    * non-core Infinispan use, such as the Java Hot Rod client.
7    *
8    * @author Manik Surtani
9    * @version 4.1
10   * @see <a href="http://www.jboss.org/jbossmarshalling">JBoss Marshalling</a>
11   */
12  public final class GenericJBossMarshaller extends AbstractJBossMarshaller {
13  
14     public GenericJBossMarshaller() {
15        super();
16        baseCfg.setClassResolver(
17              new DefaultContextClassResolver(this.getClass().getClassLoader()));
18     }
19  
20     public GenericJBossMarshaller(ClassLoader classLoader) {
21        super();
22        baseCfg.setClassResolver(
23              new DefaultContextClassResolver(classLoader != null ? classLoader : this.getClass().getClassLoader()));
24     }
25  
26  }
