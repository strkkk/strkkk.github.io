3   import org.infinispan.commons.CacheException;
4   
5   
6   /**
7    * Thrown when a timeout occurred. used by operations with timeouts, e.g. lock acquisition, or waiting for responses
8    * from all members.
9    *
10   * @author <a href="mailto:bela@jboss.org">Bela Ban</a>.
11   * @author Mircea.Markus@jboss.com
12   * @since 4.0
13   */
14  public class TimeoutException extends CacheException {
15  
16     /**
17      * The serialVersionUID
18      */
19     private static final long serialVersionUID = -8096787619908687038L;
20  
21     public TimeoutException() {
22        super();
23     }
24  
25     public TimeoutException(String msg) {
26        super(msg);
27     }
28  
29     public TimeoutException(String msg, Throwable cause) {
30        super(msg, cause);
31     }
32  
33     @Override
34     public String toString() {
35        return super.toString();
36     }
37  }
