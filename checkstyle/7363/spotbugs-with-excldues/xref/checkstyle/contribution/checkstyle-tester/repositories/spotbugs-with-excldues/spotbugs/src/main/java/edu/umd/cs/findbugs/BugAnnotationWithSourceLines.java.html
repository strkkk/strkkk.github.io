20  package edu.umd.cs.findbugs;
21  
22  /**
23   * @author pugh
24   */
25  public abstract class BugAnnotationWithSourceLines implements BugAnnotation {
26  
27      protected String sourceFileName;
28  
29      protected SourceLineAnnotation sourceLines;
30  
31      /**
32       *
33       */
34      public BugAnnotationWithSourceLines() {
35          super();
36      }
37  
38      @Override
39      public Object clone() {
40          try {
41              return super.clone();
42          } catch (CloneNotSupportedException e) {
43              throw new AssertionError(e);
44          }
45      }
46  
47      /**
48       * Get the source file name.
49       */
50      public final String getSourceFileName() {
51          return sourceFileName;
52      }
53  
54      /**
55       * Set a SourceLineAnnotation describing the source lines where the package
56       * element is defined.
57       */
58      public void setSourceLines(SourceLineAnnotation sourceLines) {
59          this.sourceLines = sourceLines;
60          sourceFileName = sourceLines.getSourceFile();
61      }
62  
63      /**
64       * Get the SourceLineAnnotation describing the source lines where the method
65       * is defined.
66       *
67       * @return the SourceLineAnnotation, or null if there is no source
68       *         information for this package element
69       */
70      public SourceLineAnnotation getSourceLines() {
71          return sourceLines;
72      }
73  
74      @Override
75      public String toString(ClassAnnotation primaryClass) {
76          return toString();
77      }
78  
79  }
