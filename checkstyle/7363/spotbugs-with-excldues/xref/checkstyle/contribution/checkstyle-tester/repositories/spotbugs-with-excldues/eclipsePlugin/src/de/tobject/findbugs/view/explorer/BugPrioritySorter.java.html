19  package de.tobject.findbugs.view.explorer;
20  
21  import java.text.Collator;
22  
23  import org.eclipse.core.resources.IMarker;
24  import org.eclipse.jface.viewers.Viewer;
25  import org.eclipse.jface.viewers.ViewerSorter;
26  
27  import de.tobject.findbugs.reporter.MarkerUtil;
28  
29  public class BugPrioritySorter extends ViewerSorter {
30  
31      public BugPrioritySorter() {
32          super();
33      }
34  
35      public BugPrioritySorter(Collator collator) {
36          super(collator);
37      }
38  
39      @Override
40      public int compare(Viewer viewer, Object e1, Object e2) {
41          int cat1 = category(e1);
42          int cat2 = category(e2);
43  
44          if (cat1 != cat2) {
45              return cat1 - cat2;
46          }
47  
48          if (e1 instanceof IMarker && e2 instanceof IMarker) {
49              return compareMarkers((IMarker) e1, (IMarker) e2);
50          }
51  
52          // Sorts groups on priority first, then on group name
53          if (e1 instanceof BugGroup && e2 instanceof BugGroup) {
54              return compareGroups((BugGroup) e1, (BugGroup) e2);
55          }
56          return super.compare(viewer, e1, e2);
57      }
58  
59      @Override
60      public int category(Object element) {
61          if (element instanceof IMarker) {
62              return 1;
63          }
64          if (element instanceof BugGroup) {
65              return 5;
66          }
67          // return smallest value: 0
68          return super.category(element);
69      }
70  
71      /**
72       * Sorts bug groups on severity first, then on bug pattern name.
73       */
74      static int compareGroups(BugGroup m1, BugGroup m2) {
75          int result = m1.compareTo(m2);
76          if (result == 0) {
77              return m1.getShortDescription().compareToIgnoreCase(m2.getShortDescription());
78          }
79          return result;
80      }
81  
82      /**
83       * Sorts markers on rank, then priority, and then on name if requested
84       *
85       * @param m1
86       * @param m2
87       * @return
88       */
89      static int compareMarkers(IMarker m1, IMarker m2) {
90          if (m1 == null || m2 == null || !m1.exists() || !m2.exists()) {
91              return 0;
92          }
93          int rank1 = MarkerUtil.findBugRankForMarker(m1);
94          int rank2 = MarkerUtil.findBugRankForMarker(m2);
95          int result = rank1 - rank2;
96          if (result != 0) {
97              return result;
98          }
99          int prio1 = MarkerUtil.findConfidenceForMarker(m1).ordinal();
100         int prio2 = MarkerUtil.findConfidenceForMarker(m2).ordinal();
101         result = prio1 - prio2;
102         if (result != 0) {
103             return result;
104         }
105         String a1 = m1.getAttribute(IMarker.MESSAGE, "");
106         String a2 = m2.getAttribute(IMarker.MESSAGE, "");
107         return a1.compareToIgnoreCase(a2);
108     }
109 
110 }
