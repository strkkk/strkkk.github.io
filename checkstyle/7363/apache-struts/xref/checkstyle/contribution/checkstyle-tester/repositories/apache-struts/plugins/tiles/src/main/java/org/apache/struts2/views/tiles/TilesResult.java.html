44  /**
45   * <!-- START SNIPPET: description -->
46   * Renders a view using struts-tiles.
47   * <!-- END SNIPPET: description -->
48   *
49   * <!-- START SNIPPET: webxml -->
50   * In your web.xml file, you need to add a TilesListener.
51   *
52   * &lt;listener&gt;
53   *      &lt;listener-class&gt;org.apache.struts2.tiles.StrutsTilesListener&lt;/listener-class&gt;
54   * &lt;/listener&gt;
55   * <!-- END SNIPPET: webxml -->
56   *
57   * <!-- START SNIPPET: strutsxml -->
58   * In struts.xml, use type="tiles" on your &lt;result&gt;.
59   *
60   * &lt;action name="editUser" class="userAction" method="edit"&gt;
61   *      &lt;result name="success" type="tiles"&gt;userForm&lt;/result&gt;
62   *      &lt;result name="input" type="tiles"&gt;userList&lt;/result&gt;
63   * &lt;/action&gt;
64   * <!-- END SNIPPET: strutsxml -->
65   *
66   *
67   * <!-- START SNIPPET: packageconfig -->
68   *
69   * Making this result type the default for the current package.
70   *
71   * &lt;result-types&gt;
72   *      &lt;result-type name="tiles"
73   * class="org.apache.struts2.views.tiles.TilesResult" default="true" /&gt;
74   * &lt;/result-types&gt;
75   * <!-- END SNIPPET: packageconfig -->
76   *
77   *
78   * <!-- START SNIPPET: tilesconfig -->
79   * You have to configure tiles itself. Therefore you can add <code>tiles.xml</code> either 
80   * to resources or WEB-INF. You may also use annotations like {@link TilesDefinition}.
81   *
82   * <!-- END SNIPPET: tilesconfig -->
83   *
84   */
85  public class TilesResult extends ServletDispatcherResult {
86  
87      private static final long serialVersionUID = -3806939435493086244L;
88  
89      private static final Logger LOG = LogManager.getLogger(TilesResult.class);
90  
91      public TilesResult() {
92          super();
93      }
94  
95      public TilesResult(String location) {
96          super(location);
97      }
98  
99      /**
100      * Dispatches to the given location. Does its forward via a RequestDispatcher. If the
101      * dispatch fails a 404 error will be sent back in the http response.
102      *
103      * @param location the location to dispatch to.
104      * @param invocation    the execution state of the action
105      * @throws Exception if an error occurs. If the dispatch fails the error will go back via the
106      *                   HTTP request.
107      */
108     public void doExecute(String location, ActionInvocation invocation) throws Exception {
109         StrutsTilesAnnotationProcessor annotationProcessor = new StrutsTilesAnnotationProcessor();
110         TilesDefinition tilesDefinition = null;
111         Object action = invocation.getAction();
112         String actionName = invocation.getInvocationContext().getName();
113 
114         if (StringUtils.isEmpty(location)) {
115             LOG.trace("location not set -> action must have one @TilesDefinition");
116             tilesDefinition = annotationProcessor.findAnnotation(action, null);
117             String tileName = StringUtils.isNotEmpty(tilesDefinition.name()) ? tilesDefinition.name() : actionName;
118             location = tileName;
119             LOG.debug("using new location name '{}' and @TilesDefinition '{}'", location, tilesDefinition);
120         }
121         setLocation(location);
122 
123         ServletContext servletContext = ServletActionContext.getServletContext();
124 
125         ApplicationContext applicationContext = ServletUtil.getApplicationContext(servletContext);
126         TilesContainer container = TilesAccess.getContainer(applicationContext);
127 
128         HttpServletRequest httpRequest = ServletActionContext.getRequest();
129         HttpServletResponse httpResponse = ServletActionContext.getResponse();
130 
131         Request request = new ServletRequest(applicationContext, httpRequest, httpResponse);
132 
133         boolean definitionValid = false;
134         try {
135             LOG.debug("checking if tiles definition exists '{}'", location);
136             definitionValid = container.isValidDefinition(location, request);
137         } catch (TilesException e) {
138             LOG.warn("got TilesException while checking if definiton exists, ignoring it", e);
139         }
140         if (!definitionValid) {
141             if (tilesDefinition == null) {
142                 LOG.trace("tilesDefinition not found yet, searching in action");
143                 tilesDefinition = annotationProcessor.findAnnotation(action, location);
144             }
145             if (tilesDefinition != null) {
146                 Definition definition = annotationProcessor.buildTilesDefinition(location, tilesDefinition);
147                 if (container instanceof MutableTilesContainer) {
148                     LOG.debug("registering tiles definition with name '{}'", definition.getName());
149                     ((MutableTilesContainer)container).register(definition, request);
150                 } else {
151                     LOG.error("cannot register tiles definition as tiles container is not mutable!");
152                 }
153             } else {
154                 LOG.warn("could not find @TilesDefinition for action: {}", actionName);
155             }
156         }
157 
158         container.render(location, request);
159     }
160 }
