19  package org.apache.struts2.el.util;
20  
21  import java.text.MessageFormat;
22  import java.util.ResourceBundle;
23  
24  /**
25   * @author Jacob Hookom [jacob@hookom.net]
26   * @version $Change: 181177 $$DateTime: 2001/06/26 08:45:09 $$Author: rjung $
27   */
28  public final class MessageFactory {
29  
30      protected final static ResourceBundle bundle = ResourceBundle
31              .getBundle("org.apache.el.Messages");
32      /**
33       * 
34       */
35      public MessageFactory() {
36          super();
37      }
38  
39      public static String get(final String key) {
40          return bundle.getString(key);
41      }
42  
43      public static String get(final String key, final Object obj0) {
44          return getArray(key, new Object[] { obj0 });
45      }
46  
47      public static String get(final String key, final Object obj0,
48              final Object obj1) {
49          return getArray(key, new Object[] { obj0, obj1 });
50      }
51  
52      public static String get(final String key, final Object obj0,
53              final Object obj1, final Object obj2) {
54          return getArray(key, new Object[] { obj0, obj1, obj2 });
55      }
56  
57      public static String get(final String key, final Object obj0,
58              final Object obj1, final Object obj2, final Object obj3) {
59          return getArray(key, new Object[] { obj0, obj1, obj2, obj3 });
60      }
61  
62      public static String get(final String key, final Object obj0,
63              final Object obj1, final Object obj2, final Object obj3,
64              final Object obj4) {
65          return getArray(key, new Object[] { obj0, obj1, obj2, obj3, obj4 });
66      }
67  
68      public static String getArray(final String key, final Object[] objA) {
69          return MessageFormat.format(bundle.getString(key), objA);
70      }
71  
72  }
