19  package org.apache.struts2.views.java.simple;
20  
21  import org.apache.struts2.views.java.DefaultTagHandlerFactory;
22  import org.apache.struts2.views.java.DefaultTheme;
23  import org.apache.struts2.views.java.TagHandlerFactory;
24  import org.apache.struts2.views.java.XHTMLTagSerializer;
25  
26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.List;
29  
30  public class SimpleTheme extends DefaultTheme {
31  
32      public SimpleTheme() {
33          setHandlerFactories(new HashMap<String, List<TagHandlerFactory>>() {
34              {
35                  put("text", new FactoryList(TextFieldHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
36                  put("textfield", new FactoryList(TextFieldHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
37                  put("datetextfield", new FactoryList(DateTextFieldHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class));
38                  put("select", new FactoryList(SelectHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
39                  put("form", new FactoryList(FormHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
40                  put("form-close", new FactoryList(FormHandler.CloseHandler.class));
41                  put("a", new FactoryList(AnchorHandler.class));
42                  put("a-close", new FactoryList(AnchorHandler.CloseHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
43                  put("checkbox", new FactoryList(CheckboxHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
44                  put("file", new FactoryList(FileHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
45                  put("password", new FactoryList(PasswordHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
46                  put("label", new FactoryList(LabelHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
47                  put("reset", new FactoryList(ResetHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
48                  put("submit", new FactoryList(SubmitHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
49                  put("submit-close", new FactoryList(SubmitHandler.CloseHandler.class));
50                  put("textarea", new FactoryList(TextAreaHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
51                  put("radiomap", new FactoryList(RadioHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
52                  put("checkboxlist", new FactoryList(CheckboxListHandler.class, ScriptingEventsHandler.class, CommonAttributesHandler.class, DynamicAttributesHandler.class));
53                  put("actionerror", new FactoryList(ActionErrorHandler.class));
54                  put("token", new FactoryList(TokenHandler.class));
55                  put("actionmessage", new FactoryList(ActionMessageHandler.class));
56                  put("head", new FactoryList(HeadHandler.class));
57                  put("hidden", new FactoryList(HiddenHandler.class));
58                  put("fielderror", new FactoryList(FieldErrorHandler.class));
59                  put("empty", new FactoryList(EmptyHandler.class));
60             }
61          });
62          setName("simple");
63      }
64  
65      private class FactoryList extends ArrayList<TagHandlerFactory> {
66  
67          private static final long serialVersionUID = -1551895041394434032L;
68  
69          public FactoryList(Class... classes) {
70              super();
71              for (Class cls : classes) {
72                  add(new DefaultTagHandlerFactory(cls));
73              }
74              add(new DefaultTagHandlerFactory(XHTMLTagSerializer.class));
75          }
76      }
77  
78  }
