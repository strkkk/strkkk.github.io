25  /**
26    A safe replacement for Optional.
27  
28    Option consists of two states
29  <pre>
30    1. Some : a value is present
31    2. None : no value is present
32   </pre>
33  
34   Unlike Optional, Option does not expose an unsafe `get` method. `fold` or `orElse` can be used instead.
35  
36   @see Maybe is a lazy / reactive sub-class of Option
37  
38   **/
39  public interface Option<T> extends To<Option<T>>,
40                                      OrElseValue<T,Option<T>>,
41                                      MonadicValue<T>,
42                                      Zippable<T>,
43                                      Recoverable<T>,
44                                      Sealed2<T, Option.None<T>>,
45                                      Iterable<T>,
46                                      Higher<option,T>,
47                                      Serializable{

53      public static <T> Option<T> attempt(CheckedSupplier<T> s){
54          try {

72      public static  <T,R> Option<R> tailRec(T initial, Function<? super T, ? extends Option<? extends Either<T, R>>> fn){
73        Option<? extends Either<T, R>> next[] = new Option[1];
74        next[0] = Option.some(Either.left(initial));
75        boolean cont = true;
76        do {
77          cont = next[0].fold(p -> p.fold(s -> {
78            next[0] = narrowK(fn.apply(s));
79            return true;
80          }, pr -> false), () -> false);
81        } while (cont);
82  
83        return next[0].map(x->x.fold(l->null, r->r));
84      }

352   public static  <T> Option<ReactiveSeq<T>> sequence(ReactiveSeq<? extends Option<T>> stream) {
353 
354     Option<ReactiveSeq<T>> identity = Option.some(ReactiveSeq.empty());
355 
356     BiFunction<Option<ReactiveSeq<T>>,Option<T>,Option<ReactiveSeq<T>>> combineToStream = (acc,next) ->acc.zip(next,(a,b)->a.append(b));
357 
358     BinaryOperator<Option<ReactiveSeq<T>>> combineStreams = (a,b)-> a.zip(b,(z1,z2)->z1.appendStream(z2));
359 
360     return stream.reduce(identity,combineToStream,combineStreams);
361   }

364     return sequence(s);

368      * Accumulates the present results, ignores zero Maybes.
