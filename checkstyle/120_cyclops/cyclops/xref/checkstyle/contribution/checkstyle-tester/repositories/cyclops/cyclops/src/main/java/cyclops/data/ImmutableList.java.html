49      default int compareTo(final T o) {
50          if (o instanceof ImmutableList) {
51              final ImmutableList l = (ImmutableList) o;
52              if (this.size() == l.size()) {
53                  final Iterator i1 = iterator();
54                  final Iterator i2 = l.iterator();
55                  if (i1.hasNext()) {
56                      if (i2.hasNext()) {
57                          final int comp = Comparator.<Comparable> naturalOrder()
58                              .compare((Comparable) i1.next(), (Comparable) i2.next());
59                          if (comp != 0)
60                              return comp;
61                      }
62                      return 1;
63                  } else {
64                      if (i2.hasNext())
65                          return -1;
66                      else
67                          return 0;
68                  }
69              }
70              return this.size() - ((ImmutableList) o).size();
71          } else
72              return 1;
73  
74      }

92      default ImmutableList<T> replaceFirst(T currentElement, T newElement){
93          ImmutableList<T> preceding = emptyUnit();
94          ImmutableList<T> tail = this;
95          while(!tail.isEmpty()){
96              ImmutableList<T> ref=  preceding;
97              ImmutableList<T> tailRef = tail;
98              Tuple3<ImmutableList<T>, ImmutableList<T>, Boolean> t3 = tail.fold(c -> {
99                  if (Objects.equals(c.head(), currentElement))
100                     return Tuple.tuple(ref, tailRef, true);
101                 return Tuple.tuple(ref.prepend(c.head()), c.tail(), false);
102             }, nil -> Tuple.tuple(ref, tailRef, true));
103 
104             preceding = t3._1();
105             tail = t3._2();
106             if(t3._3())
107                 break;
108 
109         }
110 
111         ImmutableList<T> start = preceding;
112         return tail.fold(cons->cons.tail().prepend(newElement).prependAll(start), nil->this);
113     }
