1   package com.oath.cyclops.rx2.adapter;
2   
3   import com.oath.cyclops.anym.extensability.AbstractMonadAdapter;
4   
5   
6   import cyclops.companion.rx2.Functions;
7   import cyclops.companion.rx2.Singles;
8   import cyclops.control.Future;
9   import cyclops.monads.AnyM;
10  import cyclops.monads.Rx2Witness.single;
11  import cyclops.monads.SingleAnyM;
12  import cyclops.reactive.ReactiveSeq;
13  import io.reactivex.Single;
14  import lombok.AllArgsConstructor;
15  
16  
17  import java.util.function.Function;
18  import java.util.function.Predicate;
19  
20  
21  
22  
23  @AllArgsConstructor
24  public class SingleAdapter extends AbstractMonadAdapter<single> {
25  
26      @Override
27      public <T> Iterable<T> toIterable(AnyM<single, T> t) {
28          return Future.fromPublisher(future(t).toFlowable());
29      }
30  
31      @Override
32      public <T, R> AnyM<single, R> ap(AnyM<single,? extends Function<? super T,? extends R>> fn, AnyM<single, T> apply) {
33          Single<T> f = future(apply);
34  
35          Single<? extends Function<? super T, ? extends R>> fnF = future(fn);
36  
37          Future<T> crF1 = Future.fromPublisher(f.toFlowable());
38          Future<? extends Function<? super T, ? extends R>> crFnF = Future.fromPublisher(fnF.toFlowable());
39  
40          Single<R> res = Single.fromPublisher(crF1.zip(crFnF,(a,b)->b.apply(a)));
41          return SingleAnyM.anyM(res);
42  
43      }
44  
45      @Override
46      public <T> AnyM<single, T> filter(AnyM<single, T> t, Predicate<? super T> fn) {
47  
48          return SingleAnyM.anyM(future(t).filter(Functions.rxPredicate(fn)).toSingle());
49      }
50  
51      <T> Single<T> future(AnyM<single,T> anyM){
52          return anyM.unwrap();
53      }
54      <T> Future<T> futureW(AnyM<single,T> anyM){
55          return Future.fromPublisher(anyM.unwrap());
56      }
57  
58      @Override
59      public <T> AnyM<single, T> empty() {
60          return SingleAnyM.anyM(Single.never());
61      }
62  
63  
64  
65      @Override
66      public <T, R> AnyM<single, R> flatMap(AnyM<single, T> t,
67                                       Function<? super T, ? extends AnyM<single, ? extends R>> fn) {
68          return SingleAnyM.anyM(Single.fromPublisher(futureW(t).flatMap(fn.andThen(a-> futureW(a)))));
69  
70      }
71  
72      @Override
73      public <T> AnyM<single, T> unitIterable(Iterable<T> it)  {
74          return SingleAnyM.anyM(Single.fromPublisher(Future.fromIterable(it)));
75      }
76  
77      @Override
78      public <T> AnyM<single, T> unit(T o) {
79          return SingleAnyM.anyM(Single.just(o));
80      }
81  
82      @Override
83      public <T> ReactiveSeq<T> toStream(AnyM<single, T> t) {
84          return ReactiveSeq.fromPublisher(future(t).toFlowable());
85      }
86  
87      @Override
88      public <T, R> AnyM<single, R> map(AnyM<single, T> t, Function<? super T, ? extends R> fn) {
89          return SingleAnyM.anyM(future(t).map(x->fn.apply(x)));
90      }
91  }
