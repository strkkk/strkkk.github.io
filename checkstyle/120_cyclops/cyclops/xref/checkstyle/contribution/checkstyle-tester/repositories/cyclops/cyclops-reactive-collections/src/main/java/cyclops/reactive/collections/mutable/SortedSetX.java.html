45   * @author johnmcclean

52      public static <T> SortedSetX<T> defer(Supplier<SortedSetX<T>> s){
53        return of(s)
54                .map(Supplier::get)
55                .concatMap(l->l);
56      }
57      default Tuple2<SortedSetX<T>, SortedSetX<T>> splitAt(int n) {
58          materialize();
59          return Tuple.tuple(take(n), drop(n));
60      }
61      default Tuple2<SortedSetX<T>, SortedSetX<T>> span(Predicate<? super T> pred) {
62          return tuple(takeWhile(pred), dropWhile(pred));
63      }

65      default Tuple2<SortedSetX<T>,SortedSetX<T>> splitBy(Predicate<? super T> test) {
66          return span(test.negate());
67      }
68      default Tuple2<SortedSetX<T>, SortedSetX<T>> partition(final Predicate<? super T> splitter) {
69  
70          return tuple(filter(splitter), filter(splitter.negate()));

1116         static <T, R extends ReactiveSeq<T> & Comparable<T>> R comparable(final ReactiveSeq<T> seq) {
1117             final Method compareTo = Stream.of(Comparable.class.getMethods())
1118                                            .filter(m -> m.getName()
1119                                                          .equals("compareTo"))
1120                                            .findFirst()
1121                                            .get();
1122 
1123             return (R) Proxy.newProxyInstance(SortedSetX.class.getClassLoader(), new Class[] { ReactiveSeq.class, Comparable.class },
1124                                               (proxy, method, args) -> {
1125                                                   if (compareTo.equals(method))
1126                                                       return Objects.compare(System.identityHashCode(seq), System.identityHashCode(args[0]),
1127                                                                              Comparator.naturalOrder());
1128                                                   else
1129                                                       return method.invoke(seq, args);
1130                                               });
1131 
1132         }

1135     public static  <T,R> SortedSetX<R> tailRec(T initial, Function<? super T, ? extends SortedSetX<? extends Either<T, R>>> fn) {
1136         ListX<Either<T, R>> lazy = ListX.of(Either.left(initial));
1137         ListX<Either<T, R>> next = lazy.eager();
1138         boolean newValue[] = {true};
1139         for(;;){
1140 
1141             next = next.concatMap(e -> e.fold(s -> {
1142                         newValue[0]=true;
1143                         return fn.apply(s); },
1144                     p -> {
1145                         newValue[0]=false;
1146                         return ListX.of(e);
1147                     }));
1148             if(!newValue[0])
1149                 break;
1150 
1151         }
1152         return Either.sequenceRight(next)
1153                      .orElse(ReactiveSeq.empty())
1154                      .to(ReactiveConvertableSequence::converter)
1155                      .sortedSetX(Evaluation.LAZY);
1156     }
