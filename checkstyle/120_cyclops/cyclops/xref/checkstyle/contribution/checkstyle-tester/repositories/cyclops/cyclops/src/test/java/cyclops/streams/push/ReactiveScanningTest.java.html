1   package cyclops.streams.push;
2   
3   import cyclops.companion.Reducers;
4   import org.junit.Test;
5   
6   import static cyclops.reactive.Spouts.of;
7   import static java.util.Arrays.asList;
8   import static org.hamcrest.Matchers.is;
9   import static org.junit.Assert.assertThat;
10  
11  public class ReactiveScanningTest {
12  	@Test
13  	public void testScanLeftStringConcat() {
14  		assertThat(of("a", "b", "c").scanLeft("", String::concat).toList(), is(asList("", "a", "ab", "abc")));
15  	}
16  
17  	@Test
18  	public void testScanLeftSum() {
19  		assertThat(of("a", "ab", "abc").map(str -> str.length()).scanLeft(0, (u, t) -> u + t).toList(), is(asList(0, 1, 3, 6)));
20  	}
21  
22  	@Test
23  	public void testScanLeftStringConcatMonoid() {
24  		assertThat(of("a", "b", "c").scanLeft(Reducers.toString("")).toList(), is(asList("", "a", "ab", "abc")));
25  	}
26  
27  	@Test
28  	public void testScanLeftSumMonoid() {
29  		assertThat(of("a", "ab", "abc").map(str -> str.length()).scanLeft(Reducers.toTotalInt()).toList(), is(asList(0, 1, 3, 6)));
30  	}
31  
32  	@Test
33  	public void testScanRightStringConcat() {
34  		assertThat(of("a", "b", "c").scanRight("", String::concat).toList(), is(asList("", "c", "bc", "abc")));
35  	}
36  
37  	@Test
38  	public void testScanRightSum() {
39  		assertThat(of("a", "ab", "abc").map(str -> str.length()).scanRight(0, (t, u) -> u + t).toList(), is(asList(0, 3, 5, 6)));
40  
41  	}
42  	@Test
43  	public void testScanRightStringConcatMonoid() {
44  		assertThat(of("a", "b", "c").scanRight(Reducers.toString("")).toList(), is(asList("", "c", "bc", "abc")));
45  	}
46  
47  	@Test
48  	public void testScanRightSumMonoid() {
49  		assertThat(of("a", "ab", "abc").map(str -> str.length()).scanRight(Reducers.toTotalInt()).toList(), is(asList(0, 3, 5, 6)));
50  
51  	}
52  }
