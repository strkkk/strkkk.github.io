35  /**
36   * Provide easy access to all typeclasses for a type
37   * e.g.
38   *
39   * <pre>
40   *     {@code
41   *       Active<list,Integer> active = Active.of(ListX.of(1,2,3),ListX.Instances.definitions());
42   *       Active<list,Integer> doubled = active.map(i->i*2);
43   *       Active<list,Integer> doubledPlusOne = doubled.flatMap(i->ListX.of(i+1));
44   *     }
45   *
46   * </pre>
47   *
48   * @param <W> Witness type
49   * @param <T> Data type
50   */

117      * @return Converter that works on the concrete type
118      */
