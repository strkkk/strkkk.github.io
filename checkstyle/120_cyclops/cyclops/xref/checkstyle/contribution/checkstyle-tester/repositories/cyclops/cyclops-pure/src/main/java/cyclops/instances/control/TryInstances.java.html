122   public static class TryTypeclasses<L extends Throwable,X> implements Monad<Higher<tryType, L>>,
123                                                     MonadRec<Higher<tryType, L>>,

127       @Override
128       public <T> T foldRight(Monoid<T> monoid, Higher<Higher<tryType, L>, T> ds) {
129           Try<T,L> tryType = narrowK(ds);
130           return tryType.fold(monoid);
131       }
132 
133       @Override
134       public <T> T foldLeft(Monoid<T> monoid, Higher<Higher<tryType, L>, T> ds) {
135           Try<T,L> tryType = narrowK(ds);

138       @Override
139       public <T, R> R foldMap(Monoid<R> mb, Function<? super T, ? extends R> fn, Higher<Higher<tryType, L>, T> nestedA) {
140           return foldLeft(mb, narrowK(nestedA).<R>map(fn));
141       }

143       @Override
144       public <T, R> Higher<Higher<tryType, L>, R> flatMap(Function<? super T, ? extends Higher<Higher<tryType, L>, R>> fn, Higher<Higher<tryType, L>, T> ds) {
145           Try<T,L> tryType = narrowK(ds);
146           return tryType.flatMap(fn.andThen(Try::narrowK));
147       }
148 
149       @Override
150       public <C2, T, R> Higher<C2, Higher<Higher<tryType, L>, R>> traverseA(Applicative<C2> applicative, Function<? super T, ? extends Higher<C2, R>> fn, Higher<Higher<tryType, L>, T> ds) {
151           Try<T, L> maybe = narrowK(ds);
152           Function<R, Try<R, L>> rightFn = r -> Try.success(r);
153 
154           return maybe.fold(r->applicative.map(rightFn, fn.apply(r)),l->applicative.unit(Try.failure(l)));
155       }

159       @Override
160       public <T, R> Higher<Higher<tryType, L>, R> ap(Higher<Higher<tryType, L>, ? extends Function<T, R>> fn, Higher<Higher<tryType, L>, T> apply) {
161           Try<T,L>  tryType = narrowK(apply);
162           Try<? extends Function<T, R>, L> tryTypeFn = narrowK(fn);
163           return tryTypeFn.zip(tryType,(a,b)->a.apply(b));
164       }

168           return Try.success(value);
169       }

172       public <T, R> Higher<Higher<tryType, L>, R> map(Function<? super T, ? extends R> fn, Higher<Higher<tryType, L>, T> ds) {
173           Try<T,L> tryType = narrowK(ds);

178       @Override
179       public <T, R> Higher<Higher<tryType, L>, R> tailRec(T initial, Function<? super T, ? extends Higher<Higher<tryType, L>, ? extends Either<T, R>>> fn) {
180           Try<? extends Either<T, R>,L> next[] = new Try[1];
181           next[0] = Try.success(Either.left(initial));
182           boolean cont = true;
183           do {
184               cont = next[0].fold(p -> p.fold(s -> {
185                   next[0] = narrowK(fn.apply(s));
186                   return true;
187               }, pr -> false), () -> false);
188           } while (cont);
189           return next[0].map(x->x.orElse(null));
190       }
