1   package com.oath.cyclops.rx2.adapter;
2   
3   
4   import com.oath.cyclops.anym.extensability.AbstractMonadAdapter;
5   import cyclops.monads.AnyM;
6   
7   import cyclops.monads.FlowableAnyM;
8   import cyclops.monads.Rx2Witness;
9   import cyclops.monads.Rx2Witness.flowable;
10  import cyclops.reactive.ReactiveSeq;
11  import io.reactivex.Flowable;
12  import lombok.AllArgsConstructor;
13  
14  
15  import java.util.function.Function;
16  import java.util.function.Predicate;
17  
18  
19  @AllArgsConstructor
20  public class FlowableAdapter extends AbstractMonadAdapter<flowable> {
21  
22  
23  
24  
25  
26      @Override
27      public <T> Iterable<T> toIterable(AnyM<flowable, T> t) {
28          return ()->stream(t).blockingIterable().iterator();
29      }
30  
31      @Override
32      public <T, R> AnyM<flowable, R> ap(AnyM<flowable,? extends Function<? super T,? extends R>> fn, AnyM<flowable, T> apply) {
33          Flowable<T> f = stream(apply);
34          Flowable<? extends Function<? super T, ? extends R>> fnF = stream(fn);
35          Flowable<R> res = fnF.zipWith(f, (a, b) -> a.apply(b));
36          return FlowableAnyM.anyM(res);
37  
38      }
39  
40      @Override
41      public <T> AnyM<flowable, T> filter(AnyM<flowable, T> t, Predicate<? super T> fn) {
42          return FlowableAnyM.anyM(stream(t).filter(i->fn.test(i)));
43      }
44  
45      <T> Flowable<T> stream(AnyM<flowable,T> anyM){
46          return anyM.unwrap();
47      }
48  
49      @Override
50      public <T> AnyM<flowable, T> empty() {
51          return FlowableAnyM.anyM(Flowable.empty());
52      }
53  
54  
55  
56      @Override
57      public <T, R> AnyM<flowable, R> flatMap(AnyM<flowable, T> t,
58                                       Function<? super T, ? extends AnyM<flowable, ? extends R>> fn) {
59          return FlowableAnyM.anyM(stream(t).flatMap(i->fn.andThen(a->stream(a)).apply(i)));
60  
61      }
62  
63      @Override
64      public <T> AnyM<flowable, T> unitIterable(Iterable<T> it)  {
65          return FlowableAnyM.anyM(Flowable.fromIterable(it));
66      }
67  
68      @Override
69      public <T> AnyM<flowable, T> unit(T o) {
70          return FlowableAnyM.anyM(Flowable.just(o));
71      }
72  
73      @Override
74      public <T> ReactiveSeq<T> toStream(AnyM<flowable, T> t) {
75          return ReactiveSeq.fromPublisher(Rx2Witness.flowable(t));
76      }
77  
78      @Override
79      public <T, R> AnyM<flowable, R> map(AnyM<flowable, T> t, Function<? super T, ? extends R> fn) {
80          return FlowableAnyM.anyM(stream(t).map(i->fn.apply(i)));
81      }
82  }
