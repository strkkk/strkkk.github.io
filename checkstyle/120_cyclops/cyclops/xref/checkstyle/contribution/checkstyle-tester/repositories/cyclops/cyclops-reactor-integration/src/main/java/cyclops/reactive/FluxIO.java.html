1   package cyclops.reactive;
2   
3   import cyclops.companion.reactor.Fluxs;
4   import cyclops.control.Future;
5   import cyclops.control.Try;
6   import cyclops.data.Seq;
7   import lombok.AccessLevel;
8   import lombok.AllArgsConstructor;
9   import org.reactivestreams.Publisher;
10  import reactor.core.publisher.Flux;
11  import reactor.core.scheduler.Scheduler;
12  
13  import java.util.function.BiFunction;
14  import java.util.function.Consumer;
15  import java.util.function.Function;
16  import java.util.function.Supplier;
17  
18  @AllArgsConstructor(access = AccessLevel.PRIVATE)
19  public final class FluxIO<T> implements IO<T> {
20      private final Flux<T> flowable;
21  
22      public static <T> IO<T> of(Future<T> f){
23          return of(Flux.from(f));
24      }
25      public static <T> IO<T> of(Flux<T> flowable){
26          return new FluxIO<>(flowable);
27      }
28      public static <T> IO<T> just(T s){
29          return new FluxIO<T>(Flux.just(s));
30      }
31  
32      public static <T> IO<T> of(Supplier<? extends T> s){
33          return new FluxIO<T>(Fluxs.generate(()->s.get()));
34      }
35  
36      public static <T> IO<T> of(Supplier<? extends T> s, Scheduler ex){
37          Flux<T> x = Fluxs.generate(() -> s.get());
38          x = x.subscribeOn(ex);
39          return new FluxIO<T>(x);
40      }
41  
42  
43      public static <T> IO<T> fromPublisher(Publisher<T> p){
44          return new FluxIO<T>(Flux.from(p));
45      }
46  
47      public static <T,X extends Throwable> IO<Try<T,X>> withCatch(Try.CheckedSupplier<T, X> cf, Class<? extends X>... classes){
48          return of(()-> Try.withCatch(cf,classes));
49      }
50  
51  
52      public static <T1,T2,R> IO<R> merge(Publisher<T1> p1, Publisher<T2> p2, BiFunction<? super T1, ? super T2, ? extends R> fn2){
53          Flux<T1> s1 = Flux.from(p1);
54          Flux<T2> s2 = Flux.from(p2);
55          return fromPublisher(s1.zipWith(s2,(a,b)->fn2.apply(a,b)));
56      }
57  
58      @Override
59      public <B, R> IO<R> par(IO<B> that, BiFunction<? super T, ? super B, ? extends R> fn) {
60          return IO.fromPublisher(flowable.zipWith(that,fn));
61      }
62  
63      @Override
64      public IO<T> race(IO<T> that) {
65          return fromPublisher(Flux.first(Seq.of(publisher(), that.publisher())));
66      }
67  
68      @Override
69      public <R> IO<R> map(Function<? super T, ? extends R> s) {
70          return of(flowable.map(s));
71      }
72  
73      @Override
74      public <R> IO<R> flatMap(Function<? super T, IO<? extends R>> s) {
75          return of(flowable.flatMap(s));
76      }
77  
78      @Override
79      public <R> IO<R> mergeMap(int maxConcurrency, Function<? super T, Publisher<? extends R>> s) {
80          return of(flowable.flatMap(s,maxConcurrency));
81      }
82      @Override
83      public <R extends AutoCloseable> IO<R> bracket(Function<? super T, ? extends R> fn) {
84          Managed<R> m = FluxManaged.of(map(fn));
85          return m.io();
86      }
87  
88      @Override
89      public <R> IO<R> bracket(Function<? super T, ? extends R> fn, Consumer<R> consumer) {
90          Managed<R> m = FluxManaged.of(map(fn),consumer);
91          return m.io();
92      }
93      @Override
94      public <R extends AutoCloseable,R1> Managed.Tupled<R,R1> bracketWith(Function<? super T, ? extends R> fn, Function<? super R, ? extends R1> with) {
95          Managed.Tupled<? extends R, ? extends R1> x = FluxManaged.of(map(fn)).with(with);
96          return (Managed.Tupled<R, R1> )x;
97      }
98      @Override
99      public void forEach(Consumer<? super T> consumerElement, Consumer<? super Throwable> consumerError, Runnable onComplete) {
100         flowable.subscribe(consumerElement,consumerError,onComplete);
101     }
102 
103     @Override
104     public Future<T> future() {
105         return Future.fromPublisher(flowable);
106     }
107 
108     @Override
109     public Publisher<T> publisher() {
110         return flowable;
111     }
112 
113     @Override
114     public ReactiveSeq<T> stream() {
115         return FluxReactiveSeq.reactiveSeq(flowable);
116     }
117 
118     @Override
119     public <R> IO<R> unit(Publisher<R> pub) {
120         return FluxIO.<R>fromPublisher(pub);
121     }
122 }
