1   package cyclops.reactive;
2   
3   import com.oath.cyclops.hkt.DataWitness;
4   import com.oath.cyclops.hkt.DataWitness.future;
5   import com.oath.cyclops.hkt.DataWitness.reactiveSeq;
6   import com.oath.cyclops.hkt.Higher;
7   import cyclops.control.Future;
8   import cyclops.instances.control.FutureInstances;
9   import cyclops.instances.reactive.PublisherInstances;
10  import cyclops.typeclasses.monad.Monad;
11  import lombok.AccessLevel;
12  import lombok.AllArgsConstructor;
13  import org.reactivestreams.Publisher;
14  
15  import java.util.function.Consumer;
16  import java.util.function.Function;
17  
18  @AllArgsConstructor(access = AccessLevel.PRIVATE)
19  public class IOMonad<W,T> implements IO<T> {
20      private final Monad<W> monad;
21      private final Higher<W,T> pub;
22      private final ToPublsher<W> toPublsher;
23      private final FromPublsher<W> fromPublsher;
24  
25      public static Converter<reactiveSeq> reactiveSeqConverter = new Converter<reactiveSeq>() {
26          @Override
27          public <T> Function<? super Publisher<? extends T>, ? extends Higher<reactiveSeq, T>> fromPublisherFn() {
28              return Spouts::from;
29          }
30  
31          @Override
32          public Monad<reactiveSeq> monad() {
33              return PublisherInstances.monad();
34          }
35  
36          @Override
37          public <T> Function<Higher<reactiveSeq, T>, Publisher<T>> toPublisherFn() {
38              return ReactiveSeq::narrowK;
39          }
40      };
41      public static Converter<future> futureConverter = new Converter<future>() {
42          @Override
43          public <T> Function<? super Publisher<? extends T>, ? extends Higher<future, T>> fromPublisherFn() {
44              return p->Future.fromPublisher(p);
45          }
46  
47          @Override
48          public Monad<future> monad() {
49              return FutureInstances.monad();
50          }
51  
52          @Override
53          public <T> Function<Higher<future, T>, Publisher<T>> toPublisherFn() {
54              return Future::narrowK;
55          }
56      };
57     public interface Converter<W> extends ToPublsher<W>, FromPublsher<W>{
58          Monad<W> monad();
59      }
60      public interface ToPublsher<W>{
61          <T> Function<Higher<W,T>,Publisher<T>> toPublisherFn();
62      }
63      public interface FromPublsher<W>{
64          <T> Function<? super Publisher<? extends T>,? extends Higher<W,T>> fromPublisherFn();
65      }
66  
67      public static <W,T> IOMonad<W,T> ioMonad(Monad<W> monad, Higher<W,T> hkt, ToPublsher<W> to, FromPublsher<W> from){
68          return new IOMonad<W,T>(monad,hkt,to,from);
69      }
70      public static <W,T> IOMonad<W,T> ioMonad(Converter<W>  converter,Higher<W,T> hkt){
71          return new IOMonad<W,T>(converter.monad(),hkt,converter,converter);
72      }
73  
74  
75      @Override
76      public <R> IO<R> map(Function<? super T, ? extends R> s) {
77          return new IOMonad<W,R>(monad,monad.map(s,pub),toPublsher,fromPublsher);
78      }
79  
80      @Override
81      public <R> IO<R> flatMap(Function<? super T, IO<? extends R>> s) {
82          return new IOMonad<W,R>(monad,monad.flatMap(s.andThen(io->fromPublsher.<R>fromPublisherFn().apply(io.publisher())),pub),toPublsher,fromPublsher);
83      }
84  
85      @Override
86      public <R> IO<R> mergeMap(int maxConcurrency, Function<? super T, Publisher<? extends R>> s) {
87          ReactiveSeq<R> r = Spouts.from(this.toPublsher.<T>toPublisherFn().apply(pub)).mergeMap(maxConcurrency, s);
88          return new IOMonad<W,R>(monad,fromPublsher.<R>fromPublisherFn().apply(r),toPublsher,fromPublsher);
89  
90      }
91  
92  
93      @Override
94      public Publisher<T> publisher() {
95          return toPublsher.<T>toPublisherFn().apply(pub);
96      }
97  
98      public String toString(){
99          return "IO["+ run().toString() + "]";
100     }
101 }
