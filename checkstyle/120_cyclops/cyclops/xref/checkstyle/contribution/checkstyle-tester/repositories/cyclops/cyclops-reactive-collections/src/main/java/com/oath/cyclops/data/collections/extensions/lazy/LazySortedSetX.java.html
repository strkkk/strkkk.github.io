12  /**
13   * An extended Set type {@see java.util.List}
14   * Extended Set operations execute lazily e.g.
15   * <pre>
16   * {@code
17   *    SortedSetX<Integer> q = SortedSetX.of(1,2,3)
18   *                                      .map(i->i*2);
19   * }
20   * </pre>
21   * The map operation above is not executed immediately. It will only be executed when (if) the data inside the
22   * queue is accessed. This allows lazy operations to be chained and executed more efficiently e.g.
23   *
24   * <pre>
25   * {@code
26   *    SortedSetX<Integer> q = SortedSetX.of(1,2,3)
27   *                                      .map(i->i*2);
28   *                                      .filter(i->i<5);
29   * }
30   * </pre>
31   *
32   * The operation above is more efficient than the equivalent operation with a ListX.
33   *
34   * @author johnmcclean
35   *
36   * @param <T> the type of elements held in this toX
37   */
38  public class LazySortedSetX<T> extends AbstractLazyCollection<T,SortedSet<T>> implements SortedSetX<T> {
39  
40      public static final <T> Function<ReactiveSeq<SortedSet<T>>, SortedSet<T>> asyncSortedSet() {
41          return r -> {
42              CompletableSortedSetX<T> res = new CompletableSortedSetX<>();
43              r.forEachAsync(l -> res.complete(l));
44              return res.asSortedSetX();
45          };
46      }
47  
48      public LazySortedSetX(SortedSet<T> list, ReactiveSeq<T> seq, Collector<T, ?, SortedSet<T>> collector,Evaluation strict) {
49          super(list, seq, collector,strict,asyncSortedSet());
50  
51      }
52      public LazySortedSetX(SortedSet<T> list, Collector<T, ?, SortedSet<T>> collector,Evaluation strict) {
53          super(list, null, collector,strict,asyncSortedSet());
54  
55      }
56  
57      public LazySortedSetX(ReactiveSeq<T> seq, Collector<T, ?, SortedSet<T>> collector,Evaluation strict) {
58          super(null, seq, collector,strict,asyncSortedSet());
59  
60      }
61      @Override
62      public SortedSetX<T> lazy() {
63          return new LazySortedSetX<T>(getList(),getSeq().get(),getCollectorInternal(), Evaluation.LAZY) ;
64      }
65  
66      @Override
67      public SortedSetX<T> eager() {
68          return new LazySortedSetX<T>(getList(),getSeq().get(),getCollectorInternal(),Evaluation.EAGER) ;
69      }
70      @Override
71      public LazySortedSetX<T> type(Collector<T, ?, SortedSet<T>> collector){
72          return (LazySortedSetX)new LazySortedSetX<T>(this.getList(),this.getSeq().get(),collector, evaluation());
73      }
74      //@Override
75      public SortedSetX<T> materialize() {
76          get();
77          return this;
78      }
79  
80      @Override
81      public <T1> Collector<T1, ?, SortedSet<T1>> getCollector() {
82          return (Collector)super.getCollectorInternal();
83      }
84  
85  
86  
87      @Override
88      public <X> LazySortedSetX<X> fromStream(ReactiveSeq<X> stream) {
89  
90          return new LazySortedSetX<X>((SortedSet)getList(),ReactiveSeq.fromStream(stream),(Collector)this.getCollectorInternal(), evaluation());
91      }
92  
93      @Override
94      public <T1> LazySortedSetX<T1> from(Iterable<T1> c) {
95          if(c instanceof Set)
96              return new LazySortedSetX<T1>((SortedSet)c,null,(Collector)this.getCollectorInternal(), evaluation());
97          return fromStream(ReactiveSeq.fromIterable(c));
98      }
99  
100     @Override
101     public <U> LazySortedSetX<U> unitIterable(Iterable<U> it) {
102         return fromStream(ReactiveSeq.fromIterable(it));
103     }
104 
105 
106 
107     @Override
108     public <R> LazySortedSetX<R> unit(Iterable<R> col) {
109         return from(col);
110     }
111 
112 
113     @Override
114     public Comparator<? super T> comparator() {
115         return get().comparator();
116     }
117 
118     @Override
119     public SortedSet<T> subSet(T fromElement, T toElement) {
120         return get().subSet(fromElement,toElement);
121     }
122 
123     @Override
124     public SortedSet<T> headSet(T toElement) {
125         return get().headSet(toElement);
126     }
127 
128     @Override
129     public SortedSet<T> tailSet(T fromElement) {
130         return get().tailSet(fromElement);
131     }
132 
133     @Override
134     public T first() {
135         return get().first();
136     }
137 
138     @Override
139     public T last() {
140         return get().last();
