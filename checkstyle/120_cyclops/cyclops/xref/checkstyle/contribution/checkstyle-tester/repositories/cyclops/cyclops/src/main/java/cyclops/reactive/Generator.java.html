141         public Generator<T> yieldAll(Iterable<T>... next){
142             if(next.length==0)
143                 return empty(this);
144             ReactiveSeq[] array= new ReactiveSeq[next.length];
145             int index =0;
146             for(Iterable<T> it : next){
147                 array[index++] = ReactiveSeq.fromIterable(it);
148             }
149 
150             return yieldAll(ReactiveSeq.concat(array));

153         public Generator<T> yieldAll(Iterable<T> next){
154 
155 
156             Iterator<T> it = next.iterator();
157             if(it.hasNext()) {
158                 T nextA = it.next();
159                 return yieldAll(nextA, () -> it);
160             }
161             return empty(this);

163         public Generator<T> yieldAll(T value, Iterable<T> next){
164

351     @Override
352     public Iterator<T> iterator() {
353 
354         return new Iterator<T>() {
355             Generator<T> nextCont = copy();
356             boolean ready = nextCont.value.isPresent();
357             @Override
358             public boolean hasNext() {
359                 while(!ready){
360                     if(nextCont.remainderOfWorkToBeDone==null){
361                         return false;
362                     }
363                     nextCont = nextCont.proceed().orElse(null);
364 
365                     ready = nextCont==null? false : nextCont.value.isPresent();
366                 }
367                 return true;
368 
369             }
370 
371             @Override
372             public T next() {
373                 ready = false;
374                 return nextCont.extract();
375             }
376         };
