145      *
146      * <pre>
147      * {@code
148      *      import static cyclops2.function.Predicates.some;
149      *
150      *      Seq.of(Arrays.asList(),Arrays.asList(1),null, Optional.zero(),Maybe.none())
151      *            .filter(some());
152      *
153      *       //Seq[List[1]]
154      * }
155      * </pre>
156      *
157      * @return A predicate that checks for a values presence (i.e. for standard values that they are non-null,
158      *  for Optionals that they are present, for cyclops2-react values that they are present and for Iterables that they are
159      *  non-null).
160      */
161     public static <T> Predicate<T> some() {
162         return Predicates.<T> p(t -> t != null)
163                          .and(not(optionalPresent()))
164                          .and(not(valuePresent()))
165                          .and(not(iterablePresent()));
166     }

255     /**
256      * Check for universal equality (Object#equals)
257      *
258      * Filtering example
259      * <pre>
260      * {@code
261      *  ReactiveSeq.of(1,2,3).filter(anyOf(not(eq(2)),in(1,10,20)));
262      *  //ReactiveSeq[1]
263      *
264      * }
265      * </pre>
266      *
267      * Pattern Matching Example
268      *
269      *  <pre>
270      *  {@code
271      *   Eval<String> url = Xors.url(new URL("http://www.aol.com/path?q=hello"))
272                                      .on$12_45()
273                                      .matches(c->c.is(when(eq("http"),in("www.aol.com","aol.com"),any(),not(eq("q=hello!"))), transform("correct")),otherwise("miss"));
274 
275         //Eval.now("correct");
276      *
277      *
278      *  }
279      *  </pre>
280      *
281      * @param value Value to check equality of
282      * @return Predicate for equality
283      */
284     public static <V> Predicate<V> eq(final V value) {
285 
286         return test -> Objects.equals(test, value);
287     }

293     /**
294      *  Test for equivalence
295      *  null eqv to absent, embedded value equivalency, non-values converted to values before testing
296      *.
297      * <pre>
298      * {@code
299      *
300      *   Stream.of(Maybe.of(2))
301      *         .filter(eqv(Maybe.of(2)))
302      *         .forEach(System.out::println);
303      *
304      *   //Maybe[2]
305      *
306      *   Stream.of(2)
307      *         .filter(eqv(Maybe.of(2)))
308      *         .forEach(System.out::println);
309      *
310      *   //2      (passes filter as equivalent to Maybe[2])
311      * }</pre>
312      *
313      * @param value
314      * @return
315      */
316     public static <V> Predicate<Value<? super V>> eqv(final Value<? super V> value) {
317 
318         return test -> test == null ? value == null ? true : !value.toMaybe()
319                                                                    .isPresent()
320                 : test.toMaybe()
321                       .equals(value.toMaybe());
322 
323     }

329                 : Seq.fromIterable(test)

464     /**
465      * Samples a dataset by only returning true when the modulus of the event count divided by the rate is 0
466      * e.g.
467      * To select every second member of a Stream, filter with Predicates.sample(2) - for every thid member filter with Predicates.sample(3) and so on.
468      * To take 1% of a Stream use Predicates.sample(100)
469      *
470      * @param rate Every x element to include in your sample
471      * @param <T> Data type to sample
472      * @return Sampling Predicate
473      */
474     public static <T> Predicate<T> sample(int rate){
475         return new Sample<T>(rate);
476     }
477     public static <T> Predicate<T> sample(long time,TimeUnit unit){
478         return new TimeSample<T>(unit.toNanos(time));

485     private static class TimeSample<T> implements Predicate<T>{
486         private final AtomicReference<SimpleTimer> timer = new AtomicReference(null);
487         private final  long nanos;
488 
489 
490         public TimeSample(long nanos) {
491             this.nanos = nanos;
492         }
493 
494         @Override
495         public boolean test(T t) {
496             if(timer.get()==null){
497                 timer.set(new SimpleTimer());
498                 return true;
499             }
500             if(timer.get().getElapsedNanoseconds() > nanos){
501                 timer.set(null);
502                 return true;
503             }
504             return false;
505         }
506 
507     }

509     private static class Sample<T> implements Predicate<T>{
510         private final int rate;
511 
512         private AtomicInteger count = new AtomicInteger(0);
513 
514         public Sample(int rate) {
515             this.rate = rate;
516         }
517 
518         @Override
519         public boolean test(T t) {
520             return count.incrementAndGet() % rate ==0;
521         }
