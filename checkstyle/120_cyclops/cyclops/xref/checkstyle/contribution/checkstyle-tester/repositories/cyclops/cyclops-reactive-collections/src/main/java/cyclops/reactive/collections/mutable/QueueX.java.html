41  /**
42   * An eXtended Queue type, that offers additional functional style operators such as bimap, filter and more
43   * Can operate eagerly, lazily or reactively (async push)
44   *
45   * @author johnmcclean
46   *
47   * @param <T> the type of elements held in this QueueX
48   */
49  public interface QueueX<T> extends To<QueueX<T>>,Queue<T>,
50                                     LazyCollectionX<T>,
51                                      OnEmptySwitch<T, Queue<T>>,
52                                      Higher<queue,T>{

54      public static <T> QueueX<T> defer(Supplier<QueueX<T>> s){
55        return of(s)
56                .map(Supplier::get)
57                .concatMap(l->l);
58      }
59      default Tuple2<QueueX<T>, QueueX<T>> splitAt(int n) {
60          materialize();
61          return Tuple.tuple(take(n), drop(n));
62      }
63      default Tuple2<QueueX<T>, QueueX<T>> span(Predicate<? super T> pred) {
64          return tuple(takeWhile(pred), dropWhile(pred));
65      }
66  
67      default Tuple2<QueueX<T>,QueueX<T>> splitBy(Predicate<? super T> test) {
68          return span(test.negate());
69      }
70      default Tuple2<QueueX<T>, QueueX<T>> partition(final Predicate<? super T> splitter) {
71  
72          return tuple(filter(splitter), filter(splitter.negate()));
73  
74      }

287     public static <T> QueueX<T> fromIterable(final Iterable<T> it) {
288 
289         if (it instanceof QueueX)
290             return (QueueX) it;
291         if (it instanceof Queue)
292             return new LazyQueueX<T>(
293                                      (Queue) it, defaultCollector(),Evaluation.LAZY);
294         return new LazyQueueX<T>(null,
295                                  ReactiveSeq.fromIterable(it),
296                                             defaultCollector(),Evaluation.LAZY);
297     }

299     public static <T> QueueX<T> fromIterable(final Collector<T, ?, Queue<T>> collector, final Iterable<T> it) {
300         if (it instanceof QueueX)
301             return ((QueueX) it).withCollector(collector);
302         if (it instanceof Deque)
303             return new LazyQueueX<T>(
304                                      (Queue) it, collector,Evaluation.LAZY);
305         return new LazyQueueX<T>(
306                 null,
307                 ReactiveSeq.fromIterable(it),
308                 collector,Evaluation.LAZY);
309     }

410      * @return Transformed Queue

413         return fn.andThen(r ->  this.<R>unit(r))
414                 .apply(this);

1062     public static  <T,R> QueueX<R> tailRec(T initial, Function<? super T, ? extends QueueX<? extends Either<T, R>>> fn) {
1063         ListX<Either<T, R>> lazy = ListX.of(Either.left(initial));
1064         ListX<Either<T, R>> next = lazy.eager();
1065         boolean newValue[] = {true};
1066         for(;;){
1067 
1068             next = next.concatMap(e -> e.fold(s -> {
1069                         newValue[0]=true;
1070                         return fn.apply(s); },
1071                     p -> {
1072                         newValue[0]=false;
1073                         return ListX.of(e);
1074                     }));
1075             if(!newValue[0])
1076                 break;
1077 
1078         }
1079         return Either.sequenceRight(next)
1080                      .orElse(ReactiveSeq.empty())
1081                      .to(ReactiveConvertableSequence::converter)
1082                       .queueX(Evaluation.LAZY);
1083     }
