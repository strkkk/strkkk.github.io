109     default <R1, R2, A1, A2> Tuple2<R1, R2> collect(Collector<? super T, A1, R1> c1, Collector<? super T, A2, R2> c2) {
110         return stream().collect(Collector.of(() -> Tuple.tuple(c1.supplier().get(),c2.supplier().get()),
111                 (t2, next) -> {
112                     c1.accumulator().accept(t2._1(), next);
113                     c2.accumulator().accept(t2._2(), next);
114                 },(t2, t2b) -> Tuple.tuple(c1.combiner().apply(t2._1(), t2b._1()),c2.combiner().apply(t2._2(), t2b._2())),
115                 t2 -> Tuple.tuple(c1.finisher().apply(t2._1()),c2.finisher().apply(t2._2()))));
116     }
117     default <R1, R2, R3, A1, A2, A3> Tuple3<R1, R2, R3> collect(Collector<? super T, A1, R1> c1, Collector<? super T, A2, R2> c2, Collector<? super T, A3, R3> c3) {
118         return stream().collect(Collector.of(() -> Tuple.tuple(c1.supplier().get(),c2.supplier().get(),c3.supplier().get()),
119                 (t3, next) -> {
120                     c1.accumulator().accept(t3._1(), next);
121                     c2.accumulator().accept(t3._2(), next);
122                     c3.accumulator().accept(t3._3(), next);
123                 },(t3, t3b) -> Tuple.tuple(c1.combiner().apply(t3._1(), t3b._1()),c2.combiner().apply(t3._2(), t3b._2()),c3.combiner().apply(t3._3(), t3b._3())),
124                 t3 -> Tuple.tuple(c1.finisher().apply(t3._1()),c2.finisher().apply(t3._2()),c3.finisher().apply(t3._3()))));
125     }

176     default Seq<Tuple2<T,BigDecimal>> withPercentiles(){
177 
178         Seq<T> list = stream().seq();
179 
180         int precision = new Double(Math.log10(list.size())).intValue();
181 
182 
183         return list.zipWithIndex().map(t -> t.map2(idx -> {
184             double d = (idx / new Double(list.size()));
185             return new BigDecimal((d*100),new MathContext(precision));
186         }));
187 
188     }
189     /*
190         Value at percentile denoted by a double value between 0 and 100
191         Assumes the data is already sorted
192         Returns an Option, Option.none is returned when an invalid percentile or empty dataset is used
193      */
194     default Option<T> atPercentile(double percentile){
195         List<T> list = stream().collect(Collectors.toList());
196         if(list.size()==0)
197             return Option.none();
198         Long pos = new Double(Math.ceil( (percentile / 100) * list.size())).longValue();
199         if(pos<0 || pos > list.size()){
200             return Option.none();
201         }
202         return Option.some(list.get(pos.intValue()));
203     }

206     default double variance(ToDoubleFunction<T> fn){
207         Seq<T> list = stream().seq();
208         double avg = list.collect(Collectors.<T>averagingDouble(fn));
209         return (list.map(t -> fn.applyAsDouble(t))
210                 .map(t -> t - avg)
211                 .map(t -> t * t)
212                 .sumDouble(i -> i))/(list.size()-1);
213 
214     }
215     default double populationVariance(ToDoubleFunction<T> fn){
216         Seq<T> list = stream().seq();
217         double avg = list.collect(Collectors.<T>averagingDouble(fn));
218         return (list.map(t -> fn.applyAsDouble(t))
219                 .map(t -> t - avg)
220                 .map(t -> t * t)
221                 .sumDouble(i -> i)/(list.size()));
222 
223     }

225     default double stdDeviation(ToDoubleFunction<T> fn){
226         Seq<T> list = stream().seq();
227         double avg = list.collect(Collectors.<T>averagingDouble(fn));
228         return Math.sqrt( list.stream().mapToDouble(fn)
229                 .map(i->i-avg)
230                 .map(i->i*i)
231                 .average()
232                 .getAsDouble());

757     /**
758      * Check that the predicate matches at least the specified number of times
759      */
760     default boolean atLeast(final int num, final Predicate<? super T> c) {
761         int[] matched = {0};
762         return stream().filter(n->{
763             if(c.test(n)) {
764                 matched[0]++;
765                 return true;
766             }
767             return false;
768         }).takeWhileInclusive(n->matched[0]<num).count()>=num;
769     }
770 
771     /**
772      * Check that the predicate matches at most the specified number of times
773      */
774     default boolean atMost(final int num, final Predicate<? super T> c) {
775         int[] matched = {0};
776         return stream().filter(n->{
777             if(c.test(n)) {
778                 matched[0]++;
779                 return true;
780             }
781             return false;
782         }).takeWhileInclusive(n->matched[0]<num+1).count()<num+1;
783     }
