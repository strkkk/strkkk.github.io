64      default ImmutableQueue<T> replace(T currentElement, T newElement){
65          ImmutableQueue<T> preceding = emptyUnit();
66          ImmutableQueue<T> tail = this;
67          while(!tail.isEmpty()){
68              ImmutableQueue<T> ref=  preceding;
69              ImmutableQueue<T> tailRef = tail;
70              Tuple3<ImmutableQueue<T>, ImmutableQueue<T>, Boolean> t3 = tail.fold(c -> {
71                  if (Objects.equals(c.head(), currentElement))
72                      return Tuple.tuple(ref, tailRef, true);
73                  return Tuple.tuple(ref.prepend(c.head()), c.tail(), false);
74              }, nil -> Tuple.tuple(ref, tailRef, true));
75  
76              preceding = t3._1();
77              tail = t3._2();
78              if(t3._3())
79                  break;
80  
81          }
82  
83          ImmutableQueue<T> start = preceding;
84          return tail.fold(cons->cons.tail().prepend(newElement).prependAll(start), nil->this);
85      }
86      default ImmutableQueue<T> removeFirst(Predicate<? super T> pred){
87          ImmutableQueue<T> res[] = new ImmutableQueue[]{emptyUnit()};
88          ImmutableQueue<T> rem = this;
89          boolean[] found = {false};
90          do {
91              rem = rem.fold(s -> {
92                  return s.fold((head, tail2) -> {
93                      found[0] = pred.test(head);
94                      if(!found[0]) {
95                          res[0] = res[0].prepend(head);
96                          return tail2;
97                      }
98                      return tail2;
99                  });
100             }, n -> n);
101         }while(!rem.isEmpty() && !found[0]);
102 
103         ImmutableQueue<T> ar = rem.fold(s -> s.fold((h, t) -> t), n -> n);
104         return res[0].foldLeft(ar, (a,b)->a.prepend(b));
105 
106     }
