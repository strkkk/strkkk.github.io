38  /**
39   *  Either left or right.
40   *
41   * 'Right' (or right type) biased disjunct union.
42   *
43   *  No 'projections' are provided, swap() and leftXXXX alternative methods can be used instead.
44   *
45   *  Either is used to represent values that can be one of two states (for example a validation result, lazy everything is ok - or we have an error).
46   *  It can be used to avoid a common design anti-pattern where an Object has two fields one of which is always null (or worse, both are defined as Optionals).
47   *
48   *  <pre>
49   *  {@code
50   *
51   *     public class Member{
52   *           Either<SeniorTeam,JuniorTeam> team;
53   *     }
54   *
55   *     Rather than
56   *
57   *     public class Member{
58   *           @Setter
59   *           SeniorTeam seniorTeam = null;
60   *           @Setter
61   *           JuniorTeam juniorTeam = null;
62   *     }
63   *  }
64   *  </pre>
65   *
66   *  Either's have two states
67   *  Right : Most methods operate naturally on the right type, if it is present. If it is not, nothing happens.
68   *  Left : Most methods do nothing to the left type if it is present.
69   *              To operate on the Left type first call swap() or use left analogs of the main operators.
70   *
71   *  Instantiating an Either - Right
72   *  <pre>
73   *  {@code
74   *      Either.right("hello").map(v->v+" world")
75   *    //Either.right["hello world"]
76   *  }
77   *  </pre>
78   *
79   *  Instantiating an Either - Left
80   *  <pre>
81   *  {@code
82   *      Either.left("hello").map(v->v+" world")
83   *    //Either.seconary["hello"]
84   *  }
85   *  </pre>
86   *
87   *  Either can operate (via transform/flatMap) as a Functor / Monad and via combine as an ApplicativeFunctor
88   *
89   *   Values can be accumulated via
90   *  <pre>
91   *  {@code
92   *  Either.accumulateLeft(Seq.of(Either.left("failed1"),
93                                                      Either.left("failed2"),
94                                                      Either.right("success")),
95                                                      SemigroupK.stringConcat)
96   *
97   *  //failed1failed2
98   *
99   *   Either<String,String> fail1 = Either.left("failed1");
100      fail1.swap().combine((a,b)->a+b)
101                  .combine(Either.left("failed2").swap())
102                  .combine(Either.<String,String>right("success").swap())
103  *
104  *  //failed1failed2
105  *  }
106  *  </pre>
107  *
108  *
109  * For Inclusive Ors @see Ior
110  *
111  * @author johnmcclean
112  *
113  * @param <LT> Left type
114  * @param <RT> Right type
115  */
116 public interface Either<LT, RT> extends To<Either<LT, RT>>,
117                                          BiTransformable<LT, RT>,
118                                          Sealed2<LT, RT>,Value<RT>,
119                                          OrElseValue<RT,Either<LT, RT>>,
120                                          Unit<RT>, Transformable<RT>, Filters<RT>,
121                                          Serializable,
122                                          Higher2<either, LT, RT> {

127     public static  <L,T,R> Either<L,R> tailRec(T initial, Function<? super T, ? extends Either<L,? extends Either<T, R>>> fn){
128         Either<L,? extends Either<T, R>> next[] = new Either[1];
129         next[0] = Either.right(Either.left(initial));
130         boolean cont = true;
131         do {
132             cont = next[0].fold(p -> p.fold(s -> {
133                 next[0] = narrowK(fn.apply(s));
134                 return true;
135             }, pr -> false), () -> false);
136         } while (cont);
137 
138         return next[0].map(x->x.fold(l->null, r->r));
139     }

464   public static  <L,T> Either<L,Stream<T>> sequence(Stream<? extends Either<L,T>> stream) {
465 
466     Either<L, Stream<T>> identity = Either.right(ReactiveSeq.empty());
467 
468     BiFunction<Either<L,Stream<T>>,Either<L,T>,Either<L,Stream<T>>> combineToStream = (acc,next) ->acc.zip(next,(a,b)->ReactiveSeq.fromStream(a).append(b));
469 
470     BinaryOperator<Either<L,Stream<T>>> combineStreams = (a,b)-> a.zip(b,(z1,z2)->ReactiveSeq.fromStream(z1).appendStream(z2));
471 
472    return stream.reduce(identity,combineToStream,combineStreams);
473   }
