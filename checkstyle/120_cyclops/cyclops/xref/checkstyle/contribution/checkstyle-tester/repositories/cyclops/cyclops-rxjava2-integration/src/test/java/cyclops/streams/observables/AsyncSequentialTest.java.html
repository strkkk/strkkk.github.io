49      protected <U> ReactiveSeq<U> of(U... array){
50          int[] index = {0};
51          ReactiveSeq<U> seq = Spouts.async(s->{
52  
53              new Thread(()-> {
54  
55                  for (U next : array) {
56                      s.onNext(next);
57                      if(index[0]++>100)
58                          break;
59                  }
60                  s.onComplete();
61              }).start();
62  
63          });
64          return ObservableReactiveSeq.reactiveSeq(Observables.observableFrom(seq));
65      }

488     protected <U> ReactiveSeq<U> ofWait(U... array){
489         int[] index = {0};
490         return Spouts.async(s->{
491 
492             new Thread(()-> {
493                 System.out.println("Pushing data from " + Thread.currentThread().getId());
494                 try {
495                     Thread.sleep(500);
496                 } catch (InterruptedException e) {
497                     e.printStackTrace();
498                 }
499                 for (U next : array) {
500                     s.onNext(next);
501                     if(index[0]++>100)
502                         break;
503                 }
504                 s.onComplete();
505             }).start();
506 
507         });
508     }
