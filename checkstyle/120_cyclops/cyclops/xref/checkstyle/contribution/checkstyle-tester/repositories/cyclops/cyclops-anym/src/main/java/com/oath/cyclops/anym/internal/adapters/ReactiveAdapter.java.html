1   package com.oath.cyclops.anym.internal.adapters;
2   
3   import com.oath.cyclops.anym.extensability.AbstractMonadAdapter;
4   import com.oath.cyclops.internal.stream.ReactiveStreamX;
5   import cyclops.monads.AnyM;
6   import cyclops.monads.Witness;
7   import cyclops.monads.Witness.StreamWitness;
8   import cyclops.reactive.ReactiveSeq;
9   import cyclops.reactive.Spouts;
10  import lombok.AllArgsConstructor;
11  import org.reactivestreams.Publisher;
12  
13  import java.util.Iterator;
14  import java.util.function.BiFunction;
15  import java.util.function.Function;
16  import java.util.function.Predicate;
17  import java.util.function.Supplier;
18  import java.util.stream.Stream;
19  
20  import static cyclops.monads.AnyM.fromStream;
21  
22  @AllArgsConstructor
23  public class ReactiveAdapter<W extends StreamWitness<W>> extends AbstractMonadAdapter<W> {
24  
25      private final Supplier<Stream<?>> empty;
26      private final Function<?,Stream<?>> unit;
27      private final Function<Iterator<?>,Stream<?>> unitIterator;
28      private final W witness;
29  
30  
31      public final static ReactiveAdapter reactiveSeq = new ReactiveAdapter(()->Spouts.of(), t->Spouts.of(t), it->(Stream)ReactiveSeq.fromIterator((Iterator)it),Witness.reactiveSeq.REACTIVE);
32  
33      private <U> Supplier<Stream<U>> getEmpty(){
34          return (Supplier)empty;
35      }
36      private <U> Function<U,Stream<U>>  getUnit(){
37          return (Function)unit;
38      }
39      private <U> Function<Iterator<U>,Stream<U>>  getUnitIterator(){
40          return (Function)unitIterator;
41      }
42      @Override
43      public <T> Iterable<T> toIterable(AnyM<W, T> t) {
44          return ()->stream(t).iterator();
45      }
46  
47  
48      @Override
49      public <T> AnyM<W, T> filter(AnyM<W, T> t, Predicate<? super T> fn) {
50          return fromStream(stream(t).filter(fn),witness);
51      }
52  
53      <T> ReactiveSeq<T> stream(AnyM<W,T> anyM){
54          return anyM.unwrap();
55      }
56  
57      @Override
58      public <T> AnyM<W, T> empty() {
59          return fromStream(this.<T>getEmpty().get(),witness);
60      }
61  
62  
63    @Override
64    public <T, T2, R> AnyM<W, R> zip(AnyM<W, ? extends T> t, AnyM<W, ? extends T2> t2, BiFunction<? super T, ? super T2, ? extends R> fn) {
65      return fromStream(stream(t).zip((a,b)->fn.apply(a,b),stream(t2)),witness);
66    }
67  
68    @Override
69      public <T, R> AnyM<W, R> ap(AnyM<W,? extends Function<? super T,? extends R>> fn, AnyM<W, T> apply) {
70           return fromStream(stream(apply).zip((a,b)->b.apply(a),stream(fn)),witness);
71      }
72  
73      @Override
74      public <T, R> AnyM<W, R> flatMap(AnyM<W, T> t,
75              Function<? super T, ? extends AnyM<W, ? extends R>> fn) {
76          return fromStream(((Stream)t.unwrap()).flatMap(fn.andThen(a-> (Stream)a.unwrap())),witness);
77      }
78  
79      @Override
80      public <T> AnyM<W, T> unitIterable(Iterable<T> it)  {
81  
82          if(it instanceof ReactiveSeq){
83              W witnessToUse = it instanceof ReactiveStreamX ? witness : (W)Witness.reactiveSeq.ITERATIVE;
84              return fromStream((ReactiveSeq<T>)it,witnessToUse);
85          }
86          if(it instanceof Publisher){
87              return fromStream(Spouts.from((Publisher)it),witness);
88          }
89         return fromStream(this.<T>getUnitIterator().apply(it.iterator()),witness);
90      }
91  
92      @Override
93      public <T> AnyM<W, T> unit(T o) {
94          return fromStream(this.<T>getUnit().apply(o),witness);
95      }
96  
97      @Override
98      public <T, R> AnyM<W, R> map(AnyM<W, T> t, Function<? super T, ? extends R> fn) {
99          return fromStream(((Stream)t.unwrap()).map(fn),witness);
100     }
101 }
