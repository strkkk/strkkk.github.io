425     public  <B,R> Free<F,R> zip(Functor<F> f,Free<F,B> b,BiFunction<? super T,? super B,? extends R> zipper){
426 
427         Either<Higher<F, Free<F, T>>, T> first = resume(f);
428         Either<Higher<F, Free<F, B>>, B> second = b.resume(f);
429 
430         if(first.isLeft() && second.isLeft()) {
431             return suspend(f.map_(first.leftOrElse(null), a1->{
432                 return suspend(f.map_(second.leftOrElse(null), b1->a1.zip(f,b1,zipper)));
433             }));
434         }
435         if(first.isRight() && second.isRight()){
436             return done(zipper.apply(first.orElse(null),second.orElse(null)));
437         }
438         if(first.isLeft() && second.isRight()){
439             return suspend(f.map_(first.leftOrElse(null), a1->a1.zip(f,b,zipper)));
440 
441         }
442         if(first.isRight() && second.isLeft()){
443             return suspend(f.map_(second.leftOrElse(null), a1->Free.<F,T>done(first.orElse(null)).zip(f,b,zipper)));
444         }
445         return null;
446     }

452     public  <B,C,R> Free<F,R> zip(Functor<F> f,Free<F,B> b, Free<F,C> c, Function3<? super T, ? super B, ? super C,? extends R> fn){
453 
454         Either<Higher<F, Free<F, T>>, T> first = resume(f);
455         Either<Higher<F, Free<F, B>>, B> second = b.resume(f);
456         Either<Higher<F, Free<F, C>>, C> third = c.resume(f);
457 
458         if(first.isLeft() && second.isLeft() && third.isLeft()) {
459             return suspend(f.map_(first.leftOrElse(null), a1->{
460                 return suspend(f.map_(second.leftOrElse(null), b1->{
461                     return suspend(f.map_(third.leftOrElse(null), c1->a1.zip(f,b1,c1,fn)));
462                 }));
463             }));
464         }
465 
466         if(first.isRight() && second.isRight() && third.isRight()){
467             return done(fn.apply(first.orElse(null),second.orElse(null),third.orElse(null)));
468         }
469 
470         if(first.isLeft() && second.isRight() && third.isRight()){
471             return suspend(f.map_(first.leftOrElse(null), a1->a1.zip(f,b,c,fn)));
472         }
473         if(first.isRight() && second.isLeft() && third.isRight()){
474 
475             return suspend(f.map_(second.leftOrElse(null), b1->this.zip(f,b1,c,fn)));
476 
477 
478 
479         }
480         if(first.isRight() && second.isRight() && third.isLeft()){
481             return suspend(f.map_(third.leftOrElse(null), c1->this.zip(f,b,c1,fn)));
482         }
483 
484 
485         if(first.isRight() && second.isLeft() && third.isLeft()){
486             return suspend(f.map_(second.leftOrElse(null), b1->{
487                 return suspend(f.map_(third.leftOrElse(null), c1->this.zip(f,b1,c1,fn)));
488             }));
489 
490         }
491         if(first.isLeft() && second.isRight() && third.isLeft()){
492             return suspend(f.map_(first.leftOrElse(null), a1->{
493 
494                 return suspend(f.map_(third.leftOrElse(null), c1->a1.zip(f,b,c1,fn)));
495 
496             }));
497         }
498         if(first.isLeft() && second.isLeft() && third.isRight()){
499             return suspend(f.map_(first.leftOrElse(null), a1->{
500                 return suspend(f.map_(second.leftOrElse(null), b1->a1.zip(f,b1,c,fn)));
501 
502             }));
503         }
