29  /**
30   * A right biased Lazy Either3 type. transform / flatMap operators are tail-call optimized
31   *
32   * Can be one of 3 types
33   *
34   *
35   *
36   * @author johnmcclean
37   *
38   * @param <LT1> Left1 type
39   * @param <LT2> Left2 type
40   * @param <RT> Right type (operations are performed on this type if present)
41   */
42  public interface LazyEither3<LT1, LT2, RT> extends Value<RT>,
43                                                  OrElseValue<RT,LazyEither3<LT1,LT2,RT>>,
44                                                  Unit<RT>,
45                                                  Transformable<RT>,
46                                                   Filters<RT>,
47                                                  BiTransformable<LT2, RT>,
48                                                  To<LazyEither3<LT1, LT2, RT>>,

229      *  Turn an IterableX of Either3 into a single Either with Lists of values.

235        Either3<String,String,Integer> none = Either3.left("none");

248     public static <LT1,LT2, PT> LazyEither3<LT1,LT2,ReactiveSeq<PT>> sequence(final Iterable<? extends LazyEither3<LT1, LT2, PT>> xors) {
249         Objects.requireNonNull(xors);

253   public static  <L1,L2,T> LazyEither3<L1,L2,ReactiveSeq<T>> sequence(ReactiveSeq<? extends LazyEither3<L1,L2,T>> stream) {
254 
255     LazyEither3<L1,L2, ReactiveSeq<T>> identity = right(ReactiveSeq.empty());
256 
257     BiFunction<LazyEither3<L1,L2,ReactiveSeq<T>>,LazyEither3<L1,L2,T>,LazyEither3<L1,L2,ReactiveSeq<T>>> combineToStream = (acc,next) ->acc.zip(next,(a,b)->a.append(b));
258 
259     BinaryOperator<LazyEither3<L1,L2,ReactiveSeq<T>>> combineStreams = (a,b)-> a.zip(b,(z1,z2)->z1.appendStream(z2));
260 
261     return stream.reduce(identity,combineToStream,combineStreams);
262   }
