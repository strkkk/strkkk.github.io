35                                       .unit("hello")
36                                       .applyHKT(h-> FutureInstances.functor().map((String v) ->v.length(), h))
37                                       .convert(Future::narrowK);

44              .ap(Future.ofResult(l1(this::multiplyByTwo)),Future.ofResult(1));

49      @Test
50      public void applicative(){
51  
52          Future<Function1<Integer,Integer>> optFn = FutureInstances.unit()
53                                                          .unit(Lambda.l1((Integer i) ->i*2))
54                                                          .convert(Future::narrowK);
55  
56          Future<Integer> opt = FutureInstances.unit()
57                                       .unit("hello")
58                                       .applyHKT(h-> FutureInstances.functor().map((String v) ->v.length(), h))
59                                       .applyHKT(h-> FutureInstances.applicative().ap(optFn, h))
60                                       .convert(Future::narrowK);
61  
62          assertThat(opt.toCompletableFuture().join(),equalTo(Future.ofResult("hello".length()*2).toCompletableFuture().join()));
63      }
