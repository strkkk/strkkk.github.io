28  /**
29   * A totally Lazy Either implementation with tail call optimization for transform and flatMap operators. LazyEither can operate reactively (i.e. suppports data arriving asynchronsouly
30   the monadic chain of computations will only be executed once data arrives).
31   *
32   * 'Right' (or right type) biased disjunct union.
33  
34   *  No 'projections' are provided, swap() and secondaryXXXX alternative methods can be used instead.
35   *
36   *  Either is used to represent values that can be one of two states (for example a validation result, lazy everything is ok - or we have an error).
37   *  It can be used to avoid a common design anti-pattern where an Object has two fields one of which is always null (or worse, both are defined as Optionals).
38   *
39   *  <pre>
40   *  {@code
41   *
42   *     public class Member{
43   *           LazyEither<SeniorTeam,JuniorTeam> team;
44   *     }
45   *
46   *     Rather than
47   *
48   *     public class Member{
49   *           @Setter
50   *           SeniorTeam seniorTeam = null;
51   *           @Setter
52   *           JuniorTeam juniorTeam = null;
53   *     }
54   *  }
55   *  </pre>
56   *
57   *  Either's have two states
58   *  Right : Most methods operate naturally on the right type, if it is present. If it is not, nothing happens.
59   *  Left : Most methods do nothing to the left type if it is present.
60   *              To operate on the Left type first call swap() or use left analogs of the main operators.
61   *
62   *  Instantiating a LazyEither - Right
63   *  <pre>
64   *  {@code
65   *      LazyEither.right("hello").map(v->v+" world")
66   *    //LazyEither.right["hello world"]
67   *  }
68   *  </pre>
69   *
70   *  Instantiating an LazyEither - Left
71   *  <pre>
72   *  {@code
73   *      LazyEither.left("hello").map(v->v+" world")
74   *    //LazyEither.left["hello"]
75   *  }
76   *  </pre>
77   *
78   *  LazyEither can operate (via transform/flatMap) as a Functor / Monad and via combine as an ApplicativeFunctor
79   *
80   *   Values can be accumulated via
81   *  <pre>
82   *  {@code
83   *  LazyEither.accumulateLeft(Seq.of(Either.left("failed1"),
84                                                      Either.left("failed2"),
85                                                      Either.right("success")),
86                                                      SemigroupK.stringConcat)
87   *
88   *  //failed1failed2
89   *
90   *   LazyEither<String,String> fail1 = Either.left("failed1");
91       fail1.swap().combine((a,b)->a+b)
92                   .combine(Either.left("failed2").swap())
93                   .combine(Either.<String,String>right("success").swap())
94   *
95   *  //failed1failed2
96   *  }
97   *  </pre>
98   *
99   *
100  * For Inclusive Ors @see Ior
101  *
102  * @author johnmcclean
103  *
104  * @param <LT> Left type
105  * @param <RT> Right type
106  */
107 public interface LazyEither<LT, RT> extends Either<LT, RT> {

118     public static  <L,T,R> LazyEither<L,R> tailRec(T initial, Function<? super T, ? extends LazyEither<L,? extends Either<T, R>>> fn){
119         LazyEither<L,? extends Either<T, R>> next[] = new LazyEither[1];
120         next[0] = LazyEither.right(Either.left(initial));
121         boolean cont = true;
122         do {
123             cont = next[0].fold(p -> p.fold(s -> {
124                 next[0] = fn.apply(s);
125                 return true;
126             }, pr -> false), () -> false);
127         } while (cont);
128         return next[0].map(x->x.orElse(null));
129     }

405         Objects.requireNonNull(xors);
406       LazyEither<PT, ReactiveSeq<LT1>> res = sequence(ReactiveSeq.fromIterable(xors)

411   public static  <L,T> LazyEither<L,ReactiveSeq<T>> sequence(ReactiveSeq<? extends LazyEither<L,T>> stream) {
412 
413     LazyEither<L, ReactiveSeq<T>> identity = right(ReactiveSeq.empty());
414 
415     BiFunction<LazyEither<L,ReactiveSeq<T>>,LazyEither<L,T>,LazyEither<L,ReactiveSeq<T>>> combineToStream = (acc,next) ->acc.zip(next,(a,b)->a.append(b));
416 
417     BinaryOperator<LazyEither<L,ReactiveSeq<T>>> combineStreams = (a,b)-> a.zip(b,(z1,z2)->z1.appendStream(z2));
418 
419     return stream.reduce(identity,combineToStream,combineStreams);
420   }

564     /**
565      * Lazily construct a Right Either from the supplied publisher
566      * <pre>
567      * {@code
568      *   ReactiveSeq<Integer> stream =  ReactiveSeq.of(1,2,3);
569 
570          Either<Throwable,Integer> future = Either.fromPublisher(stream);
571 
572          //Either[1]
573      *
574      * }
575      * </pre>
576      * @param pub Publisher to construct an Either from
577      * @return Either constructed from the supplied Publisher
578      */
579     public static <T> LazyEither<Throwable, T> fromPublisher(final Publisher<T> pub) {
580         if(pub instanceof LazyEither)
581             return (LazyEither<Throwable,T>)pub;
582         CompletableEither<T, T> result = LazyEither.either();
583 
584         pub.subscribe(new Subscriber<T>() {
585             Subscription sub;
586             @Override
587             public void onSubscribe(Subscription s) {
588                 sub =s;
589                 s.request(1l);
590             }
591 
592             @Override
593             public void onNext(T t) {
594                 result.complete(t);
595 
596             }
597 
598             @Override
599             public void onError(Throwable t) {
600                 result.completeExceptionally(t);
601             }
602 
603             @Override
604             public void onComplete() {
605                 if(!result.isDone())  {
606                    result.completeExceptionally(new NoSuchElementException());
607                 }
608             }
609         });
610         return result;

638     /**
639      * Create an instance of the left type. Most methods are biased to the right type,
640      * so you will need to use swap() or secondaryXXXX to manipulate the wrapped value
641      *
642      * <pre>
643      * {@code
644      *   Either.<Integer,Integer>left(10).map(i->i+1);
645      *   //Either.left[10]
646      *
647      *    Either.<Integer,Integer>left(10).swap().map(i->i+1);
648      *    //Either.right[11]
649      * }
650      * </pre>
651      *
652      *
653      * @param value to wrap
654      * @return Left instance of Either
