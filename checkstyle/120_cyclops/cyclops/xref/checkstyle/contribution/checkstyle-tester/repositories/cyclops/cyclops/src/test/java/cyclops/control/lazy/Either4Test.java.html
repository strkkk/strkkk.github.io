110     public LazyEither4<String,String,String,String> odd(LazyEither4<String,String,String,Integer> n) {

115     public LazyEither4<String,String,String,String> even(LazyEither4<String,String,String,Integer> n) {
116         return n.flatMap(x -> {
117             return x <= 0 ? LazyEither4.right("done") : odd(LazyEither4.right(x - 1));
