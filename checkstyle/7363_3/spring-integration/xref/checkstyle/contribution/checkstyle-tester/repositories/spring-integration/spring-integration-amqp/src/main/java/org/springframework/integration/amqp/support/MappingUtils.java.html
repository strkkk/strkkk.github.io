17  package org.springframework.integration.amqp.support;
18  
19  import org.springframework.amqp.core.MessageDeliveryMode;
20  import org.springframework.amqp.core.MessageProperties;
21  import org.springframework.amqp.support.AmqpHeaders;
22  import org.springframework.amqp.support.converter.ContentTypeDelegatingMessageConverter;
23  import org.springframework.amqp.support.converter.MessageConverter;
24  import org.springframework.lang.Nullable;
25  import org.springframework.messaging.Message;
26  import org.springframework.messaging.MessageHeaders;
27  import org.springframework.util.MimeType;
28  
29  /**
30   * Utility methods used during message mapping.
31   *
32   * @author Gary Russell
33   * @author Artem Bilan
34   *
35   * @since 4.3
36   *
37   */
38  public final class MappingUtils {
39  
40  	private MappingUtils() {
41  		super();
42  	}
43  
44  	/**
45  	 * Map an o.s.m.Message to an o.s.a.core.Message. When using a
46  	 * {@link ContentTypeDelegatingMessageConverter}, {@link AmqpHeaders#CONTENT_TYPE} and
47  	 * {@link MessageHeaders#CONTENT_TYPE} will be used for the selection, with the AMQP
48  	 * header taking precedence.
49  	 * @param requestMessage the request message.
50  	 * @param converter the message converter to use.
51  	 * @param headerMapper the header mapper to use.
52  	 * @param defaultDeliveryMode the default delivery mode.
53  	 * @param headersMappedLast true if headers are mapped after conversion.
54  	 * @return the mapped Message.
55  	 */
56  	public static org.springframework.amqp.core.Message mapMessage(Message<?> requestMessage,
57  			MessageConverter converter, AmqpHeaderMapper headerMapper, MessageDeliveryMode defaultDeliveryMode,
58  			boolean headersMappedLast) {
59  
60  		return doMapMessage(requestMessage, converter, headerMapper, defaultDeliveryMode, headersMappedLast, false);
61  	}
62  
63  	/**
64  	 * Map a reply o.s.m.Message to an o.s.a.core.Message. When using a
65  	 * {@link ContentTypeDelegatingMessageConverter}, {@link AmqpHeaders#CONTENT_TYPE} and
66  	 * {@link MessageHeaders#CONTENT_TYPE} will be used for the selection, with the AMQP
67  	 * header taking precedence.
68  	 * @param replyMessage the reply message.
69  	 * @param converter the message converter to use.
70  	 * @param headerMapper the header mapper to use.
71  	 * @param defaultDeliveryMode the default delivery mode.
72  	 * @param headersMappedLast true if headers are mapped after conversion.
73  	 * @return the mapped Message.
74  	 * @since 5.1.9
75  	 */
76  	public static org.springframework.amqp.core.Message mapReplyMessage(Message<?> replyMessage,
77  			MessageConverter converter, AmqpHeaderMapper headerMapper, MessageDeliveryMode defaultDeliveryMode,
78  			boolean headersMappedLast) {
79  
80  		return doMapMessage(replyMessage, converter, headerMapper, defaultDeliveryMode, headersMappedLast, true);
81  	}
82  
83  	private static org.springframework.amqp.core.Message doMapMessage(Message<?> message,
84  			MessageConverter converter, AmqpHeaderMapper headerMapper, MessageDeliveryMode defaultDeliveryMode,
85  			boolean headersMappedLast, boolean reply) {
86  
87  		MessageProperties amqpMessageProperties = new MessageProperties();
88  		org.springframework.amqp.core.Message amqpMessage;
89  		if (!headersMappedLast) {
90  			mapHeaders(message.getHeaders(), amqpMessageProperties, headerMapper, reply);
91  		}
92  		if (converter instanceof ContentTypeDelegatingMessageConverter && headersMappedLast) {
93  			String contentType = contentTypeAsString(message.getHeaders());
94  			if (contentType != null) {
95  				amqpMessageProperties.setContentType(contentType);
96  			}
97  		}
98  		amqpMessage = converter.toMessage(message.getPayload(), amqpMessageProperties);
99  		if (headersMappedLast) {
100 			mapHeaders(message.getHeaders(), amqpMessageProperties, headerMapper, reply);
101 		}
102 		checkDeliveryMode(message, amqpMessageProperties, defaultDeliveryMode);
103 		return amqpMessage;
104 	}
105 
106 	private static void mapHeaders(MessageHeaders messageHeaders, MessageProperties amqpMessageProperties,
107 			AmqpHeaderMapper headerMapper, boolean reply) {
108 
109 		if (reply) {
110 			headerMapper.fromHeadersToReply(messageHeaders, amqpMessageProperties);
111 		}
112 		else {
113 			headerMapper.fromHeadersToRequest(messageHeaders, amqpMessageProperties);
114 		}
115 	}
116 
117 	private static String contentTypeAsString(MessageHeaders headers) {
118 		Object contentType = headers.get(AmqpHeaders.CONTENT_TYPE);
119 		if (contentType instanceof MimeType) {
120 			contentType = contentType.toString();
121 		}
122 		if (contentType instanceof String) {
123 			return (String) contentType;
124 		}
125 		else if (contentType != null) {
126 			throw new IllegalArgumentException(AmqpHeaders.CONTENT_TYPE
127 					+ " header must be a MimeType or String, found: " + contentType.getClass().getName());
128 		}
129 		return null;
130 	}
131 
132 	/**
133 	 * Check the delivery mode and update with the default if not already present.
134 	 * @param requestMessage the request message.
135 	 * @param messageProperties the mapped message properties.
136 	 * @param defaultDeliveryMode the default delivery mode.
137 	 */
138 	public static void checkDeliveryMode(Message<?> requestMessage, MessageProperties messageProperties,
139 			@Nullable MessageDeliveryMode defaultDeliveryMode) {
140 		if (defaultDeliveryMode != null &&
141 				requestMessage.getHeaders().get(AmqpHeaders.DELIVERY_MODE) == null) {
142 			messageProperties.setDeliveryMode(defaultDeliveryMode);
143 		}
144 	}
