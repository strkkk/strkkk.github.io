17  package org.springframework.integration.scripting.jsr223;
18  
19  import javax.script.ScriptEngine;
20  import javax.script.ScriptEngineManager;
21  
22  import org.springframework.integration.scripting.ScriptExecutor;
23  import org.springframework.util.Assert;
24  
25  /**
26   * The scripting configuration utilities.
27   *
28   * @author David Turanski
29   * @author Artem Bilan
30   *
31   * @since 2.1
32   */
33  public final class ScriptExecutorFactory {
34  
35  	public static ScriptExecutor getScriptExecutor(String language) {
36  		if (language.equalsIgnoreCase("python") || language.equalsIgnoreCase("jython")) {
37  			return new PythonScriptExecutor();
38  		}
39  		else if (language.equalsIgnoreCase("ruby") ||  language.equalsIgnoreCase("jruby")) {
40  			return new RubyScriptExecutor();
41  		}
42  		else if (language.equalsIgnoreCase("kotlin")) {
43  			return new KotlinScriptExecutor();
44  		}
45  		return new DefaultScriptExecutor(language);
46  	}
47  
48  	/**
49  	 * Derive a scripting language from the provided script file name.
50  	 * @param scriptLocation the script file to consult for extension.
51  	 * @return the language name for the {@link ScriptExecutor}.
52  	 * @since 5.2
53  	 */
54  	public static String deriveLanguageFromFileExtension(String scriptLocation) {
55  		int index = scriptLocation.lastIndexOf('.') + 1;
56  		Assert.state(index > 0, () -> "Unable to determine language for script '" + scriptLocation + "'");
57  		String extension = scriptLocation.substring(index);
58  		if (extension.equals("kts")) {
59  			return "kotlin";
60  		}
61  		ScriptEngineManager engineManager = new ScriptEngineManager();
62  		ScriptEngine engine = engineManager.getEngineByExtension(extension);
63  		Assert.state(engine != null, () -> "No suitable scripting engine found for extension '" + extension + "'");
64  		return engine.getFactory().getLanguageName();
65  	}
66  
67  	private ScriptExecutorFactory() {
68  		super();
69  	}
70  
71  }
