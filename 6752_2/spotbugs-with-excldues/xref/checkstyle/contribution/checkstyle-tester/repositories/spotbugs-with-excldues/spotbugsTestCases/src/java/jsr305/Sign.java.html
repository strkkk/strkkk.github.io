1   package jsr305;
2   
3   import javax.annotation.meta.Exhaustive;
4   import javax.annotation.meta.TypeQualifier;
5   import javax.annotation.meta.TypeQualifierNickname;
6   import javax.annotation.meta.When;
7   
8   public @TypeQualifier
9   @interface Sign {
10      @Exhaustive
11      NumericSign value();
12  
13      When when() default When.ALWAYS;
14  
15      enum NumericSign {
16          NEGATIVE, ZERO, POSITIVE
17      };
18  
19      // Define more signess type qualifiers than we probably need
20      @Sign(NumericSign.POSITIVE)
21      @TypeQualifierNickname
22      @interface Positive {
23      }
24  
25      @Sign(NumericSign.NEGATIVE)
26      @TypeQualifierNickname
27      @interface Negative {
28      }
29  
30      @Sign(value = NumericSign.NEGATIVE, when = When.NEVER)
31      @TypeQualifierNickname
32      @interface Nonnegative {
33      }
34  
35      @Sign(value = NumericSign.NEGATIVE, when = When.MAYBE)
36      @TypeQualifierNickname
37      @interface MaybeNegative {
38      }
39  
40      @Sign(value = NumericSign.ZERO, when = When.NEVER)
41      @TypeQualifierNickname
42      @interface Nonzero {
43      }
44  
45      @Sign(value = NumericSign.POSITIVE, when = When.MAYBE)
46      @MaybeNegative
47      @TypeQualifierNickname
48      @interface Signed {
49      }
50  
51  }
