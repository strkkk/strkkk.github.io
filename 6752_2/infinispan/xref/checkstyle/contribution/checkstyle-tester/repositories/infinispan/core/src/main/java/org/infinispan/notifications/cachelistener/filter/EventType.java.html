1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.notifications.cachelistener.event.Event;
4   
5   /**
6    * Enum that provides information to allow for an event to know which type and if this event was generated due to a
7    * retry usually caused by a topology change while replicating.
8    *
9    * @author wburns
10   * @since 7.0
11   */
12  public class EventType {
13     private final Event.Type type;
14     private final boolean retried;
15     private final boolean pre;
16  
17  
18     public EventType(boolean retried, boolean pre, Event.Type type) {
19        this.retried = retried;
20        this.pre = pre;
21        this.type = type;
22     }
23  
24     public boolean isPreEvent() { return pre; };
25  
26     public boolean isRetry() {
27        return retried;
28     };
29  
30     public Event.Type getType() {
31        return type;
32     }
33  
34     public boolean isCreate() {
35        return type == Event.Type.CACHE_ENTRY_CREATED;
36     }
37  
38     public boolean isModified() {
39        return type == Event.Type.CACHE_ENTRY_MODIFIED;
40     }
41  
42     public boolean isRemove() {
43        return type == Event.Type.CACHE_ENTRY_REMOVED;
44     }
45  }
