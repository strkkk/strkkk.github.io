19  package edu.umd.cs.findbugs.detect;
20  
21  import org.apache.bcel.Const;
22  import org.apache.bcel.classfile.Method;
23  
24  import edu.umd.cs.findbugs.BugReporter;
25  import edu.umd.cs.findbugs.BytecodeScanningDetector;
26  import edu.umd.cs.findbugs.NonReportingDetector;
27  import edu.umd.cs.findbugs.ba.ClassContext;
28  import edu.umd.cs.findbugs.ba.XFactory;
29  import edu.umd.cs.findbugs.ba.XMethod;
30  import edu.umd.cs.findbugs.ba.ch.InterproceduralCallGraph;
31  import edu.umd.cs.findbugs.ba.ch.InterproceduralCallGraphVertex;
32  import edu.umd.cs.findbugs.ba.jsr305.Analysis;
33  import edu.umd.cs.findbugs.classfile.Global;
34  import edu.umd.cs.findbugs.classfile.MethodDescriptor;
35  
36  /**
37   * Build the interprocedural call graph.
38   *
39   * NOTE: at the present time, this facility is only used to find relevant type
40   * qualifiers. It could become a more general-purpose facility if there were a
41   * need.
42   *
43   * @author David Hovemeyer
44   */
45  public class BuildInterproceduralCallGraph extends BytecodeScanningDetector implements NonReportingDetector {
46  
47      private final InterproceduralCallGraph callGraph;
48  
49      private InterproceduralCallGraphVertex currentVertex;
50  
51      /**
52       * Constructor.
53       *
54       * @param bugReporter
55       *            the BugReporter to use
56       */
57      public BuildInterproceduralCallGraph(BugReporter bugReporter) {
58          if (!Analysis.FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
59              return;
60          }
61          callGraph = new InterproceduralCallGraph();
62      }
63  
64      @Override
65      public void visitClassContext(ClassContext classContext) {
66          if (!Analysis.FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
67              return;
68          }
69          super.visitClassContext(classContext);
70      }
71  
72      @Override
73      public void visitMethod(Method obj) {
74          currentVertex = findVertex(getXMethod());
75          super.visitMethod(obj);
76      }
77  
78      @Override
79      public void sawOpcode(int seen) {
80          switch (seen) {
81          case Const.INVOKESTATIC:
82          case Const.INVOKEVIRTUAL:
83          case Const.INVOKEINTERFACE:
84          case Const.INVOKESPECIAL:
85              MethodDescriptor called = getMethodDescriptorOperand();
86              XMethod calledXMethod = XFactory.createXMethod(called);
87              InterproceduralCallGraphVertex calledVertex = findVertex(calledXMethod);
88              callGraph.createEdge(currentVertex, calledVertex);
89              break;
90          default:
91              break;
92          }
93      }
94  
95      /**
96       * Find the InterproceduralCallGraphVertex for given XMethod.
97       *
98       * @param xmethod
99       *            an XMethod
100      * @return the XMethod's InterproceduralCallGraphVertex
101      */
102     private InterproceduralCallGraphVertex findVertex(XMethod xmethod) {
103         InterproceduralCallGraphVertex vertex;
104         vertex = callGraph.lookupVertex(xmethod.getMethodDescriptor());
105         if (vertex == null) {
106             vertex = new InterproceduralCallGraphVertex();
107             vertex.setXmethod(xmethod);
108             callGraph.addVertex(vertex);
109         }
110         return vertex;
111     }
112 
113     @Override
114     public void report() {
115         if (!Analysis.FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
116             return;
117         }
118         Global.getAnalysisCache().eagerlyPutDatabase(InterproceduralCallGraph.class, callGraph);
119     }
120 }
