84        while (reader.hasNext() && (reader.nextTag() != XMLStreamConstants.END_ELEMENT)) {
85           Element element = Element.forName(reader.getLocalName());

96                 parseThreads(reader, holder);
97                 break;
98              }
99              default: {
100                throw ParseUtils.unexpectedElement(reader);
101             }
102          }
103       }

132    private void parseAdvancedExternalizers(final XMLExtendedStreamReader reader, final ConfigurationBuilderHolder holder)
133          throws XMLStreamException {
134       GlobalConfigurationBuilder builder = holder.getGlobalConfigurationBuilder();
135       while (reader.hasNext() && (reader.nextTag() != XMLStreamConstants.END_ELEMENT)) {
136          Element element = Element.forName(reader.getLocalName());
137          switch (element) {
138             case ADVANCED_EXTERNALIZER: {
139                int attributes = reader.getAttributeCount();
140                AdvancedExternalizer<?> advancedExternalizer = null;
141                Integer id = null;
142                ParseUtils.requireAttributes(reader, Attribute.CLASS.getLocalName());
143                for (int i = 0; i < attributes; i++) {
144                   String value = replaceProperties(reader.getAttributeValue(i));
145                   Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
146                   switch (attribute) {
147                      case CLASS: {
148                         advancedExternalizer = Util.getInstance(value, holder.getClassLoader());
149                         break;
150                      }
151                      case ID: {
152                         id = Integer.valueOf(value);
153                         break;
154                      }
155                      default: {
156                         throw ParseUtils.unexpectedAttribute(reader, i);
157                      }
158                   }
159                }
160 
161                ParseUtils.requireNoContent(reader);
162 
163 
164                if (id != null) {
165                   builder.serialization().addAdvancedExternalizer(id, advancedExternalizer);
166                } else {
167                   builder.serialization().addAdvancedExternalizer(advancedExternalizer);
168                }
169                break;
170             }
171             default: {
172                throw ParseUtils.unexpectedElement(reader);
173             }
174          }
175       }

178    private void parseThreads(XMLExtendedStreamReader reader, ConfigurationBuilderHolder holder) throws XMLStreamException {
179       while (reader.hasNext() && (reader.nextTag() != XMLStreamConstants.END_ELEMENT)) {
180          Element element = Element.forName(reader.getLocalName());
181          switch (element) {
182             case THREAD_FACTORY: {
183                parseThreadFactory(reader);
184                break;
185             }
186             case CACHED_THREAD_POOL: {
187                parseCachedThreadPool(reader, holder);
188                break;
189             }
190             case SCHEDULED_THREAD_POOL: {
191                parseScheduledThreadPool(reader, holder);
192                break;
193             }
194             case BLOCKING_BOUNDED_QUEUE_THREAD_POOL: {
195                parseBlockingBoundedQueueThreadPool(reader, holder);
196                break;
197             }
198             default: {
199                throw ParseUtils.unexpectedElement(reader);
200             }
201          }
202       }
203 
204       // Link up thread factories with the thread pools that have referenced them
205       for (Map.Entry<String, ThreadPoolConfigurationBuilder> entry : threadPools.entrySet()) {
206          String threadFactoryName = threadPoolToThreadFactory.get(entry.getKey());
207          if (threadFactoryName != null) {
208             ThreadFactory threadFactory = threadFactories.get(threadFactoryName);
209             entry.getValue().threadFactory(threadFactory);
210          }
211       }
212    }

582    private void parseGlobalAuthorization(XMLExtendedStreamReader reader, ConfigurationBuilderHolder holder) throws XMLStreamException {
583       GlobalAuthorizationConfigurationBuilder builder = holder.getGlobalConfigurationBuilder().security().authorization().enable();
584       for (int i = 0; i < reader.getAttributeCount(); i++) {
585          String value = replaceProperties(reader.getAttributeValue(i));
586          Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
587          switch (attribute) {
588             case AUDIT_LOGGER: {
589                builder.auditLogger(Util.<AuditLogger>getInstance(value, holder.getClassLoader()));
590                break;
591             }
592             default: {
593                throw ParseUtils.unexpectedAttribute(reader, i);
594             }
595          }
596       }
597       PrincipalRoleMapper roleMapper = null;
598       while (reader.hasNext() && (reader.nextTag() != XMLStreamConstants.END_ELEMENT)) {
599          Element element = Element.forName(reader.getLocalName());
600          switch (element) {
601             case IDENTITY_ROLE_MAPPER:
602                if (roleMapper != null) {
603                   throw ParseUtils.unexpectedElement(reader);
604                }
605                ParseUtils.requireNoAttributes(reader);
606                ParseUtils.requireNoContent(reader);
607                roleMapper = new IdentityRoleMapper();
608                break;
609             case COMMON_NAME_ROLE_MAPPER:
610                if (roleMapper != null) {
611                   throw ParseUtils.unexpectedElement(reader);
612                }
613                ParseUtils.requireNoAttributes(reader);
614                ParseUtils.requireNoContent(reader);
615                roleMapper = new CommonNameRoleMapper();
616                break;
617             case CLUSTER_ROLE_MAPPER:
618                if (roleMapper != null) {
619                   throw ParseUtils.unexpectedElement(reader);
620                }
621                ParseUtils.requireNoAttributes(reader);
622                ParseUtils.requireNoContent(reader);
623                roleMapper = new ClusterRoleMapper();
624                break;
625             case CUSTOM_ROLE_MAPPER:
626                if (roleMapper != null) {
627                   throw ParseUtils.unexpectedElement(reader);
628                }
629                roleMapper = parseCustomMapper(reader, holder);
630                break;
631             case ROLE: {
632                parseGlobalRole(reader, builder);
633                break;
634             }
635             default: {
636                throw ParseUtils.unexpectedElement(reader);
637             }
638          }
639       }
640       if (roleMapper != null) {
641          builder.principalRoleMapper(roleMapper);
642       }
643    }

711       GlobalConfigurationBuilder globalBuilder = holder.getGlobalConfigurationBuilder();
712       if (holder.getGlobalConfigurationBuilder().transport().getTransport() == null) {
713          holder.getGlobalConfigurationBuilder().transport().defaultTransport();
714       }
715       TransportConfigurationBuilder transport = holder.getGlobalConfigurationBuilder().transport();
716       for (int i = 0; i < reader.getAttributeCount(); i++) {
717          String value = replaceProperties(reader.getAttributeValue(i));
718          Attribute attribute = Attribute.forName(reader.getAttributeLocalName(i));
719          switch (attribute) {
720             case STACK: {
721                boolean stackFound = transport.getProperty("stack-" + value) != null;
722                if (stackFound) {
723                   String filePath = transport.getProperty("stackFilePath-" + value);
724                   transport.addProperty("stack", value);
725                   transport.addProperty(JGroupsTransport.CONFIGURATION_FILE, filePath);
726                }
727                break;
728             }
729             case CLUSTER: {
730                globalBuilder.transport().clusterName(value);
731                break;
732             }
733             case EXECUTOR: {
734                globalBuilder.transport().transportThreadPool().read(
735                      createThreadPoolConfiguration(value, ASYNC_TRANSPORT_EXECUTOR));
736                break;
737             }
738             case TOTAL_ORDER_EXECUTOR: {
739                globalBuilder.transport().totalOrderThreadPool().read(
740                      createThreadPoolConfiguration(value, TOTAL_ORDER_EXECUTOR));
741                break;
742             }
743             case REMOTE_COMMAND_EXECUTOR: {
744                globalBuilder.transport().remoteCommandThreadPool().read(
745                      createThreadPoolConfiguration(value, REMOTE_COMMAND_EXECUTOR));
746                break;
747             }
748             case LOCK_TIMEOUT: {
749                globalBuilder.transport().distributedSyncTimeout(Long.valueOf(value));
750                break;
751             }
752             case NODE_NAME: {
753                globalBuilder.transport().nodeName(value);
754                for (DefaultThreadFactory threadFactory : threadFactories.values())
755                   threadFactory.setNode(value);
756 
757                break;
758             }
759             case MACHINE_ID: {
760                globalBuilder.transport().machineId(value);
761                break;
762             }
763             case RACK_ID: {
764                globalBuilder.transport().rackId(value);
765                break;
766             }
767             case SITE: {
768                globalBuilder.transport().siteId(value);
769                globalBuilder.site().localSite(value);
770                break;
771             }
772             default: {
773                throw ParseUtils.unexpectedAttribute(reader, i);
774             }
775          }
776       }
777       ParseUtils.requireNoContent(reader);
778    }
