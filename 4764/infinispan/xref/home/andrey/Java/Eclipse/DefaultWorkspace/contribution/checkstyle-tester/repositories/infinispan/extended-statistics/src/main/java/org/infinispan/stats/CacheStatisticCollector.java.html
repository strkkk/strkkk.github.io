22  /**
23   * Collects and maintains all the statistics for a cache.
24   *
25   * @author Roberto Palmieri
26   * @author Sebastiano Peluso
27   * @author Diego Didona
28   * @author Pedro Ruivo

31  public class CacheStatisticCollector {
32     private final static Log log = LogFactory.getLog(CacheStatisticCollector.class, Log.class);
33     private final TimeService timeService;
34     private final ConcurrentGlobalContainer globalContainer;
35     private volatile EnumMap<PercentileStatistic, ReservoirSampler> percentiles;
36  
37     public CacheStatisticCollector(TimeService timeService) {
38        this.timeService = timeService;
39        this.globalContainer = new ConcurrentGlobalContainer(timeService);
40        reset();
41     }
42  
43     /**
44      * reset all the statistics collected so far.
45      */
46     public final void reset() {
47        if (log.isTraceEnabled()) {
48           log.tracef("Resetting Node Scope Statistics");
49        }
50        globalContainer.reset();
51        percentiles = new EnumMap<PercentileStatistic, ReservoirSampler>(PercentileStatistic.class);
52        for (PercentileStatistic percentileStatistic : PercentileStatistic.values()) {
53           percentiles.put(percentileStatistic, new ReservoirSampler());
54        }
55     }
56  
57     /**
58      * Merges a transaction statistics in this cache statistics.
59      */
60     public final void merge(TransactionStatistics transactionStatistics) {
61        if (log.isTraceEnabled()) {
62           log.tracef("Merge transaction statistics %s to the node statistics", transactionStatistics);
63        }
64        ReservoirSampler reservoirSampler;
65        ExtendedStatistic percentileSample;
66        if (transactionStatistics.isLocalTransaction()) {
67           if (transactionStatistics.isReadOnly()) {
68              reservoirSampler = percentiles.get(RO_LOCAL_EXECUTION);
69              percentileSample = transactionStatistics.isCommitted() ? RO_TX_SUCCESSFUL_EXECUTION_TIME :
70                    RO_TX_ABORTED_EXECUTION_TIME;
71           } else {
72              reservoirSampler = percentiles.get(WR_LOCAL_EXECUTION);
73              percentileSample = transactionStatistics.isCommitted() ? WR_TX_SUCCESSFUL_EXECUTION_TIME :
74                    WR_TX_ABORTED_EXECUTION_TIME;
75           }
76        } else {
77           if (transactionStatistics.isReadOnly()) {
78              reservoirSampler = percentiles.get(RO_REMOTE_EXECUTION);
79              percentileSample = transactionStatistics.isCommitted() ? RO_TX_SUCCESSFUL_EXECUTION_TIME :
80                    RO_TX_ABORTED_EXECUTION_TIME;
81           } else {
82              reservoirSampler = percentiles.get(WR_REMOTE_EXECUTION);
83              percentileSample = transactionStatistics.isCommitted() ? WR_TX_SUCCESSFUL_EXECUTION_TIME :
84                    WR_TX_ABORTED_EXECUTION_TIME;
85           }
86        }
87        doMerge(transactionStatistics, reservoirSampler, percentileSample);
88     }

90     /**
91      * Adds a value to a local statistic. This value is not associated with any transaction.
92      */
93     public final void addLocalValue(ExtendedStatistic stat, double value) {
94        globalContainer.add(stat, value, true);
95     }

97     /**
98      * Adds a value to a remote statistic. This value is not associated with any transaction.
99      */
100    public final void addRemoteValue(ExtendedStatistic stat, double value) {
101       globalContainer.add(stat, value, false);
102    }

104    /**
105     * @return the percentile og the statistic.
106     * @throws IllegalArgumentException if the percentile request is not in the correct bounds (]0,100[)
107     */
108    public final double getPercentile(PercentileStatistic stat, int percentile)
109          throws IllegalArgumentException {
110       if (log.isTraceEnabled()) {
111          log.tracef("Get percentile %s from %s", percentile, stat);
112       }
113       return percentiles.get(stat).getKPercentile(percentile);
114    }

116    /**
117     * @return the current value of the statistic. If the statistic is not exported (via JMX), then the sum of the remote
118     *         and local value is returned.
119     * @throws ExtendedStatisticNotFoundException
120     *          if the statistic is not found.
121     */
122    public final double getAttribute(ExtendedStatistic stat) throws ExtendedStatisticNotFoundException {
123       final StatisticsSnapshot snapshot = globalContainer.getSnapshot();
124       double value = 0;
125       switch (stat) {
126          case LOCAL_EXEC_NO_CONT:
127             value = microAverageLocal(snapshot, LOCAL_EXEC_NO_CONT, NUM_COMMITTED_WR_TX);
128             break;
129          case LOCK_HOLD_TIME:
130             value = microAverageLocalAndRemote(snapshot, LOCK_HOLD_TIME, NUM_HELD_LOCKS);
131             break;
132          case SYNC_PREPARE_TIME:
133             value = microAverageLocal(snapshot, SYNC_PREPARE_TIME, NUM_SYNC_PREPARE);
134             break;
135          case SYNC_COMMIT_TIME:
136             value = microAverageLocal(snapshot, SYNC_COMMIT_TIME, NUM_SYNC_COMMIT);
137             break;
138          case SYNC_ROLLBACK_TIME:
139             value = microAverageLocal(snapshot, SYNC_ROLLBACK_TIME, NUM_SYNC_ROLLBACK);
140             break;
141          case SYNC_GET_TIME:
142             value = microAverageLocal(snapshot, SYNC_GET_TIME, NUM_SYNC_GET);
143             break;
144          case ASYNC_COMMIT_TIME:
145             value = microAverageLocal(snapshot, ASYNC_COMMIT_TIME, NUM_ASYNC_COMMIT);
146             break;
147          case ASYNC_COMPLETE_NOTIFY_TIME:
148             value = microAverageLocal(snapshot, ASYNC_COMPLETE_NOTIFY_TIME, NUM_ASYNC_COMPLETE_NOTIFY);
149             break;
150          case ASYNC_PREPARE_TIME:
151             value = microAverageLocal(snapshot, ASYNC_PREPARE_TIME, NUM_ASYNC_PREPARE);
152             break;
153          case ASYNC_ROLLBACK_TIME:
154             value = microAverageLocal(snapshot, ASYNC_ROLLBACK_TIME, NUM_ASYNC_ROLLBACK);
155             break;
156          case NUM_NODES_COMMIT:
157             value = averageLocal(snapshot, NUM_NODES_COMMIT, NUM_SYNC_COMMIT, NUM_ASYNC_COMMIT);
158             break;
159          case NUM_NODES_GET:
160             value = averageLocal(snapshot, NUM_NODES_GET, NUM_SYNC_GET);
161             break;
162          case NUM_NODES_PREPARE:
163             value = averageLocal(snapshot, NUM_NODES_PREPARE, NUM_SYNC_PREPARE, NUM_ASYNC_PREPARE);
164             break;
165          case NUM_NODES_ROLLBACK:
166             value = averageLocal(snapshot, NUM_NODES_ROLLBACK, NUM_SYNC_ROLLBACK, NUM_ASYNC_ROLLBACK);
167             break;
168          case NUM_NODES_COMPLETE_NOTIFY:
169             value = averageLocal(snapshot, NUM_NODES_COMPLETE_NOTIFY, NUM_ASYNC_COMPLETE_NOTIFY);
170             break;
171          case COMMIT_EXECUTION_TIME:
172             value = convertNanosToMicro(averageLocal(snapshot, COMMIT_EXECUTION_TIME, NUM_COMMITTED_WR_TX,
173                                                      NUM_COMMITTED_RO_TX));
174             break;
175          case ROLLBACK_EXECUTION_TIME:
176             value = microAverageLocal(snapshot, ROLLBACK_EXECUTION_TIME, NUM_ROLLBACK_COMMAND);
177             break;
178          case LOCK_WAITING_TIME:
179             value = microAverageLocalAndRemote(snapshot, LOCK_WAITING_TIME, NUM_WAITED_FOR_LOCKS);
180             break;
181          case WRITE_TX_PERCENTAGE: {     //computed on the locally born txs
182             double readTx = snapshot.getLocal(NUM_COMMITTED_RO_TX) +
183                   snapshot.getLocal(NUM_ABORTED_RO_TX);
184             double writeTx = snapshot.getLocal(NUM_COMMITTED_WR_TX) +
185                   snapshot.getLocal(NUM_ABORTED_WR_TX);
186             double total = readTx + writeTx;
187             if (total != 0) {
188                value = writeTx / total;
189             }
190             break;
191          }
192          case SUCCESSFUL_WRITE_TX_PERCENTAGE: { //computed on the locally born txs
193             double readSuxTx = snapshot.getLocal(NUM_COMMITTED_RO_TX);
194             double writeSuxTx = snapshot.getLocal(NUM_COMMITTED_WR_TX);
195             double total = readSuxTx + writeSuxTx;
196             if (total != 0) {
197                value = writeSuxTx / total;
198             }
199             break;
200          }
201          case NUM_GETS_RO_TX:
202             value = averageLocal(snapshot, NUM_GETS_RO_TX, NUM_COMMITTED_RO_TX);
203             break;
204          case NUM_GETS_WR_TX:
205             value = averageLocal(snapshot, NUM_GETS_WR_TX, NUM_COMMITTED_WR_TX);
206             break;
207          case NUM_REMOTE_GETS_RO_TX:
208             value = averageLocal(snapshot, NUM_REMOTE_GETS_RO_TX, NUM_COMMITTED_RO_TX);
209             break;
210          case NUM_REMOTE_GETS_WR_TX:
211             value = averageLocal(snapshot, NUM_REMOTE_GETS_WR_TX, NUM_COMMITTED_WR_TX);
212             break;
213          case REMOTE_GET_EXECUTION:
214             value = microAverageLocal(snapshot, NUM_REMOTE_GET, REMOTE_GET_EXECUTION);
215             break;
216          case NUM_PUTS_WR_TX:
217             value = averageLocal(snapshot, NUM_PUTS_WR_TX, NUM_COMMITTED_WR_TX);
218             break;
219          case NUM_REMOTE_PUTS_WR_TX:
220             value = averageLocal(snapshot, NUM_REMOTE_PUTS_WR_TX, NUM_COMMITTED_WR_TX);
221             break;
222          case REMOTE_PUT_EXECUTION:
223             value = microAverageLocal(snapshot, NUM_REMOTE_PUT, REMOTE_PUT_EXECUTION);
224             break;
225          case NUM_LOCK_FAILED_DEADLOCK:
226          case NUM_LOCK_FAILED_TIMEOUT:
227             value = snapshot.getLocal(stat);
228             break;
229          case WR_TX_SUCCESSFUL_EXECUTION_TIME:
230             value = microAverageLocal(snapshot, WR_TX_SUCCESSFUL_EXECUTION_TIME, NUM_COMMITTED_WR_TX);
231             break;
232          case WR_TX_ABORTED_EXECUTION_TIME:
233             value = microAverageLocal(snapshot, WR_TX_ABORTED_EXECUTION_TIME, NUM_ABORTED_WR_TX);
234             break;
235          case RO_TX_SUCCESSFUL_EXECUTION_TIME:
236             value = microAverageLocal(snapshot, RO_TX_SUCCESSFUL_EXECUTION_TIME, NUM_COMMITTED_RO_TX);
237             break;
238          case PREPARE_COMMAND_SIZE:
239             value = averageLocal(snapshot, PREPARE_COMMAND_SIZE, NUM_SYNC_PREPARE, NUM_ASYNC_PREPARE);
240             break;
241          case COMMIT_COMMAND_SIZE:
242             value = averageLocal(snapshot, COMMIT_COMMAND_SIZE, NUM_SYNC_COMMIT, NUM_ASYNC_COMMIT);
243             break;
244          case CLUSTERED_GET_COMMAND_SIZE:
245             value = averageLocal(snapshot, NUM_SYNC_GET, CLUSTERED_GET_COMMAND_SIZE);
246             break;
247          case NUM_LOCK_PER_LOCAL_TX:
248             value = averageLocal(snapshot, NUM_HELD_LOCKS, NUM_COMMITTED_WR_TX, NUM_ABORTED_WR_TX);
249             break;
250          case NUM_LOCK_PER_REMOTE_TX:
251             value = averageRemote(snapshot, NUM_HELD_LOCKS, NUM_COMMITTED_WR_TX, NUM_ABORTED_WR_TX);
252             break;
253          case NUM_HELD_LOCKS_SUCCESS_LOCAL_TX:
254             value = averageLocal(snapshot, NUM_HELD_LOCKS_SUCCESS_LOCAL_TX, NUM_COMMITTED_WR_TX);
255             break;
256          case LOCAL_ROLLBACK_EXECUTION_TIME:
257             value = microAverageLocal(snapshot, ROLLBACK_EXECUTION_TIME, NUM_ROLLBACK_COMMAND);
258             break;
259          case REMOTE_ROLLBACK_EXECUTION_TIME:
260             value = microAverageRemote(snapshot, ROLLBACK_EXECUTION_TIME, NUM_ROLLBACK_COMMAND);
261             break;
262          case LOCAL_COMMIT_EXECUTION_TIME:
263             value = microAverageLocal(snapshot, COMMIT_EXECUTION_TIME, NUM_COMMIT_COMMAND);
264             break;
265          case REMOTE_COMMIT_EXECUTION_TIME:
266             value = microAverageRemote(snapshot, COMMIT_EXECUTION_TIME, NUM_COMMIT_COMMAND);
267             break;
268          case LOCAL_PREPARE_EXECUTION_TIME:
269             value = microAverageLocal(snapshot, PREPARE_EXECUTION_TIME, NUM_PREPARE_COMMAND);
270             break;
271          case REMOTE_PREPARE_EXECUTION_TIME:
272             value = microAverageRemote(snapshot, PREPARE_EXECUTION_TIME, NUM_PREPARE_COMMAND);
273             break;
274          case ABORT_RATE:
275             double totalAbort = snapshot.getLocal(NUM_ABORTED_RO_TX) +
276                   snapshot.getLocal(NUM_ABORTED_WR_TX);
277             double totalCommitAndAbort = snapshot.getLocal(NUM_COMMITTED_RO_TX) +
278                   snapshot.getLocal(NUM_COMMITTED_WR_TX) + totalAbort;
279             if (totalCommitAndAbort != 0) {
280                value = totalAbort / totalCommitAndAbort;
281             }
282             break;
283          case ARRIVAL_RATE:
284             double localCommittedTx = snapshot.getLocal(NUM_COMMITTED_RO_TX) +
285                   snapshot.getLocal(NUM_COMMITTED_WR_TX);
286             double localAbortedTx = snapshot.getLocal(NUM_ABORTED_RO_TX) +
287                   snapshot.getLocal(NUM_ABORTED_WR_TX);
288             double remoteCommittedTx = snapshot.getRemote(NUM_COMMITTED_RO_TX) +
289                   snapshot.getRemote(NUM_COMMITTED_WR_TX);
290             double remoteAbortedTx = snapshot.getRemote(NUM_ABORTED_RO_TX) +
291                   snapshot.getRemote(NUM_ABORTED_WR_TX);
292             double totalBornTx = localAbortedTx + localCommittedTx + remoteAbortedTx + remoteCommittedTx;
293             value = totalBornTx / convertNanosToSeconds(timeService.timeDuration(snapshot.getLastResetTime(), NANOSECONDS));
294             break;
295          case THROUGHPUT:
296             double totalLocalBornTx = snapshot.getLocal(NUM_COMMITTED_RO_TX) +
297                   snapshot.getLocal(NUM_COMMITTED_WR_TX);
298             value = totalLocalBornTx / convertNanosToSeconds(timeService.timeDuration(snapshot.getLastResetTime(), NANOSECONDS));
299             break;
300          case LOCK_HOLD_TIME_LOCAL:
301             value = microAverageLocal(snapshot, LOCK_HOLD_TIME, NUM_HELD_LOCKS);
302             break;
303          case LOCK_HOLD_TIME_REMOTE:
304             value = microAverageRemote(snapshot, LOCK_HOLD_TIME, NUM_HELD_LOCKS);
305             break;
306          case NUM_COMMITTED_TX:
307             value = snapshot.getLocal(NUM_COMMITTED_RO_TX) + snapshot.getLocal(NUM_COMMITTED_WR_TX) +
308                   snapshot.getRemote(NUM_COMMITTED_RO_TX) + snapshot.getRemote(NUM_COMMITTED_WR_TX);
309             break;
310          case NUM_LOCAL_COMMITTED_TX:
311             value = snapshot.getLocal(NUM_COMMITTED_RO_TX) + snapshot.getLocal(NUM_COMMITTED_WR_TX);
312             break;
313          case WRITE_SKEW_PROBABILITY:
314             double totalTxs = snapshot.getLocal(NUM_COMMITTED_RO_TX) +
315                   snapshot.getLocal(NUM_COMMITTED_WR_TX) +
316                   snapshot.getLocal(NUM_ABORTED_RO_TX) +
317                   snapshot.getLocal(NUM_ABORTED_WR_TX);
318             if (totalTxs != 0) {
319                double writeSkew = snapshot.getLocal(NUM_WRITE_SKEW);
320                value = writeSkew / totalTxs;
321             }
322             break;
323          case NUM_GET:
324             value = snapshot.getLocal(NUM_GETS_WR_TX) +
325                   snapshot.getLocal(NUM_GETS_RO_TX);
326             break;
327          case NUM_REMOTE_GET:
328             value = snapshot.getLocal(NUM_REMOTE_GETS_WR_TX) +
329                   snapshot.getLocal(NUM_REMOTE_GETS_RO_TX);
330             break;
331          case NUM_PUT:
332             value = snapshot.getLocal(NUM_PUTS_WR_TX);
333             break;
334          case NUM_REMOTE_PUT:
335             value = snapshot.getLocal(NUM_REMOTE_PUTS_WR_TX);
336             break;
337          case LOCAL_GET_EXECUTION:
338             double num = snapshot.getLocal(NUM_GET);
339             if (num != 0) {
340                double local_get_time = snapshot.getLocal(ALL_GET_EXECUTION) -
341                      snapshot.getLocal(SYNC_GET_TIME);
342                value = convertNanosToMicro(local_get_time) / num;
343             }
344             break;
345          case RESPONSE_TIME:
346             double succWrTot = convertNanosToMicro(snapshot.getLocal(WR_TX_SUCCESSFUL_EXECUTION_TIME));
347             double abortWrTot = convertNanosToMicro(snapshot.getLocal(WR_TX_ABORTED_EXECUTION_TIME));
348             double succRdTot = convertNanosToMicro(snapshot.getLocal(RO_TX_SUCCESSFUL_EXECUTION_TIME));
349 
350             double numWr = snapshot.getLocal(NUM_COMMITTED_WR_TX);
351             double numRd = snapshot.getLocal(NUM_COMMITTED_RO_TX);
352 
353             if ((numWr + numRd) > 0) {
354                value = (succRdTot + succWrTot + abortWrTot) / (numWr + numRd);
355             }
356             break;
357          default:
358             if (log.isTraceEnabled()) {
359                log.tracef("Attribute %s is not exposed via JMX. Calculating raw value", stat);
360             }
361             if (stat.isLocal()) {
362                value += snapshot.getLocal(stat);
363             }
364             if (stat.isRemote()) {
365                value += snapshot.getRemote(stat);
366             }
367       }
368       if (log.isTraceEnabled()) {
369          log.tracef("Get attribute %s = %s", stat, value);
370       }
371       return value;
372    }

391    /**
392     * @return the conversion of nanoseconds to microseconds without losing precision.
393     */
394    public static double convertNanosToMicro(double nanos) {
395       return nanos / 1E3;
396    }

398    /**
399     * @return the conversion of nanoseconds to seconds without losing precision.

402       return nanos / 1E9;
403    }

405    private void doMerge(TransactionStatistics transactionStatistics,
406                         ReservoirSampler reservoirSampler, ExtendedStatistic percentileSample) {
407       transactionStatistics.flushTo(globalContainer);
408       try {
409          reservoirSampler.insertSample(transactionStatistics.getValue(percentileSample));
410       } catch (ExtendedStatisticNotFoundException e) {
411          log.extendedStatisticNotFoundForPercentile(percentileSample, e);
412       }
413    }

415    private double averageLocal(StatisticsSnapshot snapshot, ExtendedStatistic numeratorStat,
416                                ExtendedStatistic... denominatorStats) throws ExtendedStatisticNotFoundException {
417       double denominator = 0;
418       for (ExtendedStatistic denominatorStat : denominatorStats) {
419          denominator += snapshot.getLocal(denominatorStat);
420       }
421       if (denominator != 0) {
422          double numerator = snapshot.getLocal(numeratorStat);
423          return numerator / denominator;
424       }
425       return 0;
426    }
427 
428    private double averageRemote(StatisticsSnapshot snapshot, ExtendedStatistic numeratorStat,
429                                 ExtendedStatistic... denominatorStats) throws ExtendedStatisticNotFoundException {
430       double denominator = 0;
431       for (ExtendedStatistic denominatorStat : denominatorStats) {
432          denominator += snapshot.getRemote(denominatorStat);
433       }
434       if (denominator != 0) {
435          double numerator = snapshot.getRemote(numeratorStat);
436          return numerator / denominator;
437       }
438       return 0;
439    }
440 
441    private double averageLocalAndRemote(StatisticsSnapshot snapshot, ExtendedStatistic numeratorStat,
442                                         ExtendedStatistic... denominatorStats) throws ExtendedStatisticNotFoundException {
443       double denominator = 0;
444       for (ExtendedStatistic denominatorStat : denominatorStats) {
445          denominator += snapshot.getRemote(denominatorStat);
446          denominator += snapshot.getLocal(denominatorStat);
447       }
448       if (denominator != 0) {
449          double numerator = snapshot.getRemote(numeratorStat);
450          numerator += snapshot.getLocal(numeratorStat);
451          return numerator / denominator;
452       }
453       return 0;
454    }
455 
456    private double microAverageLocal(StatisticsSnapshot snapshot, ExtendedStatistic numerator,
457                                     ExtendedStatistic denominator) throws ExtendedStatisticNotFoundException {
458       return convertNanosToMicro(averageLocal(snapshot, numerator, denominator));
459    }
460 
461    private double microAverageRemote(StatisticsSnapshot snapshot, ExtendedStatistic numerator,
462                                      ExtendedStatistic denominator) throws ExtendedStatisticNotFoundException {
463       return convertNanosToMicro(averageRemote(snapshot, numerator, denominator));
464    }
465 
466    private double microAverageLocalAndRemote(StatisticsSnapshot snapshot, ExtendedStatistic numerator,
467                                              ExtendedStatistic denominator) throws ExtendedStatisticNotFoundException {
468       return convertNanosToMicro(averageLocalAndRemote(snapshot, numerator, denominator));
469    }
